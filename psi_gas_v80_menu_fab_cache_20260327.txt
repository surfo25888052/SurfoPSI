function doGet(e) {
  const p = (e && e.parameter) ? e.parameter : {};
  const type = String(p.type || "").trim();

  const callback = String(p.callback || "").trim();
  const jsonpOk = callback && /^[A-Za-z_$][0-9A-Za-z_$]*(\.[0-9A-Za-z_$]+)*$/.test(callback);

  const respond = (obj) => {
    const json = JSON.stringify(obj);
    const body = jsonpOk ? `${callback}(${json})` : json;
    const out = ContentService.createTextOutput(body);
    return out.setMimeType(jsonpOk ? ContentService.MimeType.JAVASCRIPT : ContentService.MimeType.JSON);
  };

  try {
    if (type === "products") return apiProducts_(respond);
    if (type === "manageProduct") return apiManageProduct_(p, respond);

    if (type === "members") return apiMembersLogin_(p, respond);

    // 客戶（電商）登入
    if (type === "customerLogin") return apiCustomerLogin_(p, respond);

    // 客戶（電商）註冊（寫入 customers）
    if (type === "register") return apiCustomerRegister_(p, respond);

    // 後台使用者（Members）管理
    if (type === "manageMember") return apiManageMember_(p, respond);

    // 前台（商城）下單/查單
    if (type === "order") return apiOrderCreate_(p, respond);
    if (type === "myOrders") return apiMyOrders_(p, respond);
    if (type === "getOrder") return apiGetOrder_(p, respond);

    if (type === "suppliers") return apiSuppliers_(respond);
    if (type === "manageSupplier") return apiManageSupplier_(p, respond);

    if (type === "customers") return apiCustomers_(respond);
    if (type === "manageCustomer") return apiManageCustomer_(p, respond);

    if (type === "purchases") return apiPurchases_(p, respond);
    if (type === "managePurchase") return apiManagePurchase_(p, respond);

    if (type === "pickups") return apiPickups_(respond);
    if (type === "managePickup") return apiManagePickup_(p, respond);

    if (type === "orders") return apiOrders_(respond);
    if (type === "manageOrder") return apiManageOrder_(p, respond);

    if (type === "manageSale") return apiManageSale_(p, respond);

    if (type === "stockLedger") return apiStockLedger_(respond);
    if (type === "productLedger") return apiProductLedger_(p, respond);
    if (type === "productMarketHistory") return apiProductMarketHistory_(p, respond);
    if (type === "stockAdjust") return apiStockAdjust_(p, respond);

    if (type === "syncReferencePrices") return apiSyncReferencePrices_(respond);
    if (type === "marketPriceBoard") return apiMarketPriceBoard_(respond);
    if (type === "weeklyMenu") return apiWeeklyMenu_(p, respond);
    if (type === "menuDashboard") return apiMenuDashboard_(p, respond);
    if (type === "menuMappingConfig") return apiMenuMappingConfig_(p, respond);
    if (type === "manageMenuMapping") return apiManageMenuMapping_(p, respond);

    if (type === "systemSettings") return apiSystemSettings_(respond);
    if (type === "manageSystemSettings") return apiManageSystemSettings_(p, respond);

    if (type === "lastUpdate") return apiLastUpdate_(respond);

    return respond({ status: "error", message: "Invalid Request" });
  } catch (err) {
    return respond({ status: "error", message: String(err && err.message ? err.message : err) });
  }
}

// ----------------- Helpers -----------------
function findSheetByTrimName_(ss, name){
  const target = String(name || "").trim();
  const sheets = ss.getSheets();
  for (let i = 0; i < sheets.length; i++){
    const n = String(sheets[i].getName() || "").trim();
    if (n === target) return sheets[i];
  }
  return null;
}

function getSheetByAnyName_(ss, names) {
  for (let i = 0; i < names.length; i++) {
    const sh = ss.getSheetByName(names[i]);
    if (sh) return sh;
  }
  for (let i = 0; i < names.length; i++) {
    const sh = ss.getSheetByName(String(names[i]).trim());
    if (sh) return sh;
  }
  return null;
}

function ensureSheet_(ss, name, headers) {
  let sh = ss.getSheetByName(name) || findSheetByTrimName_(ss, name);
  if (!sh) {
    sh = ss.insertSheet(name);
  } else {
    // 若是被空白或大小寫造成的名字差異，嘗試統一名稱
    if (sh.getName() !== name) {
      try { sh.setName(name); } catch(e) {}
    }
  }
  const lastCol = Math.max(sh.getLastColumn(), headers.length);
  const row1 = sh.getRange(1, 1, 1, lastCol).getValues()[0];
  const hasAny = row1.some(v => String(v || "").trim() !== "");
  if (!hasAny) {
    sh.getRange(1, 1, 1, headers.length).setValues([headers]);
  } else {
    // Fill missing header cells in order (do not overwrite existing)
    for (let i = 0; i < headers.length; i++) {
      if (!row1[i] || String(row1[i]).trim() === "") sh.getRange(1, i + 1).setValue(headers[i]);
    }
  }
  return sh;
}

function getHeaders_(sheet) {
  const lastCol = Math.max(sheet.getLastColumn() || 1, 1);
  return sheet.getRange(1, 1, 1, lastCol).getValues()[0].map(h => String(h || "").trim());
}


function normalizeShopEnabled_(v, defaultEnabled) {
  const raw = String(v == null ? "" : v).trim().toLowerCase();
  if (!raw) return defaultEnabled === false ? "0" : "1";
  return (["0","false","no","off","n","hide","hidden"].includes(raw)) ? "0" : "1";
}

function headerIndexMap_(headers) {
  const m = {};
  headers.forEach((h, i) => { if (h) m[h] = i; });
  return m;
}

function readObjects_(sheet) {
  const values = sheet.getDataRange().getValues();
  if (values.length <= 1) return [];
  const headers = values[0].map(h => String(h || "").trim());
  const out = [];
  for (let r = 1; r < values.length; r++) {
    const row = values[r];
    if (!row || row.every(v => v === "" || v === null)) continue;
    const obj = {};
    for (let c = 0; c < headers.length; c++) {
      const key = headers[c];
      if (!key) continue;
      obj[key] = row[c];
    }
    out.push(obj);
  }
  return out;
}

function readObjectsWithRowNo_(sheet) {
  const values = sheet.getDataRange().getValues();
  if (values.length <= 1) return [];
  const headers = values[0].map(h => String(h || "").trim());
  const out = [];
  for (let r = 1; r < values.length; r++) {
    const row = values[r];
    if (!row || row.every(v => v === "" || v === null)) continue;
    const obj = { __rowNo: r + 1 };
    for (let c = 0; c < headers.length; c++) {
      const key = headers[c];
      if (!key) continue;
      obj[key] = row[c];
    }
    out.push(obj);
  }
  return out;
}

function appendObject_(sheet, obj) {
  const headers = getHeaders_(sheet);
  const row = new Array(headers.length).fill("");
  headers.forEach((h, i) => {
    if (!h) return;
    if (Object.prototype.hasOwnProperty.call(obj, h)) row[i] = obj[h];
  });
  sheet.appendRow(row);
}

function appendObjectsBatch_(sheet, objs) {
  const rows = Array.isArray(objs) ? objs : [];
  if (!rows.length) return;
  const headers = getHeaders_(sheet);
  const values = rows.map(obj => headers.map(h => (Object.prototype.hasOwnProperty.call(obj || {}, h) ? obj[h] : "")));
  sheet.getRange(sheet.getLastRow() + 1, 1, values.length, headers.length).setValues(values);
}

function toROCDateString_(d) {
  if (!(d instanceof Date)) return String(d || "").trim();
  const y = d.getFullYear() - 1911;
  const m = ("0" + (d.getMonth() + 1)).slice(-2);
  const day = ("0" + d.getDate()).slice(-2);
  return `${y}.${m}.${day}`;
}

function toISODateLocal_(d){
  if (!d) return "";
  try{
    // Date object
    if (Object.prototype.toString.call(d) === "[object Date]" && !isNaN(d.getTime())) {
      return Utilities.formatDate(d, "Asia/Taipei", "yyyy-MM-dd");
    }
    const s = String(d).trim();
    if (!s) return "";
    // ISO with time (Z) -> parse then format local
    if (/^\d{4}-\d{2}-\d{2}T/.test(s)) {
      const dt = new Date(s);
      if (!isNaN(dt.getTime())) return Utilities.formatDate(dt, "Asia/Taipei", "yyyy-MM-dd");
      return s.slice(0,10);
    }
    // ISO date
    if (/^\d{4}-\d{2}-\d{2}/.test(s)) return s.slice(0,10);
    // ROC YYY.MM.DD / etc
    if (/^\d{2,3}[\.\/\-]\d{1,2}[\.\/\-]\d{1,2}/.test(s) && !/^\d{4}[\.\/\-]/.test(s)) {
      const base = s.split(/[\sT]/)[0];
      const parts = base.split(/[\.\/\-]/);
      const y = (parseInt(parts[0],10) + 1911);
      const m = ("0"+parts[1]).slice(-2);
      const dd = ("0"+parts[2]).slice(-2);
      return `${y}-${m}-${dd}`;
    }
    // Fallback Date parse
    const dt2 = new Date(s);
    if (!isNaN(dt2.getTime())) return Utilities.formatDate(dt2, "Asia/Taipei", "yyyy-MM-dd");
    return "";
  } catch(e){
    return "";
  }
}


function decimalPlacesText_(v){
  const s = String(v == null ? "" : v).replace(/,/g, "").trim();
  const m = s.match(/\.(\d+)$/);
  return m ? m[1].length : 0;
}

function cleanDecimalInput_(v, maxScale){
  maxScale = (maxScale === undefined) ? 6 : Number(maxScale || 0);
  const s = String(v == null ? "" : v).replace(/,/g, "").trim();
  if (!s) return 0;
  const n = Number(s);
  if (!isFinite(n)) return 0;
  const scale = Math.min(decimalPlacesText_(s), maxScale);
  return Number(n.toFixed(scale));
}

function mulDecimalInput_(a, b, maxScale){
  maxScale = (maxScale === undefined) ? 6 : Number(maxScale || 0);
  const na = Number(String(a == null ? "" : a).replace(/,/g, "").trim() || 0);
  const nb = Number(String(b == null ? "" : b).replace(/,/g, "").trim() || 0);
  if (!isFinite(na) || !isFinite(nb)) return 0;
  const scale = Math.min(decimalPlacesText_(a) + decimalPlacesText_(b), maxScale);
  return Number((na * nb).toFixed(scale));
}

function addDecimalInput_(a, b, maxScale){
  maxScale = (maxScale === undefined) ? 6 : Number(maxScale || 0);
  const na = Number(a || 0);
  const nb = Number(b || 0);
  if (!isFinite(na) || !isFinite(nb)) return 0;
  const scale = Math.min(Math.max(decimalPlacesText_(a), decimalPlacesText_(b)), maxScale);
  return Number((na + nb).toFixed(scale));
}

function calcRoundedSalePriceFromCost_(cost, markupPercent){
  const base = Number(cost || 0);
  const markup = Number(markupPercent || 0);
  if (!isFinite(base) || base <= 0) return "";
  if (!isFinite(markup)) return Math.round(base);
  return Math.round(base * (1 + markup / 100));
}

function normalizeDate_(d) {
  if (d instanceof Date) return toROCDateString_(d);
  return String(d || "").trim();
}

function compareDate_(d1, d2) {
  d1 = normalizeDate_(d1);
  d2 = normalizeDate_(d2);
  if (!d1 && !d2) return 0;
  if (!d1) return -1;
  if (!d2) return 1;
  const a = d1.split(".").map(Number);
  const b = d2.split(".").map(Number);
  if (a.length < 3 || b.length < 3 || a.some(isNaN) || b.some(isNaN)) {
    return d1 === d2 ? 0 : (d1 > d2 ? 1 : -1);
  }
  if (a[0] !== b[0]) return a[0] - b[0];
  if (a[1] !== b[1]) return a[1] - b[1];
  return a[2] - b[2];
}

function getLedgerSheet_(ss) {
  return getSheetByAnyName_(ss, ["stock_ledger", "stockLedger", "StockLedger", "inventoryLog", "InventoryLog", "lnventoryLog"]);
}

function ensureLedgerSheet_(ss) {
  // 以既有欄位為主（不重排、不覆蓋），必要時才補齊缺漏欄位
  const baseHeaders = ["ledger_id","product_id","product_name","change","direction","reason","ref_id","operator","date","target"];
  const sh = getLedgerSheet_(ss) || ensureSheet_(ss, "stock_ledger", baseHeaders);

  // 若 sheet 已有表頭，確保 baseHeaders 都存在（缺的就補在最後，不改動既有順序）
  const headers = getHeaders_(sh);
  const idx = headerIndexMap_(headers);
  let changed = false;
  let lastCol = sh.getLastColumn();
  baseHeaders.forEach(h => {
    if (idx[h] === undefined) {
      lastCol += 1;
      sh.getRange(1, lastCol).setValue(h);
      changed = true;
    }
  });
  if (changed) {
    // No-op, just makes sure header row is committed
  }

// 若需要顯示/追蹤歷史成本，追加欄位（只加在最後，不重排）
const extraHeaders = ["cost","sku","unit"];
extraHeaders.forEach(h => {
  if (idx[h] === undefined) {
    lastCol += 1;
    sh.getRange(1, lastCol).setValue(h);
    changed = true;
  }
});
  return sh;
}

function writeLedger_(ss, obj) {
  const ledger = ensureLedgerSheet_(ss);
  appendObject_(ledger, obj);
}

/**
 * 供應商代碼規則：yyyyMMdd + 3位流水（例：20260317001）
 * - 每日序號從 001 起算
 * - 只掃描 suppliers.id 欄位中符合當日前綴的純數字代碼
 * - 建議搭配 LockService 使用，避免併發重號
 */
function generateSupplierDailyId_(sheet, dateObj) {
  const d = dateObj instanceof Date ? dateObj : new Date();
  const prefix = Utilities.formatDate(d, "Asia/Taipei", "yyyyMMdd");
  const headers = getHeaders_(sheet);
  const idx = headerIndexMap_(headers);
  const col = idx.id;
  if (col === undefined) throw new Error('找不到欄位：id');

  const lastRow = sheet.getLastRow();
  let maxSeq = 0;
  if (lastRow >= 2) {
    const vals = sheet.getRange(2, col + 1, lastRow - 1, 1).getValues();
    for (let i = 0; i < vals.length; i++) {
      const s = String(vals[i][0] || '').trim();
      if (!s) continue;
      if (/^\d+$/.test(s) && s.indexOf(prefix) === 0 && s.length >= 11) {
        const suf = parseInt(s.slice(8), 10);
        if (!isNaN(suf)) maxSeq = Math.max(maxSeq, suf);
      }
    }
  }

  const next = maxSeq + 1;
  const suf3 = ('000' + next).slice(-3);
  return prefix + suf3;
}

/**
 * 單號規則：yyyyMMdd + 4位流水（例：202602090001）
 * - 每日序號從 0001 起算
 * - 以指定 sheet + 欄位掃描當日最大值後 +1
 * - 建議在 LockService 區塊內呼叫，避免併發重號
 */
function generateDailyDocNo_(sheet, idField, dateObj) {
  const d = dateObj instanceof Date ? dateObj : new Date();
  const prefix = Utilities.formatDate(d, "Asia/Taipei", "yyyyMMdd");

  const headers = getHeaders_(sheet);
  const idx = headerIndexMap_(headers);
  const col = idx[idField];
  if (col === undefined) throw new Error(`找不到欄位：${idField}`);

  const lastRow = sheet.getLastRow();
  let maxSeq = 0;

  if (lastRow >= 2) {
    const vals = sheet.getRange(2, col + 1, lastRow - 1, 1).getValues();
    for (let i = 0; i < vals.length; i++) {
      const s = String(vals[i][0] || "").trim();
      if (!s) continue;
      // 只看符合 prefix 的純數字單號（舊格式會被忽略）
      if (s.length >= 12 && s.indexOf(prefix) === 0) {
        const suf = parseInt(s.slice(8), 10);
        if (!isNaN(suf)) maxSeq = Math.max(maxSeq, suf);
      }
    }
  }

  const next = maxSeq + 1;
  const suf4 = ("0000" + next).slice(-4);
  return prefix + suf4;
}

function toISODateOnly_(s){
  // Accept YYYY-MM-DD or ROC like 115.02.03 or 115/02/03
  s = String(s || "").trim();
  if (!s) return Utilities.formatDate(new Date(), Session.getScriptTimeZone(), "yyyy-MM-dd");
  if (/^\d{4}-\d{2}-\d{2}$/.test(s)) return s;

  // ROC format
  const m = s.match(/^(\d{2,3})[\.\/\-](\d{1,2})[\.\/\-](\d{1,2})/);
  if (m){
    const y = Number(m[1]) + 1911;
    const mo = String(m[2]).padStart(2,"0");
    const d = String(m[3]).padStart(2,"0");
    return `${y}-${mo}-${d}`;
  }

  // Fallback parse
  const dt = new Date(s);
  if (!isNaN(dt.getTime())) return Utilities.formatDate(dt, Session.getScriptTimeZone(), "yyyy-MM-dd");
  return Utilities.formatDate(new Date(), Session.getScriptTimeZone(), "yyyy-MM-dd");
}

function toLocalDateTime_(d){

  return Utilities.formatDate(dt, "Asia/Taipei", "yyyy-MM-dd HH:mm:ss");
}


function mustJson_(s) {
  const raw = String(s == null ? "" : s);
  if (!raw) return null;
  try {
    return JSON.parse(raw);
  } catch (e1) {
    try {
      return JSON.parse(decodeURIComponent(raw));
    } catch (e2) {
      throw e2;
    }
  }
}

function syncPingtungRawAndReferencePrices_(options) {

  const syncTime = Utilities.formatDate(new Date(), Session.getScriptTimeZone(), "yyyy-MM-dd HH:mm:ss");
  const syncSource = String(options.source || '').trim() || 'manual';
  const batchId = generateId_('MPH');
  let triggerInfo = { installed: false, trigger_count: 0, handler: DAILY_MARKET_PRICE_TRIGGER_HANDLER_ };
  if (options.ensure_trigger !== false) {
    try {
      triggerInfo = ensureDailyNoonMarketPriceTrigger_();
    } catch (triggerErr) {
      triggerInfo = { installed: false, trigger_count: 0, handler: DAILY_MARKET_PRICE_TRIGGER_HANDLER_, warning: String(triggerErr && triggerErr.message || triggerErr || '') };
    }
  }
  const rawSh = ensureSheet_(ss, "PingtungRaw", PINGTUNG_RAW_HEADERS_);
  const existingRows = readNormalizedPingtungRawRows_(rawSh);

  let fetched = null;
  let usedFallback = false;
  try {
    fetched = fetchPingtungRawFromGist_();
  } catch (err) {
    const msg = String(err && err.message || err || '');
    const looksLikeHoliday = msg.indexOf('沒有可辨識的蔬菜行情資料') >= 0 || msg.indexOf('沒有可辨識的行情資料') >= 0;
    if (!looksLikeHoliday || !existingRows.length) throw err;
    fetched = { rows: [] };
    usedFallback = true;
  }

  const mergedRows = mergeLatestRawRows_(existingRows, fetched.rows || []);
  if (!mergedRows.length) throw new Error('目前沒有可保留的果菜市場價格資料');

  rawSh.clearContents();
  rawSh.getRange(1, 1, 1, PINGTUNG_RAW_HEADERS_.length).setValues([PINGTUNG_RAW_HEADERS_]);

    r.market_date,
    r.item_name,
    r.spec,
    r.market,
    r.upper_price,
    r.middle_price,
    r.lower_price,
    r.raw_json
  ]));

  const applied = applyReferencePricesFromRaw_(ss, mergedRows, { batch_id: batchId, sync_time: syncTime, sync_source: syncSource });
  const latestMarketDate = mergedRows.reduce((acc, r) => String(r.market_date || '') > acc ? String(r.market_date || '') : acc, '');
  const suffix = usedFallback
    ? `（本次未取得可用新行情，已保留最後一次更新價格，最新行情日期 ${latestMarketDate || '—'}）`
    : '';

  return {
    message: `同步完成：PingtungRaw ${mergedRows.length} 筆，商品主檔參考價格更新 ${applied.updated_count} 筆${suffix}`,
    raw_count: mergedRows.length,
    fetched_count: (fetched && Array.isArray(fetched.rows)) ? fetched.rows.length : 0,
    preserved_count: Math.max(0, mergedRows.length - (((fetched && Array.isArray(fetched.rows)) ? fetched.rows.length : 0))),
    updated_count: applied.updated_count,
    mapped_count: applied.mapped_count,
    unmatched_mapping_count: applied.unmatched_mapping_count,
    latest_market_date: latestMarketDate,
    used_fallback: usedFallback,
    batch_id: batchId,
    history_logged_count: applied.history_logged_count || 0,
    trigger_installed: !!triggerInfo.installed,
    trigger_count: Number(triggerInfo.trigger_count || 0),
    trigger_handler: String(triggerInfo.handler || DAILY_MARKET_PRICE_TRIGGER_HANDLER_),
    trigger_warning: String(triggerInfo.warning || ''),
    synced_at: syncTime
  };
}

function mergeLatestRawRows_(existingRows, fetchedRows) {
  const bucket = {};
  const put_ = (row, preferNew) => {
    if (!row) return;
    const market = normalizeMarketLabel_(row.market || '');
    const itemName = String(row.item_name || '').trim();
    const spec = String(row.spec || '').trim();
    if (!market || !itemName) return;
    const key = [market, normalizeLookupKey_(itemName), normalizeLookupKey_(spec)].join('||');
    const normalized = {
      sync_time: String(row.sync_time || '').trim(),
      source_file: String(row.source_file || '').trim(),
      market_date: normalizeLooseDate_(row.market_date || ''),
      item_name: itemName,
      spec: spec,
      market: market,
      upper_price: parseLooseNumber_(row.upper_price),
      middle_price: parseLooseNumber_(row.middle_price),
      lower_price: parseLooseNumber_(row.lower_price),
      raw_json: String(row.raw_json || '')
    };
    const current = bucket[key];
    if (!current) {
      bucket[key] = normalized;
      return;
    }
    const currentDate = String(current.market_date || '');
    const nextDate = String(normalized.market_date || '');
    if (nextDate > currentDate) {
      bucket[key] = normalized;
      return;
    }
    if (nextDate < currentDate) return;

    const currentScore = [current.upper_price, current.middle_price, current.lower_price].filter(v => v !== '' && v !== null && v !== undefined).length;
    const nextScore = [normalized.upper_price, normalized.middle_price, normalized.lower_price].filter(v => v !== '' && v !== null && v !== undefined).length;
    if (nextScore > currentScore) {
      bucket[key] = normalized;
      return;
    }
    if (nextScore < currentScore) return;

    const currentSync = String(current.sync_time || '');
    const nextSync = String(normalized.sync_time || '');
    if (nextSync > currentSync || (preferNew && nextSync === currentSync)) {
      bucket[key] = normalized;
    }
  };

  (Array.isArray(existingRows) ? existingRows : []).forEach(row => put_(row, false));
  (Array.isArray(fetchedRows) ? fetchedRows : []).forEach(row => put_(row, true));

  return Object.keys(bucket).map(k => bucket[k]).sort((a, b) => {
    const marketCmp = marketSortPriority_(b.market) - marketSortPriority_(a.market);
    if (marketCmp !== 0) return marketCmp;
    const dateCmp = String(b.market_date || '').localeCompare(String(a.market_date || ''));
    if (dateCmp !== 0) return dateCmp;
    const nameCmp = String(a.item_name || '').localeCompare(String(b.item_name || ''), 'zh-Hant');
    if (nameCmp !== 0) return nameCmp;
    return String(a.spec || '').localeCompare(String(b.spec || ''), 'zh-Hant');
  });
}

function fetchPingtungRawFromGist_() {

      Accept: "application/vnd.github+json",
      "User-Agent": "Google-Apps-Script"
    }
  });

    let content = String(f.content || "");
    if ((!content || f.truncated) && f.raw_url) {
      const rawRes = UrlFetchApp.fetch(String(f.raw_url), {
        muteHttpExceptions: true,
        headers: { "User-Agent": "Google-Apps-Script" }
      });
      if (rawRes.getResponseCode() >= 200 && rawRes.getResponseCode() < 300) content = rawRes.getContentText();

function parsePingtungMarketFile_(content, filename, mimeType) {
  const text = String(content || "").replace(/^﻿/, "").trim();
  if (!text) return [];

  const lowerName = String(filename || "").toLowerCase();
  let candidates = [];

  const tryJson = () => {
    try {
      const data = JSON.parse(text);
      candidates = extractRowsFromAnyJson_(data);
      return true;
    } catch (e) {
      return false;
    }
  };

  const tryDelimited = (delimiter) => {
    try {
      const rows = Utilities.parseCsv(text, delimiter);
      if (!rows || rows.length < 2) return false;
      const headers = rows[0].map(v => String(v || "").trim());
      candidates = rows.slice(1).map(r => {
        const obj = {};
        headers.forEach((h, idx) => { if (h) obj[h] = r[idx]; });
        return obj;
      });
      return true;
    } catch (e) {
      return false;
    }
  };

  if (lowerName.endsWith('.json') || /json/i.test(mimeType)) {
    if (!tryJson()) { tryDelimited(',') || tryDelimited('\t'); }
  } else if (lowerName.endsWith('.csv')) {
    if (!tryDelimited(',')) tryJson();
  } else if (lowerName.endsWith('.tsv') || lowerName.endsWith('.txt')) {
    if (!tryDelimited('\t')) { if (!tryDelimited(',')) tryJson(); }
  } else {
    if (!tryJson()) { if (!tryDelimited(',')) tryDelimited('\t'); }
  }

  if (!candidates.length && looksLikeMarketTextBlock_(text)) {
    candidates = parseMarketTextBlock_(text, filename, '');
  }

  const out = [];
  candidates.forEach(obj => {
    const norm = normalizeMarketRow_(obj);
    if (!norm) return;
    out.push({
      market_date: norm.market_date,
      item_name: norm.item_name,
      spec: norm.spec,
      market: norm.market,
      upper_price: norm.upper_price,
      middle_price: norm.middle_price,
      lower_price: norm.lower_price,
      raw_json: JSON.stringify(obj)
    });
  });
  return out;
}

function extractRowsFromAnyJson_(node) {
  const out = [];
  const seenTextBlocks = {};

  function pushTextBlock_(text, marketHint, dateHint) {
    const s = String(text || '').trim();
    if (!s) return;
    const key = `${String(marketHint || '')}__${String(dateHint || '')}__${s.slice(0, 120)}`;
    if (seenTextBlocks[key]) return;
    seenTextBlocks[key] = true;
    parseMarketTextBlock_(s, marketHint, dateHint).forEach(row => out.push(row));
  }

  function walk(v, keyHint, inheritedDateHint) {
    if (Array.isArray(v)) {
      v.forEach(item => walk(item, keyHint, inheritedDateHint));
      return;
    }
    if (typeof v === 'string') {
      if (looksLikeMarketTextBlock_(v)) pushTextBlock_(v, keyHint, inheritedDateHint);
      return;
    }
    if (!v || typeof v !== 'object') return;

    if (looksLikeMarketRowObject_(v)) out.push(v);

    const localDateHint = pickValueByAliases_(v, ['updatedAt','updated_at','date','market_date','trade_date','日期','交易日期']) || inheritedDateHint || '';
    Object.keys(v).forEach(k => {
      const child = v[k];
      if (typeof child === 'string') {
        if (looksLikeMarketTextBlock_(child)) pushTextBlock_(child, k, localDateHint);
        return;
      }
      if (Array.isArray(child) || (child && typeof child === 'object')) walk(child, k, localDateHint);
    });
  }

  walk(node, '', '');
  return out;
}


function looksLikeMarketTextBlock_(text) {
  const s = String(text || '').trim();
  if (!s) return false;
  return /上價\s*[\d\-]/.test(s) && /中價\s*[\d\-]/.test(s) && /下價\s*[\d\-]/.test(s) && /交易日期/.test(s);
}

function parseMarketTextBlock_(text, marketHint, dateHint) {
  const out = [];
  const lines = String(text || '').split(/\r?\n+/).map(v => String(v || '').trim()).filter(Boolean);
  lines.forEach(line => {
    const cleaned = line.replace(/^[•‧・●○◦▪■□◆◇★☆※◎◉\-]+\s*/, '').trim();
    if (!cleaned) return;

    const itemPart = String((cleaned.split(/[：:]/)[0] || '')).trim();
    const upperM = cleaned.match(/上價\s*([^，,]+)/);
    const middleM = cleaned.match(/中價\s*([^，,]+)/);
    const lowerM = cleaned.match(/下價\s*([^，,]+)/);
    const dateM = cleaned.match(/交易日期\s*([0-9\.\/-]+)/);

    const upper = parseLooseNumber_(upperM ? upperM[1] : '');
    const middle = parseLooseNumber_(middleM ? middleM[1] : '');
    const lower = parseLooseNumber_(lowerM ? lowerM[1] : '');
    const marketDate = normalizeLooseDate_(dateM ? dateM[1] : dateHint);

    if (!itemPart || (upper === '' && middle === '' && lower === '')) return;

    const split = splitMarketItemAndSpec_(itemPart);
    out.push({
      item_name: split.item_name,
      spec: split.spec,
      market: normalizeMarketLabel_(marketHint),
      market_date: marketDate,
      upper_price: upper,
      middle_price: middle,
      lower_price: lower,
      raw_line: cleaned
    });
  });
  return out;
}

function splitMarketItemAndSpec_(label) {
  const s = String(label || '').trim();
  if (!s) return { item_name: '', spec: '' };
  const parts = s.split(/\s*[\-－–—]\s*/);
  if (parts.length <= 1) return { item_name: s, spec: '' };
  return {
    item_name: String(parts.shift() || '').trim(),
    spec: String(parts.join('-') || '').trim()
  };
}

function normalizeMarketLabel_(marketHint) {
  const raw = String(marketHint || '').trim();
  const nk = normalizeLookupKey_(raw);
  if (nk.indexOf('pingtung') >= 0 || raw.indexOf('屏東') >= 0) return 'Pingtung';
  if (nk.indexOf('firstmarket') >= 0 || raw.indexOf('第一市場') >= 0) return 'FirstMarket';
  if (nk.indexOf('secondmarket') >= 0 || raw.indexOf('第二市場') >= 0) return 'SecondMarket';
  return raw;
}

function marketSortPriority_(market) {
  const raw = String(market || '').trim();
  const nk = normalizeLookupKey_(raw);
  if (nk.indexOf('pingtung') >= 0 || raw.indexOf('屏東') >= 0) return 30;
  if (nk.indexOf('secondmarket') >= 0 || raw.indexOf('第二市場') >= 0) return 20;
  if (nk.indexOf('firstmarket') >= 0 || raw.indexOf('第一市場') >= 0) return 10;
  return 0;
}


function looksLikeMarketRowObject_(obj) {
  const name = pickValueByAliases_(obj, ["item_name","name","item","product","crop","commodity","菜名","品名","名稱","作物名稱","作物","蔬菜名稱"]);
  const upper = pickValueByAliases_(obj, ["upper_price","high_price","high","upper","max_price","最高價","上價"]);
  const middle = pickValueByAliases_(obj, ["middle_price","mid_price","mid","middle","avg_price","中價","平均價"]);
  const lower = pickValueByAliases_(obj, ["lower_price","low_price","low","lower","min_price","最低價","下價"]);
  return !!String(name || "").trim() && [upper, middle, lower].some(v => String(v || "").trim() !== "");
}

function normalizeMarketRow_(obj) {
  const combinedName = String(pickValueByAliases_(obj, ['item_name','name','item','product','crop','commodity','菜名','品名','名稱','作物名稱','作物','蔬菜名稱']) || '').trim();
  if (!combinedName) return null;

  let spec = String(pickValueByAliases_(obj, ['spec','grade','size','規格','規格別','品級','品規']) || '').trim();
  let itemName = combinedName;
  if (!spec) {
    const split = splitMarketItemAndSpec_(combinedName);
    itemName = split.item_name;
    spec = split.spec;
  }

  const market = String(pickValueByAliases_(obj, ['market','market_name','市場','交易市場']) || '').trim();
  const marketDate = normalizeLooseDate_(pickValueByAliases_(obj, ['date','market_date','trade_date','updatedAt','updated_at','日期','交易日期','更新日期','查詢日期']) || '');
  const upper = parseLooseNumber_(pickValueByAliases_(obj, ['upper_price','high_price','high','upper','max_price','最高價','上價']));
  const middle = parseLooseNumber_(pickValueByAliases_(obj, ['middle_price','mid_price','mid','middle','avg_price','中價','平均價']));
  const lower = parseLooseNumber_(pickValueByAliases_(obj, ['lower_price','low_price','low','lower','min_price','最低價','下價']));
  if (upper === '' && middle === '' && lower === '') return null;
  return { item_name: itemName, spec: spec, market: market, market_date: marketDate, upper_price: upper, middle_price: middle, lower_price: lower };
}

function pickValueByAliases_(obj, aliases) {
  const map = {};
  Object.keys(obj || {}).forEach(k => { map[normalizeLookupKey_(k)] = obj[k]; });
  for (let i = 0; i < aliases.length; i++) {
    const nk = normalizeLookupKey_(aliases[i]);
    if (Object.prototype.hasOwnProperty.call(map, nk) && map[nk] !== "" && map[nk] !== null && map[nk] !== undefined) return map[nk];
  }
  const normKeys = Object.keys(map);
  for (let i = 0; i < aliases.length; i++) {
    const ak = normalizeLookupKey_(aliases[i]);
    const hit = normKeys.find(k => k.indexOf(ak) >= 0 || ak.indexOf(k) >= 0);
    if (hit && map[hit] !== "" && map[hit] !== null && map[hit] !== undefined) return map[hit];
  }
  return "";
}

function normalizeLookupKey_(s) {
  return String(s || "").toLowerCase().replace(/[\s_\-\/\\(\)\[\]\{\}\.。:：]/g, "").trim();
}

function parseLooseNumber_(v) {
  if (v === null || v === undefined) return "";
  if (typeof v === 'number') return Number.isFinite(v) ? v : "";
  const s = String(v).trim();
  if (!s || s === '-' || s === '--') return "";
  const m = s.replace(/[,，\s]/g, '').match(/-?\d+(?:\.\d+)?/);
  if (!m) return "";
  const n = Number(m[0]);
  return Number.isFinite(n) ? n : "";
}

function normalizeLooseDate_(v) {
  if (!v) return '';
  if (Object.prototype.toString.call(v) === '[object Date]' && !isNaN(v.getTime())) return Utilities.formatDate(v, Session.getScriptTimeZone(), 'yyyy-MM-dd');
  const s = String(v).trim();
  if (!s) return '';
  if (/^\d{4}-\d{2}-\d{2}/.test(s)) return s.slice(0,10);
  if (/^\d{4}[\.\/]\d{1,2}[\.\/]\d{1,2}/.test(s)) {
    const parts = s.split(/[\sT]/)[0].split(/[\.\/]/);
    return `${parts[0]}-${('0'+parts[1]).slice(-2)}-${('0'+parts[2]).slice(-2)}`;
  }
  if (/^\d{2,3}[\.\/-]\d{1,2}[\.\/-]\d{1,2}/.test(s)) {
    const parts = s.split(/[\sT]/)[0].split(/[\.\/-]/);
    const rocYear = Number(parts[0]);
    if (Number.isFinite(rocYear)) {
      const year = rocYear < 1911 ? (rocYear + 1911) : rocYear;
      return `${year}-${('0'+parts[1]).slice(-2)}-${('0'+parts[2]).slice(-2)}`;
    }
  }
  const dt = new Date(s);
  if (!isNaN(dt.getTime())) return Utilities.formatDate(dt, Session.getScriptTimeZone(), 'yyyy-MM-dd');
  return s;
}

  };

  if (synRows.length) {
    synRows.forEach(row => {
      if (!enabledValue_(row)) return;
      const rawName = String(pickValueByAliases_(row, ['raw_name','market_name','raw','alias','synonym','keyword','原始名稱','外部名稱','行情名稱','對照名稱','比對名稱']) || '').trim();
      const productName = String(pickValueByAliases_(row, ['product_name','product','商品名稱','商品','系統商品','對應商品','name']) || '').trim();
      const spec = String(pickValueByAliases_(row, ['spec','規格','對應規格']) || '').trim();
      const productId = String(pickValueByAliases_(row, ['product_id','id','商品id','商品ID']) || '').trim();
      const lookupName = rawName || productName;
      if (!lookupName) return;
      const hits = (rawLookup[normalizeLookupKey_(lookupName)] || []).filter(hit => !spec || String(hit.spec || '').trim() === spec);
      if (!hits.length) { unmatchedMappingCount++; return; }
      const rowNo = findProductRow_(productId, productName || lookupName, spec);
      if (rowNo < 0) { unmatchedMappingCount++; return; }
      mappedCount++;
      setRowPrice_(rowNo, hits[0], lookupName, true);
    });
  } else {
    for (let r = 1; r < values.length; r++) {
      const rowName = idx.name !== undefined ? String(values[r][idx.name] || '').trim() : '';
      const rowSpec = idx.spec !== undefined ? String(values[r][idx.spec] || '').trim() : '';
      if (!rowName) continue;
      const baseHits = (rawLookup[normalizeLookupKey_(rowName)] || []).filter(hit => !rowSpec || String(hit.spec || '').trim() === rowSpec);
      const combinedHits = rowSpec ? (rawLookup[normalizeLookupKey_(`${rowName}-${rowSpec}`)] || []) : [];
      const hits = baseHits.length ? baseHits : combinedHits;
      if (!hits.length) continue;
      mappedCount++;
      setRowPrice_(r, hits[0], rowName, false);
    }
  }

  prodSh.getRange(1, 1, values.length, values[0].length).setValues(values);

  const existingHistorySignatures = {};
  readObjects_(historySh).forEach(row => {
    const sig = makeHistoryDedupKey_(row);
    if (sig) existingHistorySignatures[sig] = true;
  });

  const pendingHistorySignatures = {};
  const historyRows = synRows.length ? Object.keys(historyByProduct).sort().map(key => historyByProduct[key]).filter(row => {
    const sig = makeHistoryDedupKey_(row);
    if (!sig) return false;
    if (existingHistorySignatures[sig] || pendingHistorySignatures[sig]) return false;
    pendingHistorySignatures[sig] = true;
    return true;
  }) : [];
  historyRows.forEach((row, idxRow) => {
    row.history_id = `${batchId}-${('0000' + (idxRow + 1)).slice(-4)}`;
  });
  if (historyRows.length) appendObjectsBatch_(historySh, historyRows);

  return { updated_count: updatedCount, mapped_count: mappedCount, unmatched_mapping_count: unmatchedMappingCount, history_logged_count: historyRows.length, batch_id: batchId };
}


function buildMarketPriceBoard_(ss) {
  const rawSh = ensureSheet_(ss, "PingtungRaw", PINGTUNG_RAW_HEADERS_);
  const synSh = ensureSheet_(ss, 'Synonyms', ['raw_name','product_name','spec','product_id','enabled','note']);
  const prodSh = ensureSheet_(ss, 'Products', ['id','name','sku','price','image','stock','category','date','cost','unit','safety_stock','supplier_ids','expiry_date','last_purchase_date','spec','reference_price','reference_price_date','reference_price_name','shop_enabled']);

  const rawRows = readObjects_(rawSh).map(r => ({
    sync_time: String(r.sync_time || '').trim(),
    source_file: String(r.source_file || '').trim(),
    market_date: normalizeLooseDate_(r.market_date || ''),
    item_name: String(r.item_name || '').trim(),
    spec: String(r.spec || '').trim(),
    market: normalizeMarketLabel_(r.market || ''),
    upper_price: parseLooseNumber_(r.upper_price),
    middle_price: parseLooseNumber_(r.middle_price),
    lower_price: parseLooseNumber_(r.lower_price),
    raw_json: String(r.raw_json || '')
  })).filter(r => r.item_name);
  const synRows = readObjects_(synSh);
  const products = readObjects_(prodSh).map(p => ({
    id: String(p.id || '').trim(),
    name: String(p.name || '').trim(),
    sku: String(p.sku || p.part_no || p.code || '').trim(),
    spec: String(p.spec || '').trim(),
    unit: String(p.unit || '').trim(),
    category: String(p.category || '').trim(),
    cost: parseLooseNumber_(p.cost),
    price: parseLooseNumber_(p.price),
    last_purchase_date: normalizeLooseDate_(p.last_purchase_date || '')
  })).filter(p => p.name);

  const rawByMarket = buildRawLookupByMarket_(rawRows);
  const groups = { FirstMarket: [], SecondMarket: [], Pingtung: [] };

  products.forEach(product => {
    const aliases = getProductMarketAliases_(product, synRows);
    ['FirstMarket', 'SecondMarket', 'Pingtung'].forEach(marketKey => {
      const hit = chooseLatestHitByMarket_(rawByMarket[marketKey] || {}, aliases, product.name, product.spec);
      if (!hit) return;
      groups[marketKey].push({
        product_id: product.id,
        sku: product.sku,
        product_name: product.name,
        product_spec: product.spec,
        unit: product.unit,
        category: product.category,
        cost: product.cost,
        price: product.price,
        last_purchase_date: String(product.last_purchase_date || ''),
        market: marketKey,
        market_label: marketDisplayLabel_(marketKey),
        market_date: String(hit.market_date || ''),
        raw_name: String(hit.item_name || ''),
        raw_spec: String(hit.spec || ''),
        upper_price: hit.upper_price,
        middle_price: hit.middle_price,
        lower_price: hit.lower_price
      });
    });
  });

  Object.keys(groups).forEach(k => groups[k].sort((a, b) => {
    const catCmp = String(a.category || '').localeCompare(String(b.category || ''), 'zh-Hant');
    if (catCmp !== 0) return catCmp;
    const nameCmp = String(a.product_name || '').localeCompare(String(b.product_name || ''), 'zh-Hant');
    if (nameCmp !== 0) return nameCmp;
    return String(a.product_spec || '').localeCompare(String(b.product_spec || ''), 'zh-Hant');
  }));

  const latestSync = rawRows.reduce((acc, r) => String(r.sync_time || '') > acc ? String(r.sync_time || '') : acc, '');
  const latestMarketDate = rawRows.reduce((acc, r) => String(r.market_date || '') > acc ? String(r.market_date || '') : acc, '');
  const matchedCount = Object.keys(groups).reduce((sum, k) => sum + groups[k].length, 0);

  return {
    synced_at: latestSync,
    latest_market_date: latestMarketDate,
    raw_count: rawRows.length,
    product_count: products.length,
    matched_count: matchedCount,
    groups: groups
  };
}

function buildRawLookupByMarket_(rawRows) {
  const byMarket = { FirstMarket: {}, SecondMarket: {}, Pingtung: {} };
  const addHit_ = (marketKey, key, row) => {
    if (!byMarket[marketKey]) byMarket[marketKey] = {};
    const nk = normalizeLookupKey_(key);
    if (!nk) return;
    if (!byMarket[marketKey][nk]) byMarket[marketKey][nk] = [];
    byMarket[marketKey][nk].push(row);
  };

  rawRows.forEach(row => {
    const marketKey = normalizeMarketLabel_(row.market || '');
    if (!byMarket[marketKey]) return;
    const itemName = String(row.item_name || '').trim();
    const spec = String(row.spec || '').trim();
    if (!itemName) return;
    addHit_(marketKey, itemName, row);
    if (spec) addHit_(marketKey, `${itemName}-${spec}`, row);
  });

  Object.keys(byMarket).forEach(marketKey => {
    Object.keys(byMarket[marketKey]).forEach(key => {
      byMarket[marketKey][key].sort((a, b) => compareMarketHitDesc_(a, b));
    });
  });
  return byMarket;
}

function compareMarketHitDesc_(a, b) {
  const ad = String(a && a.market_date || '');
  const bd = String(b && b.market_date || '');
  const dateCmp = bd.localeCompare(ad);
  if (dateCmp !== 0) return dateCmp;
  const as = String(a && a.spec || '');
  const bs = String(b && b.spec || '');
  if (!!bs !== !!as) return bs ? 1 : -1;
  return String(b && b.item_name || '').localeCompare(String(a && a.item_name || ''), 'zh-Hant');
}

function marketDisplayLabel_(market) {
  const key = normalizeMarketLabel_(market);
  if (key === 'FirstMarket') return '台北第一果菜市場';
  if (key === 'SecondMarket') return '台北第二果菜市場';
  if (key === 'Pingtung') return '屏東果菜市場';
  return String(market || '');
}

function getProductMarketAliases_(product, synRows) {
  const pid = String(product && product.id || '').trim();
  const pname = String(product && product.name || '').trim();
  const pspec = String(product && product.spec || '').trim();
  const out = [];
  const seen = {};
  const pushAlias_ = (rawName, spec) => {
    const rn = String(rawName || '').trim();
    const sp = String(spec || '').trim();
    if (!rn) return;
    const key = `${normalizeLookupKey_(rn)}__${normalizeLookupKey_(sp)}`;
    if (seen[key]) return;
    seen[key] = true;
    out.push({ raw_name: rn, spec: sp });
  };

  synRows.forEach(row => {
    if (!enabledSynonymRow_(row)) return;
    const targetId = String(pickValueByAliases_(row, ['product_id','id','商品id','商品ID']) || '').trim();
    const targetName = String(pickValueByAliases_(row, ['product_name','product','商品名稱','商品','系統商品','對應商品','name']) || '').trim();
    const targetSpec = String(pickValueByAliases_(row, ['spec','規格','對應規格']) || '').trim();
    const rawName = String(pickValueByAliases_(row, ['raw_name','market_name','raw','alias','synonym','keyword','原始名稱','外部名稱','行情名稱','對照名稱','比對名稱']) || '').trim();

    let matched = false;
    if (pid && targetId && pid === targetId) matched = true;
    if (!matched && targetName && pname === targetName) {
      matched = (!targetSpec || !pspec || targetSpec === pspec);
    }
    if (!matched) return;
    pushAlias_(rawName || targetName || pname, targetSpec || pspec);
  });

  pushAlias_(pname, pspec);
  return out;
}

function enabledSynonymRow_(rowObj) {
  const v = pickValueByAliases_(rowObj, ['enabled','enable','active','啟用','是否啟用']);
  if (v === '' || v === null || v === undefined) return true;
  const s = String(v).trim().toLowerCase();
  return !(s === '0' || s === 'false' || s === 'no' || s === 'n' || s === '停用' || s === '否');
}

function chooseLatestHitByMarket_(marketLookup, aliases, productName, productSpec) {
  let best = null;
  const candidateAliases = Array.isArray(aliases) && aliases.length ? aliases : [{ raw_name: productName, spec: productSpec }];
  candidateAliases.forEach(alias => {
    const rawName = String(alias && alias.raw_name || productName || '').trim();
    const targetSpec = String(alias && alias.spec || productSpec || '').trim();
    if (!rawName) return;

    let hits = [];
    const byName = marketLookup[normalizeLookupKey_(rawName)] || [];
    const byCombo = targetSpec ? (marketLookup[normalizeLookupKey_(`${rawName}-${targetSpec}`)] || []) : [];
    if (targetSpec) {
      const exact = byName.filter(hit => String(hit.spec || '').trim() === targetSpec);
      hits = exact.length ? exact.concat(byCombo) : byCombo.concat(byName);
    } else {
      hits = byName.concat(byCombo);
    }

    hits.forEach(hit => {
      if (!best || compareMarketHitDesc_(hit, best) < 0) {
        best = hit;
      }
    });
  });
  return best;
}

function buildOrderLookupMaps_(ss) {
  const sh = ensureSheet_(ss, "Orders", orderHeaders_());
  const rows = sh.getDataRange().getValues();
  const out = { linkedPurchaseToOrderId: {}, orderIdToDate: {} };
  if (rows.length <= 1) return out;
  const idx = headerIndexMap_(rows[0].map(h => String(h || "").trim()));
  const orderIdIdx = idx.order_id;
  const linkedPurchaseIdx = idx.linked_purchase_id;
  const dateIdx = idx.date;
  for (let r = 1; r < rows.length; r++) {
    const row = rows[r] || [];
    const orderId = String(orderIdIdx !== undefined ? (row[orderIdIdx] || "") : "").trim();
    if (!orderId) continue;
    if (dateIdx !== undefined) {
      out.orderIdToDate[orderId] = toISODateLocal_(row[dateIdx] || "");
    }
    if (linkedPurchaseIdx !== undefined) {
      const linkedPurchaseId = String(row[linkedPurchaseIdx] || "").trim();
      if (linkedPurchaseId && !out.linkedPurchaseToOrderId[linkedPurchaseId]) {
        out.linkedPurchaseToOrderId[linkedPurchaseId] = orderId;
      }
    }
  }
  return out;
}

function findOrderIdByLinkedPurchaseId_(ss, poId, lookup){
  const target = String(poId || "").trim();
  if (!target) return "";
  if (lookup && lookup.linkedPurchaseToOrderId) {
    return String(lookup.linkedPurchaseToOrderId[target] || "").trim();
  }
  const built = buildOrderLookupMaps_(ss);
  return String(built.linkedPurchaseToOrderId[target] || "").trim();
}

function ensurePurchasesSheet_(ss){
  const expected = purchaseHeaders_();
  const sh = ensureSheet_(ss, "purchases", expected);
  const current = getHeaders_(sh);
  const normalizedCurrent = current.map(h => String(h || "").trim());
  const hasDup = normalizedCurrent.some((h, i) => !!h && normalizedCurrent.indexOf(h) !== i);
  const sameOrder = expected.length === normalizedCurrent.length && expected.every((h, i) => normalizedCurrent[i] === h);
  if (sameOrder && !hasDup) return sh;

  const vals = sh.getDataRange().getValues();
  const currentHeaders = (vals[0] || []).map(h => String(h || "").trim());
  const firstIdx = {};
  currentHeaders.forEach((h, i) => {
    if (h && firstIdx[h] === undefined) firstIdx[h] = i;
  });
  const aliases = {
    po_id: ["po_id", "purchase_id", "purchaseId", "id", "ref_id"],
    date: ["date"],
    arrival_date: ["arrival_date", "receive_date"],
    form_no: ["form_no", "form_code"],
    form_name: ["form_name", "form_title"],
    supplier_id: ["supplier_id", "supplierId", "supplier"],
    supplier_name: ["supplier_name", "supplierName"],
    total: ["total"],
    items: ["items"],
    expiry_date: ["expiry_date"],
    status: ["status"],
    operator: ["operator"],
    created_at: ["created_at"],
    source_order_id: ["source_order_id"],
    stock_applied: ["stock_applied"],
    completed_at: ["completed_at"],
    auto_generated: ["auto_generated"]
  };

  const orderLookup = buildOrderLookupMaps_(ss);
  const out = [expected.slice()];
  for (let r = 1; r < vals.length; r++) {
    const row = vals[r] || [];
    if (row.every(v => v === "" || v === null)) continue;
    const obj = {};
    expected.forEach((h) => {
      const names = aliases[h] || [h];
      let hitIdx = -1;
      for (let i = 0; i < names.length; i++) {
        const idx = firstIdx[names[i]];
        if (idx !== undefined) { hitIdx = idx; break; }
      }
      obj[h] = hitIdx >= 0 ? row[hitIdx] : "";
    });

    if (!String(obj.form_no || "").trim()) {
      const info = getPurchaseFormInfoByCategory_(String(obj.form_name || "").trim());
      obj.form_no = info.form_no;
    }
    if (!String(obj.form_name || "").trim()) {
      const info = getPurchaseFormInfoByCategory_(String(obj.form_name || "").trim());
      obj.form_name = info.form_name;
    }
    if (!String(obj.source_order_id || "").trim() && Number(obj.auto_generated || 0)) {
      const linkedOrderId = findOrderIdByLinkedPurchaseId_(ss, obj.po_id, orderLookup);
      if (linkedOrderId) obj.source_order_id = linkedOrderId;
    }

    out.push(expected.map(h => Object.prototype.hasOwnProperty.call(obj, h) ? obj[h] : ""));
  }

  const maxRows = sh.getMaxRows();
  const maxCols = sh.getMaxColumns();
  if (maxRows < out.length) sh.insertRowsAfter(maxRows, out.length - maxRows);
  if (maxCols < expected.length) sh.insertColumnsAfter(maxCols, expected.length - maxCols);
  sh.clearContents();
  sh.getRange(1, 1, out.length, expected.length).setValues(out);
  if (sh.getMaxColumns() > expected.length) {
    sh.deleteColumns(expected.length + 1, sh.getMaxColumns() - expected.length);
  }
  return sh;
}

function applyPurchaseStock_(ss, prodSh, po_id, items, operator, dateValue){
  const ph = getHeaders_(prodSh);
  const pidx = headerIndexMap_(ph);
  const prodVals = prodSh.getDataRange().getValues();
  const byId = {};
  for (let r = 1; r < prodVals.length; r++) {
    byId[String(prodVals[r][pidx.id]).trim()] = { r: r, row: prodVals[r] };
  }
  const now = new Date();
  const purchaseDateIso = toISODateLocal_(dateValue || now);
  for (let i = 0; i < items.length; i++) {
    const it = items[i] || {};
    const pid = String(it.product_id || it.id || "").trim();
    const qtyRaw = (it.qty_raw !== undefined && it.qty_raw !== null && String(it.qty_raw).trim() !== "") ? it.qty_raw : it.qty;
    const costRaw = (it.cost_raw !== undefined && it.cost_raw !== null && String(it.cost_raw).trim() !== "") ? it.cost_raw : (it.cost ?? 0);
    const qty = cleanDecimalInput_(qtyRaw, 6);
    const unitCost = cleanDecimalInput_(costRaw, 6);
    if (!pid || !(qty > 0)) throw new Error("品項資料不完整，無法完成驗收入庫");
    const hit = byId[pid];
    if (!hit) throw new Error(`找不到商品 id=${pid}`);
    const current = cleanDecimalInput_(hit.row[pidx.stock] || 0, 6);
    const next = addDecimalInput_(current, qty, 6);
    prodSh.getRange(hit.r + 1, pidx.stock + 1).setValue(next);
    if (isFinite(unitCost) && unitCost > 0 && pidx.cost !== undefined) {
      prodSh.getRange(hit.r + 1, pidx.cost + 1).setValue(unitCost);
      hit.row[pidx.cost] = unitCost;
      const nextSalePrice = calcRoundedSalePriceFromCost_(unitCost, 25);
      if (nextSalePrice !== "" && pidx.price !== undefined) {
        prodSh.getRange(hit.r + 1, pidx.price + 1).setValue(nextSalePrice);
        hit.row[pidx.price] = nextSalePrice;
      }
    }
    if (purchaseDateIso && pidx.last_purchase_date !== undefined) {
      const oldLast = toISODateLocal_(hit.row[pidx.last_purchase_date] || "");
      const finalLast = (!oldLast || oldLast < purchaseDateIso) ? purchaseDateIso : oldLast;
      prodSh.getRange(hit.r + 1, pidx.last_purchase_date + 1).setValue(finalLast);
      hit.row[pidx.last_purchase_date] = finalLast;
    }
    const itemExpiry = toISODateLocal_(it.expiry_date || it.expire_date || it.expiry || it.valid_date || it.valid_until || "");
    if (itemExpiry && pidx.expiry_date !== undefined) {
      prodSh.getRange(hit.r + 1, pidx.expiry_date + 1).setValue(itemExpiry);
      hit.row[pidx.expiry_date] = itemExpiry;
      it.expiry_date = itemExpiry;
    }
    writeLedger_(ss, {
      ledger_id: generateId_("LG"),
      product_id: pid,
      product_name: String(hit.row[pidx.name] || it.product_name || ""),
      sku: (pidx.sku !== undefined) ? String(hit.row[pidx.sku] || it.sku || "") : String(it.sku || ""),
      unit: (pidx.unit !== undefined) ? String(hit.row[pidx.unit] || it.unit || "") : String(it.unit || ""),
      change: qty,
      direction: "IN",
      reason: "purchase",
      ref_id: String(po_id),
      operator: operator,
      date: toLocalDateTime_(now),
      cost: (isFinite(unitCost) && unitCost > 0) ? unitCost : (pidx.cost !== undefined ? (Number(hit.row[pidx.cost] || 0) || "") : ""),
      target: String(it.supplier_name || "")
    });
  }
}

    const current = cleanDecimalInput_(hit.row[pidx.stock] || 0, 6);
    const next = addDecimalInput_(current, -qty, 6);
    if (next < 0) throw new Error(`庫存不足：${hit.row[pidx.name]} current=${current}, qty=${qty}`);
    prodSh.getRange(hit.r + 1, pidx.stock + 1).setValue(next);
    writeLedger_(ss, {
      ledger_id: generateId_("LG"),
      product_id: pid,
      product_name: String(hit.row[pidx.name] || it.name || ""),
      sku: (pidx.sku !== undefined) ? String(hit.row[pidx.sku] || "") : "",
      unit: (pidx.unit !== undefined) ? String(hit.row[pidx.unit] || "") : "",
      change: -qty,
      direction: "OUT",
      reason: "order",
      ref_id: String(order_id),
      operator: operator,
      date: toLocalDateTime_(now),
      cost: (pidx.cost !== undefined) ? (Number(hit.row[pidx.cost] || 0) || "") : "",
      target: target
    });
  }
}

function normalizePurchaseRowForApi_(ss, x, i, includeItems, lookup) {
  x = x || {};
  x.date = toISODateLocal_(x.date);
  x.arrival_date = toISODateLocal_(x.arrival_date || x.receive_date || "");
  x.created_at = toISODateLocal_(x.created_at);
  x.completed_at = toISODateLocal_(x.completed_at || "");
  x.po_id = x.po_id || x.purchase_id || x.purchaseId || x.id || x.ref_id || "";
  x.form_no = x.form_no || x.form_code || "";
  x.form_name = x.form_name || x.form_title || "";
  x.supplier_id = x.supplier_id || x.supplierId || x.supplier || "";
  x.supplier_name = x.supplier_name || x.supplierName || "";
  x.status = String(x.status || "待驗收").trim() || "待驗收";
  x.stock_applied = Number(x.stock_applied || 0) ? 1 : 0;
  x.source_order_id = String(x.source_order_id || "").trim();
  x.auto_generated = Number(x.auto_generated || 0) ? 1 : 0;
  if (!x.source_order_id) {
    const fallbackSourceOrderId = String(findOrderIdByLinkedPurchaseId_(ss, x.po_id, lookup) || "").trim();
    if (fallbackSourceOrderId) {
      x.source_order_id = fallbackSourceOrderId;
      if (!x.auto_generated) x.auto_generated = 1;
    }
  }
  if (!String(x.form_no || "").trim()) {
    const info = getPurchaseFormInfoByCategory_(String(x.form_name || "").trim());
    x.form_no = info.form_no;
    if (!String(x.form_name || "").trim()) x.form_name = info.form_name;
  }
  if (x.auto_generated && x.source_order_id) {
    const linkedOrderDate = getOrderDateById_(ss, x.source_order_id, lookup);
    if (linkedOrderDate) x.date = linkedOrderDate;
  }

  const raw = x.items;
  if (includeItems) {
    if (raw && typeof raw === "string") {
      try { x.items = JSON.parse(raw); } catch (e) { x.items = []; }
    }
    if (!Array.isArray(x.items)) x.items = [];
    x.items_loaded = 1;
    x.item_count = x.items.length;
  } else {
    x.items = [];
    x.items_loaded = 0;
    x.item_count = 0;
  }

  if (!String(x.po_id || "").trim()) {
    const hasPurchaseLikeData = String(x.date || "").trim() || String(x.supplier_id || x.supplier_name || "").trim() || String(x.total || "").trim() || String(raw || "").trim();
    if (hasPurchaseLikeData) x.po_id = `ROW${i + 2}`;
  }
  return x;
}

function apiPurchases_(p, respond) {
  if (typeof p === "function") {
    respond = p;
    p = {};
  }
  p = p || {};

  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sh = ensurePurchasesSheet_(ss);
  const rawList = readObjects_(sh);
  const targetPoId = String(p.po_id || p.id || "").trim();
  const includeItems = targetPoId || Number(p.detail || p.include_items || 0) ? 1 : 0;
  const orderLookup = buildOrderLookupMaps_(ss);

  if (targetPoId) {
    for (let i = 0; i < rawList.length; i++) {
      const raw = rawList[i] || {};
      const rawPoId = String(raw.po_id || raw.purchase_id || raw.purchaseId || raw.id || raw.ref_id || "").trim();
      if (rawPoId !== targetPoId) continue;

    .map((x, i) => normalizePurchaseRowForApi_(ss, x, i, includeItems, orderLookup))
    .filter(x => String(x.po_id || "").trim() !== "");

  return respond({ status: "ok", data: list, count: list.length });

function findRecentDuplicatePurchasePoId_(rows, idx, payload) {
  var incomingPoId = String(payload && payload.po_id || "").trim();
  if (incomingPoId) return "";
  var targetItemsSig = purchaseDedupeItemSignature_(payload && payload.items);
  if (!targetItemsSig || targetItemsSig === "[]") return "";
  var targetSourceOrderId = String(payload && payload.source_order_id || "").trim();
  var targetOperator = String(payload && payload.operator || "").trim();
  var targetDate = String(payload && payload.date || "").trim();
  var targetArrivalDate = String(payload && payload.arrival_date || "").trim();
  var targetFormNo = String(payload && payload.form_no || "").trim();
  var targetSupplierId = String(payload && payload.supplier_id || "").trim();
  var now = Date.now();
  for (var r = rows.length - 1; r >= 1; r--) {
    var rowPoId = String(rows[r][idx.po_id] || "").trim();
    if (!rowPoId) continue;
    var createdAt = rows[r][idx.created_at];
    var createdMs = createdAt instanceof Date ? createdAt.getTime() : new Date(createdAt || "").getTime();
    if (isFinite(createdMs)) {
      var ageMs = now - createdMs;
      if (ageMs > 20000) break;
      if (ageMs < -5000) continue;
    }
    if (String(rows[r][idx.source_order_id] || "").trim() !== targetSourceOrderId) continue;
    if (String(rows[r][idx.operator] || "").trim() !== targetOperator) continue;
    if (String(rows[r][idx.date] || "").trim() !== targetDate) continue;
    if (String(rows[r][idx.arrival_date] || "").trim() !== targetArrivalDate) continue;
    if (String(rows[r][idx.form_no] || "").trim() !== targetFormNo) continue;
    if (String(rows[r][idx.supplier_id] || "").trim() !== targetSupplierId) continue;
    var rowItems = [];
    try {
      rowItems = JSON.parse(rows[r][idx.items] || "[]");
    } catch (err) {
      rowItems = [];
    }
    if (purchaseDedupeItemSignature_(rowItems) !== targetItemsSig) continue;
    return rowPoId;
  }
  return "";
}

function parsePurchasePayload_(p) {

  return mustJson_(p && p.purchase);
}

function apiManagePurchase_(p, respond) {
  const action = String(p.action || "").trim();
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const purchasesSh = ensurePurchasesSheet_(ss);
  const prodSh = ensureSheet_(ss, "Products", ["id","name","sku","price","image","stock","category","date","cost","unit","safety_stock","supplier_ids","expiry_date","last_purchase_date","spec","reference_price","reference_price_date","reference_price_name","shop_enabled"]);

  const lock = LockService.getScriptLock();
  lock.waitLock(10000);
  try {
    if (action === "add" || action === "update") {
      const payload = parsePurchasePayload_(p) || {};
      let po_id = String(payload.po_id || p.po_id || "").trim();
      if (!po_id) po_id = String(generateDailyDocNo_(purchasesSh, "po_id", new Date())).trim();
      const headers = getHeaders_(purchasesSh);
      const idx = headerIndexMap_(headers);
      const rows = purchasesSh.getDataRange().getValues();
      let foundRow = 0;
      let oldStockApplied = 0;
      const form_no = String(payload.form_no || "F-02-B-01-1").trim();
      const form_name = String(payload.form_name || "").trim() || getPurchaseFormInfoByCategory_(form_no).form_name;
      const items = Array.isArray(payload.items) ? payload.items : [];
      if (!items.length) return respond({ status: "error", message: "缺少 items" });
      const operator = String(payload.operator || p.operator || "").trim();
      const total = cleanDecimalInput_(payload.total || 0, 6);
      for (let r = 1; r < rows.length; r++) {
        if (String(rows[r][idx.po_id] || "").trim() === po_id) {
          foundRow = r + 1;
          oldStockApplied = Number(rows[r][idx.stock_applied] || 0) ? 1 : 0;
          break;
        }
      }

      const incomingSourceOrderId = String(payload.source_order_id || "").trim();
      const fallbackSourceOrderId = String(findOrderIdByLinkedPurchaseId_(ss, po_id) || "").trim();
      const linkedSourceOrderId = String(incomingSourceOrderId || (foundRow ? rows[foundRow - 1][idx.source_order_id] : "") || fallbackSourceOrderId || "").trim();
      const linkedAutoGenerated = Number(payload.auto_generated || (foundRow ? rows[foundRow - 1][idx.auto_generated] : 0) || (linkedSourceOrderId && fallbackSourceOrderId && linkedSourceOrderId === fallbackSourceOrderId ? 1 : 0) || 0) ? 1 : 0;
      const linkedOrderDate = linkedSourceOrderId ? getOrderDateById_(ss, linkedSourceOrderId) : "";
      const date = linkedOrderDate || payload.date || toISODateLocal_(new Date());
      const arrival_date = toISODateLocal_(payload.arrival_date || payload.receive_date || "");

      const supplierMap = buildSupplierNameMap_(ss);
      const prodVals = prodSh.getDataRange().getValues();
      const ph = prodVals[0].map(h => String(h || "").trim());
      const pidx = headerIndexMap_(ph);
      const byId = {};
      for (let r = 1; r < prodVals.length; r++) {
        byId[String(prodVals[r][pidx.id]).trim()] = { r: r, row: prodVals[r] };
      }

      let exactTotal = 0;
      const normalizedItems = items.map(it => {
        const pid = String(it.product_id || it.id || "").trim();
        if (!pid) throw new Error("品項資料不完整");
        const hit = byId[pid];
        if (!hit) throw new Error(`找不到商品 id=${pid}`);
        const qtyRaw = (it.qty_raw !== undefined && it.qty_raw !== null && String(it.qty_raw).trim() !== "") ? it.qty_raw : it.qty;
        const costRaw = (it.cost_raw !== undefined && it.cost_raw !== null && String(it.cost_raw).trim() !== "") ? it.cost_raw : (it.cost ?? hit.row[pidx.cost] ?? 0);
        const qty = cleanDecimalInput_(qtyRaw, 6);
        const cost = cleanDecimalInput_(costRaw, 6);
        if (!(qty > 0)) throw new Error("品項數量需大於 0");
        const supplierIds = parseSupplierIdsFromProductRow_(hit.row, pidx);
        let supplier_id = String(it.supplier_id || supplierIds[0] || "").trim();
        const supplier_name = String(it.supplier_name || supplierMap[supplier_id] || "").trim();
        exactTotal = addDecimalInput_(exactTotal, mulDecimalInput_(qtyRaw, costRaw, 6), 6);
        return {
          product_id: pid,
          product_name: String(it.product_name || hit.row[pidx.name] || ""),
          qty_raw: qtyRaw,
          cost_raw: costRaw,
          qty: qty,
          suggested_qty: cleanDecimalInput_(it.suggested_qty || qty, 6),
          customer_order_qty: (function(){
            var raw = String(it.customer_order_qty || "").trim();
            return raw === "" ? "" : cleanDecimalInput_(raw, 6);
          })(),
          cost: cost,
          supplier_id: supplier_id,
          supplier_name: supplier_name,
          expiry_date: toISODateLocal_(it.expiry_date || ""),
          unit: (pidx.unit !== undefined) ? String(it.unit || hit.row[pidx.unit] || "") : String(it.unit || ""),
          sku: (pidx.sku !== undefined) ? String(it.sku || hit.row[pidx.sku] || "") : String(it.sku || ""),
          spec: (pidx.spec !== undefined) ? String(it.spec || hit.row[pidx.spec] || "") : String(it.spec || ""),
          receive_date: String(it.receive_date || "").trim(),
          inspection_priority: String(it.inspection_priority || "").trim(),
          receipt_weight: String(it.receipt_weight || "").trim(),
          accept_weight: String(it.accept_weight || "").trim(),
          acceptance_result: String(it.acceptance_result || "").trim(),
          pesticide_result: String(it.pesticide_result || "").trim(),
          note: String(it.note || "").trim()
        };
      });

      const uniqSupIds = Array.from(new Set(normalizedItems.map(it => String(it.supplier_id || "").trim()).filter(Boolean)));
      const uniqSupNames = Array.from(new Set(normalizedItems.map(it => String(it.supplier_name || "").trim()).filter(Boolean)));
      const headerSupplierId = String(payload.supplier_id || ((uniqSupIds.length === 1) ? uniqSupIds[0] : "MULTI")).trim();
      const headerSupplierName = String(payload.supplier_name || ((uniqSupNames.length === 1) ? uniqSupNames[0] : "多供應商")).trim();
      const duplicatePoId = (!foundRow && action === "add") ? String(findRecentDuplicatePurchasePoId_(rows, idx, {
        po_id: String(payload.po_id || p.po_id || "").trim(),
        source_order_id: linkedSourceOrderId,
        operator: operator,
        date: date,
        arrival_date: arrival_date,
        form_no: form_no,
        supplier_id: headerSupplierId,
        items: normalizedItems
      }) || "").trim() : "";
      if (duplicatePoId) {
        return respond({ status: "ok", message: `採購驗收單已儲存：${duplicatePoId}`, po_id: duplicatePoId, deduped: 1 });
      }
      const applyStockNow = !!payload.apply_stock && !oldStockApplied;
      if (applyStockNow) {
        applyPurchaseStock_(ss, prodSh, po_id, normalizedItems, operator, date);
      }

      const rowObj = {
        po_id: po_id,
        date: date,
        arrival_date: arrival_date,
        form_no: form_no,
        form_name: form_name,
        supplier_id: headerSupplierId,
        supplier_name: headerSupplierName,
        total: (exactTotal || total),
        items: JSON.stringify(normalizedItems),
        expiry_date: "",
        status: applyStockNow ? "已入庫" : String(payload.status || (oldStockApplied ? "已入庫" : "待驗收")).trim(),
        operator: operator,
        created_at: foundRow ? (rows[foundRow - 1][idx.created_at] || new Date()) : new Date(),
        source_order_id: linkedSourceOrderId,
        stock_applied: applyStockNow || oldStockApplied ? 1 : 0,
        completed_at: (applyStockNow ? new Date() : (foundRow ? rows[foundRow - 1][idx.completed_at] : "")),
        auto_generated: linkedAutoGenerated ? 1 : 0
      };

      if (foundRow) {
        setSheetRowByObject_(purchasesSh, foundRow, rowObj);
      } else {
        appendObject_(purchasesSh, rowObj);
      }

      return respond({ status: "ok", message: applyStockNow ? `採購驗收單已完成：${po_id}` : `採購驗收單已儲存：${po_id}`, po_id: po_id });
    }

    if (action === "delete") {
      const po_id = String(p.po_id || p.id || "").trim();
      if (!po_id) return respond({ status: "error", message: "缺少 po_id" });
      const rows = purchasesSh.getDataRange().getValues();
      const headers = rows[0].map(h => String(h || "").trim());
      const idx = headerIndexMap_(headers);
      for (let r = 1; r < rows.length; r++) {
        if (String(rows[r][idx.po_id]).trim() === po_id) {
          purchasesSh.deleteRow(r + 1);
          return respond({ status: "ok", message: "採購驗收單已刪除（若已入庫，不會自動回滾庫存）" });
        }
      }
      return respond({ status: "error", message: "找不到進貨單" });
    }

    return respond({ status: "error", message: "未知 action" });
  } finally {
    try { lock.releaseLock(); } catch (e2) {}
  }
}

/**
 * 領貨單號：yyyyMMdd + 4位流水（例：202602090001）
 * - 會同時掃描 pickups 上可能的單號欄位（pickup_id / id / doc_no / docNo / pickupId）
 * - 避免「舊資料單號在別欄位」造成序號重複
 */
function generatePickupDocNo_(pickupsSh, dateObj){
  const d = dateObj instanceof Date ? dateObj : new Date();
  const prefix = Utilities.formatDate(d, "Asia/Taipei", "yyyyMMdd");

  const headers = getHeaders_(pickupsSh);
  const idx = headerIndexMap_(headers);

  const candidates = ["pickup_id","id","doc_no","docNo","pickupId"];
  const cols = candidates.map(k => idx[k]).filter(v => v !== undefined);

  if (!cols.length) {
    // 若真的沒有任何欄位，就先確保 pickup_id 存在（已在 ensurePickupsSheet_ 保障）
    return generateDailyDocNo_(pickupsSh, "pickup_id", d);
  }

  const lastRow = pickupsSh.getLastRow();
  let maxSeq = 0;

  if (lastRow >= 2) {
    // 逐欄掃描（避免抓整表太大）
    for (let c = 0; c < cols.length; c++){
      const col = cols[c] + 1;
      const vals = pickupsSh.getRange(2, col, lastRow - 1, 1).getValues();
      for (let i = 0; i < vals.length; i++){
        const s = String(vals[i][0] || "").trim();
        if (!s) continue;
        if (s.length >= 12 && s.indexOf(prefix) === 0) {
          const suf = parseInt(s.slice(8), 10);
          if (!isNaN(suf)) maxSeq = Math.max(maxSeq, suf);
        }
      }
    }
  }

  const next = maxSeq + 1;
  const suf4 = ("0000" + next).slice(-4);
  return prefix + suf4;
}

      const receiver = String(payload.receiver || "").trim();
      const note = String(payload.note || "").trim();
      const items = Array.isArray(payload.items) ? payload.items : [];
      const operator = String(payload.operator || p.operator || "").trim();

      if (!department) return respond({ status:"error", message:"缺少領用單位/門市（department）" });
      if (!items.length) return respond({ status:"error", message:"缺少 items" });

      // 領貨單號：yyyyMMdd + 4位流水（例：202602090001）
      const pickup_id = String(generatePickupDocNo_(pickupsSh, new Date())).trim();

      // products map
      const prodVals = prodSh.getDataRange().getValues();
      const ph = prodVals[0].map(h => String(h || "").trim());
      const pidx = headerIndexMap_(ph);
      const byId = {};
      for (let r = 1; r < prodVals.length; r++){
        byId[String(prodVals[r][pidx.id]).trim()] = { r: r, row: prodVals[r] };
      }

      const now = new Date();
      let total = 0;

      // 扣庫存 + 寫 ledger
      for (let i = 0; i < items.length; i++){
        const it = items[i];
        const pid = String(it.product_id || it.id || "").trim();
        const qty = Number(it.qty || it.quantity || 0);
        const unitCost = (it.cost !== undefined && it.cost !== "") ? Number(it.cost) : Number(it.unit_cost || 0);

        if (!pid || !qty || qty <= 0) return respond({ status:"error", message:"品項資料不完整（product_id/qty）" });

        const hit = byId[pid];
        if (!hit) return respond({ status:"error", message:`找不到商品 id=${pid}` });

        const current = Number(hit.row[pidx.stock] || 0);
        const next = current - qty;
        if (next < 0) return respond({ status:"error", message:`庫存不足：${hit.row[pidx.name]} current=${current}, qty=${qty}` });

        // update stock
        prodSh.getRange(hit.r + 1, pidx.stock + 1).setValue(next);

        const pname = String(hit.row[pidx.name] || "");
        const target = receiver ? `${department} / ${receiver}` : department;

        writeLedger_(ss, {
          ledger_id: generateId_("LG"),
          product_id: pid,
          product_name: pname,
                    sku: (pidx.sku !== undefined) ? String(hit.row[pidx.sku] || "") : "",
          unit: (pidx.unit !== undefined) ? String(hit.row[pidx.unit] || "") : "",
change: -qty,                // signed change: out = negative
          direction: "OUT",
          reason: "pickup",
          ref_id: pickup_id,
          operator: operator,
          date: toLocalDateTime_(now),
          cost: (pidx.cost !== undefined) ? (Number(hit.row[pidx.cost] || 0) || "") : "",
          target: target
        });

        // total cost (optional)
        if (!isNaN(unitCost) && unitCost > 0) total += unitCost * qty;
      }

      // 寫 pickups（items 存 JSON）
      // 同時寫入 pickup_id 與 id（若有）以兼容舊表頭（appendObject_ 只會寫入表頭存在的欄位）
      appendObject_(pickupsSh, {
        pickup_id: pickup_id,
        id: pickup_id,
        doc_no: pickup_id,
        date: payload.date || toROCDateString_(new Date()),
        department: department,
        receiver: receiver,
        total: total,
        items: JSON.stringify(items),
        note: note,
        operator: operator,
        created_at: now
      });

      return respond({ status:"ok", message:`領貨完成：${pickup_id}`, pickup_id: pickup_id });
    }

    if (action === "delete"){
      const pickup_id = String(p.pickup_id || p.id || "").trim();
      if (!pickup_id) return respond({ status:"error", message:"缺少 pickup_id" });

      const rows = pickupsSh.getDataRange().getValues();
      const headers = rows[0].map(h => String(h || "").trim());
      const idx = headerIndexMap_(headers);

      const candidates = ["pickup_id","id","doc_no","docNo","pickupId"];
      const cols = candidates.map(k => idx[k]).filter(v => v !== undefined);
      const col = cols.length ? cols[0] : 0;

      for (let r = 1; r < rows.length; r++){
        if (String(rows[r][col] || "").trim() === pickup_id){
          pickupsSh.deleteRow(r + 1);
          return respond({ status:"ok", message:"領貨單已刪除（未回滾庫存）" });
        }
      }
      return respond({ status:"error", message:"找不到領貨單" });
    }

    return respond({ status:"error", message:"未知 action" });
  } finally {
    try { lock.releaseLock(); } catch(e2) {}
  }
}

function apiManageOrder_(p, respond) {
  const action = String(p.action || "").trim();
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sh = ensureSheet_(ss, "Orders", orderHeaders_());
  const prodSh = ensureSheet_(ss, "Products", ["id","name","sku","price","image","stock","category","date","cost","unit","safety_stock","supplier_ids","expiry_date","last_purchase_date","spec","reference_price","reference_price_date","reference_price_name","shop_enabled"]);
  const rows = sh.getDataRange().getValues();
  const headers = rows[0].map(h => String(h || "").trim());
  const idx = headerIndexMap_(headers);

  const order_id = String(p.order_id || p.id || "").trim();
  if (!order_id) return respond({ status: "error", message: "缺少 order_id" });

  if (action === "update") {
    const status = String(p.status || "").trim();
    if (!status) return respond({ status: "error", message: "缺少 status" });
    for (let r = 1; r < rows.length; r++) {
      if (String(rows[r][idx.order_id]).trim() === order_id) {
        const source = String(rows[r][idx.source] || "").trim();
        const linkedPurchaseId = String(rows[r][idx.linked_purchase_id] || "").trim();
        const stockApplied = Number(rows[r][idx.stock_applied] || 0) ? 1 : 0;
        let items = rows[r][idx.items];
        if (typeof items === "string" && items) {
          try { items = JSON.parse(items); } catch (e) { items = []; }
        }
        if (!Array.isArray(items)) items = [];

        const incomingOperator = String(p.operator || "").trim();
        const existingOperator = String(rows[r][idx.operator] || "").trim();
        const shipmentOperator = incomingOperator || existingOperator;

        if ((status === "已出貨" || status === "已完成") && !stockApplied) {
          applyOrderShipment_(ss, prodSh, order_id, items, String(rows[r][idx.name] || "").trim(), shipmentOperator);
          if (idx.stock_applied !== undefined) sh.getRange(r + 1, idx.stock_applied + 1).setValue(1);
        }
        if (idx.status !== undefined) sh.getRange(r + 1, idx.status + 1).setValue(status);
        if (idx.operator !== undefined && incomingOperator) {
          if (status === "已出貨" || (!existingOperator && status === "已完成")) {
            sh.getRange(r + 1, idx.operator + 1).setValue(incomingOperator);
          }
        }
        return respond({ status: "ok", message: "訂單已更新" });
      }
    }
    return respond({ status: "error", message: "找不到訂單" });
  }

  if (action === "delete") {
    for (let r = 1; r < rows.length; r++) {
      if (String(rows[r][idx.order_id]).trim() === order_id) {
        sh.deleteRow(r + 1);
        return respond({ status: "ok", message: "訂單已刪除（未回滾庫存）" });
      }
    }
    return respond({ status: "error", message: "找不到訂單" });
  }

  return respond({ status: "error", message: "未知 action" });
}

function apiProductLedger_(p, respond) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sh = ensureLedgerSheet_(ss);
  const pid = String(p.product_id || p.id || "").trim();
  const sku = String(p.sku || "").trim();
  if (!sku && !pid) return respond({ status: "error", message: "缺少 sku 或 product_id" });

  const limit = Math.max(1, Math.min(1000, Number(p.limit || 300)));

  const from = String(p.from || "").trim(); // yyyy-MM-dd (optional)
  const to = String(p.to || "").trim();     // yyyy-MM-dd (optional)

  const list = readObjects_(sh).filter(x => {
    const xs = String(x.sku || "").trim();
    if (sku && xs && xs === sku) return true;
    return pid ? (String(x.product_id || x.id || "").trim() === pid) : false;
  });

  // 日期篩選（用 date，若無則不篩）
  let rows = list;
  if (from) {
    const ft = Date.parse(from + "T00:00:00");
    rows = rows.filter(x => {
      const t = Date.parse(String(x.date || "").replace(" ", "T"));
      return isNaN(t) ? true : (t >= ft);
    });
  }
  if (to) {
    const tt = Date.parse(to + "T23:59:59");
    rows = rows.filter(x => {
      const t = Date.parse(String(x.date || "").replace(" ", "T"));
      return isNaN(t) ? true : (t <= tt);
    });
  }

  // 依時間倒序（date）
  rows.sort((a,b) => {
    const ta = Date.parse(String(a.date || "").replace(" ", "T"));
    const tb = Date.parse(String(b.date || "").replace(" ", "T"));
    return (isNaN(tb)?0:tb) - (isNaN(ta)?0:ta);
  });

  return respond({ status: "ok", data: rows.slice(0, limit) });
}


function pickHistoryReferencePrice_(row) {
  const ref = parseLooseNumber_(row && row.reference_price);
  if (ref !== '' && ref !== null && ref !== undefined && !isNaN(Number(ref))) return Number(ref);
  const upper = parseLooseNumber_(row && row.upper_price);
  if (upper !== '' && upper !== null && upper !== undefined && !isNaN(Number(upper))) return Number(upper);
  const middle = parseLooseNumber_(row && row.middle_price);
  if (middle !== '' && middle !== null && middle !== undefined && !isNaN(Number(middle))) return Number(middle);
  const lower = parseLooseNumber_(row && row.lower_price);
  if (lower !== '' && lower !== null && lower !== undefined && !isNaN(Number(lower))) return Number(lower);
  return '';
}

function apiProductMarketHistory_(p, respond) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const histSh = ensureSheet_(ss, 'MarketPriceHistory', MARKET_PRICE_HISTORY_HEADERS_);
  const prodSh = ensureSheet_(ss, 'Products', ['id','name','sku','price','image','stock','category','date','cost','unit','safety_stock','supplier_ids','expiry_date','last_purchase_date','spec','reference_price','reference_price_date','reference_price_name','shop_enabled']);

  const pid = String(p.product_id || p.id || '').trim();
  const sku = String(p.sku || '').trim();
  if (!pid && !sku) return respond({ status: 'error', message: '缺少 sku 或 product_id' });

  const limit = Math.max(1, Math.min(2000, Number(p.limit || 1000)));
  const from = normalizeLooseDate_(String(p.from || '').trim());
  const to = normalizeLooseDate_(String(p.to || '').trim());

  const products = readObjects_(prodSh);
  let product = null;
  for (let i = 0; i < products.length; i++) {
    const row = products[i] || {};
    const rowId = String(row.id || '').trim();
    const rowSku = String(row.sku || row.part_no || row.code || '').trim();
    if (pid && rowId === pid) { product = row; break; }
    if (!product && sku && rowSku && rowSku === sku) product = row;
  }

  const productId = pid || String(product && product.id || '').trim();
  const productName = String(product && product.name || '').trim();
  const productSpec = String(product && product.spec || '').trim();

  const rows = readObjects_(histSh).map(r => ({
    history_id: String(r.history_id || '').trim(),
    batch_id: String(r.batch_id || '').trim(),
    sync_time: String(r.sync_time || '').trim(),
    sync_source: String(r.sync_source || '').trim(),
    product_id: String(r.product_id || '').trim(),
    product_name: String(r.product_name || '').trim(),
    product_spec: String(r.product_spec || '').trim(),
    reference_price: pickHistoryReferencePrice_(r),
    reference_price_date: normalizeLooseDate_(r.reference_price_date || ''),
    reference_price_name: String(r.reference_price_name || '').trim(),
    market: normalizeMarketLabel_(r.market || ''),
    market_label: String(r.market_label || '').trim() || marketDisplayLabel_(normalizeMarketLabel_(r.market || '')),
    market_date: normalizeLooseDate_(r.market_date || ''),
    raw_name: String(r.raw_name || '').trim(),
    raw_spec: String(r.raw_spec || '').trim(),
    upper_price: parseLooseNumber_(r.upper_price),
    middle_price: parseLooseNumber_(r.middle_price),
    lower_price: parseLooseNumber_(r.lower_price),
    matched: String(r.matched || '').trim()
  })).filter(r => {
    if (!r.market_date) return false;
    if (productId && r.product_id === productId) return true;
    if (!productId && productName && r.product_name === productName && (!productSpec || r.product_spec === productSpec)) return true;
    return false;
  }).filter(r => {
    if (from && r.market_date < from) return false;
    if (to && r.market_date > to) return false;
    return true;
  });

  rows.sort((a, b) => {
    const dateCmp = String(b.market_date || '').localeCompare(String(a.market_date || ''));
    if (dateCmp !== 0) return dateCmp;
    const syncCmp = String(b.sync_time || '').localeCompare(String(a.sync_time || ''));
    if (syncCmp !== 0) return syncCmp;
    return String(b.history_id || '').localeCompare(String(a.history_id || ''));
  });

  const deduped = [];
  const seenDate = {};
  rows.forEach(r => {
    const key = String(r.market_date || '').trim();
    if (!key) return;
    if (seenDate[key]) return;
    seenDate[key] = true;
    deduped.push(r);
  });

  return respond({
    status: 'ok',
    product_id: productId,
    product_name: productName,
    product_spec: productSpec,
    data: deduped.slice(0, limit)
  });
}

function apiStockAdjust_(p, respond) {
  const product_id = String(p.product_id || "").trim();
  const delta = Number(p.delta || 0);
  const reason = String(p.reason || "adjust").trim();
  const operator = String(p.operator || "").trim();
  if (!product_id) return respond({ status: "error", message: "缺少 product_id" });
  if (!delta || isNaN(delta)) return respond({ status: "error", message: "delta 必須是非 0 數字" });

  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const prodSh = ensureSheet_(ss, "Products", ["id","name","sku","price","image","stock","category","date","cost","unit","safety_stock","supplier_ids","expiry_date","last_purchase_date","spec","reference_price","reference_price_date","reference_price_name","shop_enabled"]);

  const lock = LockService.getScriptLock();
  lock.waitLock(10000);
  try {
    const vals = prodSh.getDataRange().getValues();
    const headers = vals[0].map(h => String(h || "").trim());
    const idx = headerIndexMap_(headers);

    let row = -1;
    for (let r = 1; r < vals.length; r++) {
      if (String(vals[r][idx.id]).trim() === product_id) { row = r; break; }
    }
    if (row === -1) return respond({ status: "error", message: "找不到商品" });

    const current = Number(vals[row][idx.stock] || 0);
    const next = current + delta;
    if (next < 0) return respond({ status: "error", message: `庫存不足 current=${current}, delta=${delta}` });

    prodSh.getRange(row + 1, idx.stock + 1).setValue(next);

    const pname = String(vals[row][idx.name] || "");
    const now = new Date();
    writeLedger_(ss, {
      ledger_id: generateId_("LG"),
      product_id: product_id,
      product_name: pname,
            sku: (idx.sku !== undefined) ? String(vals[row][idx.sku] || "") : "",
      unit: (idx.unit !== undefined) ? String(vals[row][idx.unit] || "") : "",
change: delta,                 // signed change：增加為正、減少為負
      direction: "ADJ",
      reason: "adjust",
      ref_id: "ADJ-" + Utilities.formatDate(now, "Asia/Taipei", "yyyyMMddHHmmss"),
      operator: operator,
      date: toLocalDateTime_(now),
      cost: (idx.cost !== undefined) ? (Number(vals[row][idx.cost] || 0) || "") : "",
      target: reason                 // 調整原因放在 target（便於前台顯示）
    });

    return respond({ status: "ok", message: "庫存已調整", before: current, after: next });
  } finally {
    try { lock.releaseLock(); } catch (e2) {}
  }
}

    const u = String(rows[i][idx.username] || "").trim();
    if (u === username) {
      return respond({ status: "error", message: "帳號已存在" });
    }
  }

  // 產生新 id（取目前最大值 + 1）
  let newId = 1;
  if (rows.length > 1 && idx.id !== undefined) {
    let maxId = 0;
    for (let i = 1; i < rows.length; i++) {
      const v = Number(String(rows[i][idx.id] || "").trim());
      if (!isNaN(v)) maxId = Math.max(maxId, v);
    }
    newId = maxId + 1;
  }

  appendObject_(sh, {
    id: newId,
    name: name,
    username: username,
    password: password,
    created_at: new Date(),
    role: "user"
  });

  return respond({ status: "ok", message: "註冊成功", id: newId });
}

function apiOrderCreate_(p, respond) {

  const member_id = String(p.member_id || "").trim();

    const po_id = String(generateDailyDocNo_(purchasesSh, "po_id", now)).trim();

      if (!firstCategory && category) firstCategory = category;

      const safety = cleanDecimalInput_(row[pidx.safety_stock] || 0, 6);
      const cost = cleanDecimalInput_(row[pidx.cost] || 0, 6);

      });

        qty: qty,
        qty_raw: qty,
        suggested_qty: suggestedQty,
        customer_order_qty: qty,

  }

function apiGetOrder_(p, respond) {
  const order_id = String(p.order_id || p.id || "").trim();
  if (!order_id) return respond({ status: "error", message: "缺少 order_id" });

  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sh = ensureSheet_(ss, "Orders", orderHeaders_());

  const list = readObjects_(sh);
  const hit = list.find(o => String(o.order_id || "").trim() === order_id);
  if (!hit) return respond({ status: "error", message: "查無此訂單" });

  let items = hit.items;
  if (items && typeof items === "string") {
    try { items = JSON.parse(items); } catch (e) {}
  }
  hit.items = formatOrderItemsForDisplay_(items);
  hit.date = toISODateLocal_(hit.date);
  hit.shipping_date = toISODateLocal_(hit.shipping_date || "");

  return respond({ status: "ok", order: hit });
}

function formatOrderItemsForDisplay_(items) {
  if (!Array.isArray(items) || items.length === 0) return "";
  return items.map(it => {
    const name = String(it.name || it.product_name || it.product || "");
    const qty = Number(it.qty || it.quantity || it.Quantity || 0);
    const price = Number(it.price || it.UnitPrice || 0);
    const subtotal = (!isNaN(Number(it.subtotal))) ? Number(it.subtotal) : (qty * price);
    return `${name} x${qty} = $${subtotal}`;
  }).join("<br>");
}

function findMenuHeaderRow_(sheet){
  const maxRows = Math.min(Math.max(sheet.getLastRow() || 0, 3), 12);
  const maxCols = Math.max(sheet.getLastColumn() || 0, menuSheetHeaders_().length);
  if (!maxRows || !maxCols) return 0;
  const values = sheet.getRange(1, 1, maxRows, maxCols).getDisplayValues();
  for (let i = 0; i < values.length; i++) {
    if (looksLikeMenuHeaderRow_(values[i])) return i + 1;
  }
  return 0;
}

function normalizeMenuDishMatchText_(v){
  return String(v || "").replace(/[\s　]+/g, "").trim().toLowerCase();
}

function loadMenuProductLookup_(ss){
  const prodSh = ensureSheet_(ss, "Products", ["id","name","sku","price","image","stock","category","date","cost","unit","safety_stock","supplier_ids","expiry_date","last_purchase_date","spec","reference_price","reference_price_date","reference_price_name","shop_enabled"]);
  const products = readObjects_(prodSh).map(o => ({
    id: String(o.id || "").trim(),
    sku: String(o.sku || o.id || "").trim(),
    name: String(o.name || "").trim(),
    spec: String(o.spec || "").trim(),
    price: Number(o.price || 0),
    image: String(o.image || "").trim(),
    stock: Number(o.stock || 0),
    unit: String(o.unit || "").trim(),
    shop_enabled: normalizeShopEnabled_(o.shop_enabled, true)
  }));

  const byId = {};
  const byName = {};
  const byNameSpec = {};
  products.forEach(p => {
    if (!p.id || !p.name) return;
    const nameKey = normalizeMenuDishMatchText_(p.name);
    const specKey = normalizeMenuDishMatchText_(p.spec);
    byId[p.id] = p;
    if (nameKey) {
      if (!byName[nameKey]) byName[nameKey] = [];
      byName[nameKey].push(p);
      const exactKey = nameKey + "__" + specKey;
      if (specKey && (!byNameSpec[exactKey] || normalizeShopEnabled_(byNameSpec[exactKey].shop_enabled, true) === "0")) {
        byNameSpec[exactKey] = p;
      }
    }
  });
  return { byId: byId, byName: byName, byNameSpec: byNameSpec };
}

function resolveMenuIngredientProduct_(row, productLookup){
  const productId = String(row.product_id || "").trim();
  if (productId && productLookup.byId[productId]) return productLookup.byId[productId];

  const productName = String(row.product_name || "").trim();
  const spec = String(row.spec || "").trim();
  const nameKey = normalizeMenuDishMatchText_(productName);
  const specKey = normalizeMenuDishMatchText_(spec);

  if (nameKey && specKey) {
    const hit = productLookup.byNameSpec[nameKey + "__" + specKey];
    if (hit) return hit;
  }

  if (nameKey && productLookup.byName[nameKey] && productLookup.byName[nameKey].length) {
    const visible = productLookup.byName[nameKey].find(p => normalizeShopEnabled_(p.shop_enabled, true) !== "0");
    return visible || productLookup.byName[nameKey][0];
  }
  return null;
}

function loadMenuIngredientMap_(ss){
  const sh = ensureMenuIngredientSheet_(ss);
  const rows = readObjects_(sh);
  const productLookup = loadMenuProductLookup_(ss);
  const map = {};

  rows.forEach((row, index) => {
    const dishName = String(row.dish_name || "").trim();
    if (!dishName) return;
    if (normalizeShopEnabled_(row.enabled, true) === "0") return;

    const product = resolveMenuIngredientProduct_(row, productLookup);
    if (!product || normalizeShopEnabled_(product.shop_enabled, true) === "0") return;

    const key = normalizeMenuDishMatchText_(dishName);
    if (!key) return;

    if (!map[key]) map[key] = [];
    map[key].push({
      dish_name: dishName,
      sort_no: Number(row.sort_no || index + 1) || (index + 1),
      default_qty: Math.max(1, parseInt(String(row.default_qty || "1"), 10) || 1),
      note: String(row.note || "").trim(),
      product: product
    });
  });

  Object.keys(map).forEach(key => {
    const seen = {};
    map[key] = map[key]
      .sort((a, b) => {
        const sa = Number(a.sort_no || 0);
        const sb = Number(b.sort_no || 0);
        if (sa !== sb) return sa - sb;
        return String(a.product.name || "").localeCompare(String(b.product.name || ""), "zh-Hant");
      })
      .filter(row => {
        const pid = String((row.product && row.product.id) || "").trim();
        if (!pid) return false;
        if (seen[pid]) return false;
        seen[pid] = true;
        return true;
      });
  });

  return { sheet: sh, map: map };
}

function normalizeMenuBomNumber_(v, fallback){
  const n = Number(String(v == null ? "" : v).replace(/%/g, "").trim());
  return Number.isFinite(n) ? n : (fallback || 0);
}

function normalizeMenuBomLossRate_(v){
  let n = normalizeMenuBomNumber_(v, 0);
  if (n > 1) n = n / 100;
  if (n < 0) n = 0;
  if (n >= 1) n = 0.9999;
  return n;
}

function normalizeMenuBomQtyUnit_(row, product){
  const custom = String((row && row.qty_unit) || "").trim();
  if (custom) return custom;
  const productUnit = String((product && product.unit) || "").trim();
  if (productUnit) return productUnit;
  return "g";
}

function loadMenuBomMap_(ss){
  const sh = ensureMenuBomSheet_(ss);
  const rows = readObjects_(sh);
  const productLookup = loadMenuProductLookup_(ss);
  const map = {};

  rows.forEach((row, index) => {
    const dishName = String(row.dish_name || "").trim();
    if (!dishName) return;
    if (normalizeShopEnabled_(row.enabled, true) === "0") return;

    const resolvedProduct = resolveMenuIngredientProduct_(row, productLookup);
    const product = resolvedProduct ? resolvedProduct : {
      id: "",
      sku: "",
      name: String(row.product_name || "").trim(),
      spec: String(row.spec || "").trim(),
      price: 0,
      image: "",
      stock: 0,
      unit: String(row.qty_unit || "").trim(),
      shop_enabled: "1"
    };
    if (!String(product.name || "").trim()) return;
    if (resolvedProduct && normalizeShopEnabled_(product.shop_enabled, true) === "0") return;

    const perPersonQty = Math.max(0, normalizeMenuBomNumber_(row.per_person_qty, 0));
    if (!(perPersonQty > 0)) return;

    const key = normalizeMenuDishMatchText_(dishName);
    if (!key) return;

    if (!map[key]) map[key] = [];
    map[key].push({
      dish_name: dishName,
      sort_no: Number(row.sort_no || index + 1) || (index + 1),
      category: String(row.category || "").trim(),
      per_person_qty: perPersonQty,
      loss_rate: normalizeMenuBomLossRate_(row.loss_rate),
      qty_unit: normalizeMenuBomQtyUnit_(row, product),
      note: String(row.note || "").trim(),
      product: product,
      has_linked_product: !!(resolvedProduct && String(resolvedProduct.id || "").trim())
    });
  });

  Object.keys(map).forEach(key => {
    const seen = {};
    map[key] = map[key]
      .sort((a, b) => {
        const sa = Number(a.sort_no || 0);
        const sb = Number(b.sort_no || 0);
        if (sa !== sb) return sa - sb;
        return String((a.product && a.product.name) || "").localeCompare(String((b.product && b.product.name) || ""), "zh-Hant");
      })
      .filter(row => {
        const pid = String((row.product && row.product.id) || "").trim();
        const pname = normalizeMenuDishMatchText_((row.product && row.product.name) || "");
        const pspec = normalizeMenuDishMatchText_((row.product && row.product.spec) || "");
        const q = String(row.qty_unit || "").trim();
        const uniq = (pid ? pid : (pname + "__" + pspec)) + "__" + q;
        if (!uniq || /^__+$/.test(uniq)) return false;
        if (seen[uniq]) return false;
        seen[uniq] = true;
        return true;
      });
  });

  return { sheet: sh, map: map };
}

function menuHeaderIndexMap_(headers){
  const normalized = (headers || []).map(normalizeMenuHeaderText_);
  const idx = {
    date: -1,
    weekday: -1,
    lunar: -1,
    meal: -1,
    main: -1,
    side: -1,
    light: -1,
    dark: -1,
    root: -1
  };
  normalized.forEach((h, i) => {
    if (h === "日期" && idx.date < 0) idx.date = i;
    else if (h === "星期" && idx.weekday < 0) idx.weekday = i;
    else if (h === "農曆" && idx.lunar < 0) idx.lunar = i;
    else if (h === "餐別" && idx.meal < 0) idx.meal = i;
    else if (h.indexOf("主菜") >= 0 && idx.main < 0) idx.main = i;
    else if (h.indexOf("小配菜") >= 0 && idx.side < 0) idx.side = i;
    else if (h.indexOf("淺色蔬菜") >= 0 && idx.light < 0) idx.light = i;
    else if (h.indexOf("深色蔬菜") >= 0 && idx.dark < 0) idx.dark = i;
    else if (h.indexOf("根莖瓜果") >= 0 && idx.root < 0) idx.root = i;
  });
  return idx;
}

function parseMenuDateToISO_(v){
  if (!v && v !== 0) return "";
  if (Object.prototype.toString.call(v) === "[object Date]" && !isNaN(v.getTime())) {
    return Utilities.formatDate(v, "Asia/Taipei", "yyyy-MM-dd");
  }
  const s = String(v).trim();
  if (!s) return "";

  if (/^\d{4}-\d{1,2}-\d{1,2}/.test(s)) {
    const parts = s.split(/[T\s]/)[0].split("-");
    return `${parts[0]}-${("0"+parts[1]).slice(-2)}-${("0"+parts[2]).slice(-2)}`;
  }
  if (/^\d{4}\/\d{1,2}\/\d{1,2}/.test(s)) {
    const parts = s.split(/[T\s]/)[0].split("/");
    return `${parts[0]}-${("0"+parts[1]).slice(-2)}-${("0"+parts[2]).slice(-2)}`;
  }
  if (/^\d{2,3}[\/\.\-]\d{1,2}[\/\.\-]\d{1,2}/.test(s) && !/^\d{4}[\/\.\-]/.test(s)) {
    const parts = s.split(/[T\s]/)[0].split(/[\/\.\-]/);
    const year = parseInt(parts[0], 10) + 1911;
    return `${year}-${("0"+parts[1]).slice(-2)}-${("0"+parts[2]).slice(-2)}`;
  }

  const dt = new Date(s);
  if (!isNaN(dt.getTime())) return Utilities.formatDate(dt, "Asia/Taipei", "yyyy-MM-dd");
  return "";
}

function normalizeMealKey_(v){
  const s = String(v || "").trim();
  if (!s) return "";
  if (s.indexOf("午") >= 0) return "lunch";
  if (s.indexOf("晚") >= 0) return "dinner";
  return s;
}

function mealLabelFromKey_(key){
  if (key === "lunch") return "午餐";
  if (key === "dinner") return "晚餐";
  return String(key || "").trim();
}

function buildDateLabelZh_(iso){
  const dt = new Date(iso + "T00:00:00");
  if (isNaN(dt.getTime())) return iso;
  const mm = ("0" + (dt.getMonth() + 1)).slice(-2);
  const dd = ("0" + dt.getDate()).slice(-2);
  return `${mm}月${dd}日`;
}

function weekdayLabelZh_(iso){
  const dt = new Date(iso + "T00:00:00");
  if (isNaN(dt.getTime())) return "";
  return ["週日","週一","週二","週三","週四","週五","週六"][dt.getDay()] || "";
}

function startOfWeekMonday_(baseDate){
  const safe = (Object.prototype.toString.call(baseDate) === "[object Date]" && !isNaN(baseDate.getTime())) ? baseDate : new Date();
  const y = Number(Utilities.formatDate(safe, "Asia/Taipei", "yyyy"));
  const m = Number(Utilities.formatDate(safe, "Asia/Taipei", "MM"));
  const d = Number(Utilities.formatDate(safe, "Asia/Taipei", "dd"));
  const local = new Date(y, m - 1, d, 0, 0, 0, 0);
  const dow = local.getDay();
  const diff = (dow === 0 ? -6 : 1 - dow);
  local.setDate(local.getDate() + diff);
  local.setHours(0, 0, 0, 0);
  return local;
}

function addDaysLocal_(dateObj, days){
  const dt = new Date(dateObj.getTime());
  dt.setDate(dt.getDate() + Number(days || 0));
  dt.setHours(0, 0, 0, 0);
  return dt;
}

function parseMenuRows_(sheet){
  const headerRow = findMenuHeaderRow_(sheet);
  if (!headerRow) return [];
  const lastRow = sheet.getLastRow();
  const lastCol = Math.max(sheet.getLastColumn() || 0, menuSheetHeaders_().length);
  if (lastRow <= headerRow) return [];

  const rawValues = sheet.getRange(headerRow, 1, lastRow - headerRow + 1, lastCol).getValues();
  const displayValues = sheet.getRange(headerRow, 1, lastRow - headerRow + 1, lastCol).getDisplayValues();
  const headers = displayValues[0] || [];
  const idx = menuHeaderIndexMap_(headers);

  let currentDate = "";
  let currentLunar = "";
  const out = [];

  for (let r = 1; r < rawValues.length; r++) {
    const rawRow = rawValues[r] || [];
    const displayRow = displayValues[r] || [];

    const dateRaw = idx.date >= 0 ? rawRow[idx.date] : "";
    const dateDisplay = idx.date >= 0 ? displayRow[idx.date] : "";
    const parsedDate = parseMenuDateToISO_(dateRaw) || parseMenuDateToISO_(dateDisplay);
    if (parsedDate) currentDate = parsedDate;

    const lunar = idx.lunar >= 0 ? String(displayRow[idx.lunar] || rawRow[idx.lunar] || "").trim() : "";
    if (lunar) currentLunar = lunar;

    const mealValue = idx.meal >= 0 ? String(displayRow[idx.meal] || rawRow[idx.meal] || "").trim() : "";
    const mealKey = normalizeMealKey_(mealValue);

    const item = {
      date: currentDate,
      lunar: lunar || currentLunar,
      meal_key: mealKey,
      meal_label: mealLabelFromKey_(mealKey || mealValue),
      main: idx.main >= 0 ? String(displayRow[idx.main] || rawRow[idx.main] || "").trim() : "",
      side: idx.side >= 0 ? String(displayRow[idx.side] || rawRow[idx.side] || "").trim() : "",
      light_veg: idx.light >= 0 ? String(displayRow[idx.light] || rawRow[idx.light] || "").trim() : "",
      dark_veg: idx.dark >= 0 ? String(displayRow[idx.dark] || rawRow[idx.dark] || "").trim() : "",
      root_veg: idx.root >= 0 ? String(displayRow[idx.root] || rawRow[idx.root] || "").trim() : ""
    };

    if (!item.date || !item.meal_key) continue;
    if (!item.main && !item.side && !item.light_veg && !item.dark_veg && !item.root_veg) continue;

    out.push(item);
  }

  return out;
}

function menuProcurementAggKey_(product, row){
  const productId = String((product && product.id) || "").trim();
  const nameKey = normalizeMenuDishMatchText_((product && product.name) || (row && row.product_name) || "");
  const specKey = normalizeMenuDishMatchText_((product && product.spec) || (row && row.spec) || "");
  const qtyUnit = String(((row && row.qty_unit) || (product && product.unit) || "g")).trim() || "g";
  if (productId) return [productId, qtyUnit].join("__");
  if (nameKey || specKey) return [nameKey || "item", specKey || "na", qtyUnit].join("__");
  return "";
}

function buildMenuDayPayload_(day, ingredientData, bomData){
  const groups = [];
  const unmapped = [];
  const procurementMap = {};
  const procurementUnmapped = [];
  const mealDefs = menuMealDefs_();
  const dishDefs = menuDishDefs_();

  mealDefs.forEach(mealInfo => {
    const meal = day.meals[mealInfo.key] || {};
    dishDefs.forEach(dishInfo => {
      const dishName = String(meal[dishInfo.key] || "").trim();
      if (!dishName) return;
      const mapKey = normalizeMenuDishMatchText_(dishName);
      const rows = (ingredientData.map[mapKey] || []).map(row => ({
        product_id: String((row.product && row.product.id) || "").trim(),
        sku: String((row.product && row.product.sku) || (row.product && row.product.id) || "").trim(),
        product_name: String((row.product && row.product.name) || "").trim(),
        spec: String((row.product && row.product.spec) || "").trim(),
        price: Number((row.product && row.product.price) || 0),
        unit: String((row.product && row.product.unit) || "").trim(),
        stock: Number((row.product && row.product.stock) || 0),
        image: String((row.product && row.product.image) || "").trim(),
        default_qty: Math.max(1, Number(row.default_qty || 1) || 1),
        note: String(row.note || "").trim()
      }));

      if (!rows.length) {
        unmapped.push({
          meal_key: mealInfo.key,
          meal_label: mealInfo.label,
          dish_key: dishInfo.key,
          dish_label: dishInfo.label,
          dish_name: dishName
        });
      } else {
        groups.push({
          meal_key: mealInfo.key,
          meal_label: mealInfo.label,
          dish_key: dishInfo.key,
          dish_label: dishInfo.label,
          dish_name: dishName,
          items: rows
        });
      }

      const bomRows = bomData.map[mapKey] || [];
      if (!bomRows.length) {
        procurementUnmapped.push({
          meal_key: mealInfo.key,
          meal_label: mealInfo.label,
          dish_key: dishInfo.key,
          dish_label: dishInfo.label,
          dish_name: dishName
        });
        return;
      }

      bomRows.forEach(row => {
        const product = row.product || {};
        const qtyUnit = String(row.qty_unit || product.unit || "g").trim() || "g";
        const aggKey = menuProcurementAggKey_(product, row);
        if (!aggKey) return;
        if (!procurementMap[aggKey]) {
          procurementMap[aggKey] = {
            product_id: String(product.id || "").trim(),
            sku: String(product.sku || product.id || "").trim(),
            product_name: String(product.name || row.product_name || "").trim(),
            spec: String(product.spec || row.spec || "").trim(),
            price: Number(product.price || 0),
            unit: String(product.unit || "").trim(),
            stock: Number(product.stock || 0),
            image: String(product.image || "").trim(),
            category: String(row.category || dishInfo.label || "").trim(),
            qty_unit: qtyUnit,
            per_person_net: 0,
            per_person_loss: 0,
            per_person_gross: 0,
            source_dishes: [],
            note_list: [],
            has_linked_product: !!String(product.id || "").trim()
          };
        }
        const agg = procurementMap[aggKey];
        const net = Math.max(0, Number(row.per_person_qty || 0) || 0);
        const rate = normalizeMenuBomLossRate_(row.loss_rate);
        const gross = rate >= 1 ? net : (net / (1 - rate));
        agg.per_person_net += net;
        agg.per_person_gross += gross;
        agg.per_person_loss += (gross - net);
        const sourceLabel = mealInfo.label + "｜" + dishName;
        if (agg.source_dishes.indexOf(sourceLabel) < 0) agg.source_dishes.push(sourceLabel);
        const note = String(row.note || "").trim();
        if (note && agg.note_list.indexOf(note) < 0) agg.note_list.push(note);
      });
    });
  });

  day.ingredient_groups = groups;
  day.unmapped_dishes = unmapped;
  day.procurement_unmapped_dishes = procurementUnmapped;
  day.procurement_items = Object.keys(procurementMap).map(key => procurementMap[key]).sort((a, b) => {
    const ca = String(a.category || "");
    const cb = String(b.category || "");
    if (ca !== cb) return ca.localeCompare(cb, "zh-Hant");
    return String(a.product_name || "").localeCompare(String(b.product_name || ""), "zh-Hant");
  }).map(row => ({
    product_id: row.product_id,
    sku: row.sku,
    product_name: row.product_name,
    spec: row.spec,
    price: row.price,
    unit: row.unit,
    stock: row.stock,
    image: row.image,
    category: row.category,
    qty_unit: row.qty_unit,
    per_person_net: Number(row.per_person_net || 0),
    per_person_loss: Number(row.per_person_loss || 0),
    per_person_gross: Number(row.per_person_gross || 0),
    source_dishes: row.source_dishes,
    note: row.note_list.join("｜"),
    has_linked_product: !!row.has_linked_product
  }));
  day.nutrition_summary = analyzeMenuNutrition_(day);
  return day;
}

function menuNutritionTargets_(){
  return { calories: 1800, protein: 65, vegetables: 300, grains: 220 };
}

function menuNutritionDishDefaults_(){
  return { main: 70, side: 40, light_veg: 150, dark_veg: 150, root_veg: 80 };
}

function menuNutritionCategoryKey_(category, dishKey){
  const raw = String(category || "") + " " + String(dishKey || "");
  if (/主菜/.test(raw) || String(dishKey) === "main") return "main";
  if (/小配菜/.test(raw) || String(dishKey) === "side") return "side";
  if (/根莖|瓜果|全穀|雜糧/.test(raw) || String(dishKey) === "root_veg") return "root";
  if (/淺色|綠色|深色|蔬菜/.test(raw) || String(dishKey) === "light_veg" || String(dishKey) === "dark_veg") return "veg";
  return "side";
}

function menuNutritionFactors_(bucket, text){
  const source = String(text || "");
  if (bucket === "main") {
    const beanLike = /豆腐|豆干|豆包|蛋|雞蛋|豆皮/.test(source);
    return { calories: beanLike ? 1.65 : 2.2, protein: beanLike ? 0.11 : 0.18 };
  }
  if (bucket === "side") {
    const proteinSide = /豆腐|豆干|蛋|雞蛋|蝦|魚|肉|雞|豬|牛|丸|捲/.test(source);
    return { calories: proteinSide ? 1.45 : 1.05, protein: proteinSide ? 0.08 : 0.03 };
  }
  if (bucket === "veg") return { calories: 0.28, protein: 0.018 };
  if (bucket === "root") return { calories: 0.9, protein: 0.012 };
  return { calories: 0.8, protein: 0.02 };
}

function analyzeMenuNutrition_(day){
  const targets = menuNutritionTargets_();
  const defaults = menuNutritionDishDefaults_();
  let calories = 1180;
  let protein = 34;
  let vegetables = 0;
  let rootNet = 0;

  (day.procurement_items || []).forEach(item => {
    const bucket = menuNutritionCategoryKey_(item.category, "");
    const text = [item.product_name, item.spec].concat(item.source_dishes || []).join(" ");
    const net = Math.max(0, Number(item.per_person_net || 0) || 0);
    const factors = menuNutritionFactors_(bucket, text);
    calories += net * factors.calories;
    protein += net * factors.protein;
    if (bucket === "veg") vegetables += net;
    if (bucket === "root") rootNet += net;
  });

  (day.procurement_unmapped_dishes || []).forEach(item => {
    const bucket = menuNutritionCategoryKey_(item.dish_label, item.dish_key);
    const text = String(item.dish_name || "");
    const net = defaults[item.dish_key] || 0;
    const factors = menuNutritionFactors_(bucket, text);
    calories += net * factors.calories;
    protein += net * factors.protein;
    if (bucket === "veg") vegetables += net;
    if (bucket === "root") rootNet += net;
  });

  const grains = 180 + (rootNet * 0.35);
  const result = {
    targets: targets,
    calories: Math.round(calories),
    protein: Math.round(protein * 10) / 10,
    vegetables: Math.round(vegetables),
    grains: Math.round(grains),
    formula_note: "依菜單管理中的採購試算設定、餐別結構與中央廚房日供餐基準估算，作為每人平均對標分析。"
  };
  result.progress = {
    calories: Math.max(0, Math.min(1.4, result.calories / targets.calories)),
    protein: Math.max(0, Math.min(1.4, result.protein / targets.protein)),
    vegetables: Math.max(0, Math.min(1.4, result.vegetables / targets.vegetables)),
    grains: Math.max(0, Math.min(1.4, result.grains / targets.grains))
  };
  return result;
}

function parseMenuMonthKey_(v){
  const s = String(v || "").trim();
  if (/^\d{4}-\d{2}$/.test(s)) return s;
  const iso = parseMenuDateToISO_(s);
  if (/^\d{4}-\d{2}-\d{2}$/.test(iso)) return iso.slice(0, 7);
  return Utilities.formatDate(new Date(), "Asia/Taipei", "yyyy-MM");
}

function monthStartLocal_(monthKey){
  const parts = String(monthKey || "").split("-");
  const y = parseInt(parts[0], 10);
  const m = parseInt(parts[1], 10);
  const dt = new Date(y, (m || 1) - 1, 1, 0, 0, 0, 0);
  return isNaN(dt.getTime()) ? new Date() : dt;
}

function monthEndLocal_(monthKey){
  const start = monthStartLocal_(monthKey);
  return new Date(start.getFullYear(), start.getMonth() + 1, 0, 0, 0, 0, 0);
}


function menuApiCacheVersion_(){
  const props = PropertiesService.getScriptProperties();
  let v = String(props.getProperty('MENU_API_CACHE_VERSION') || '').trim();
  if (!v) {
    v = '1';
    props.setProperty('MENU_API_CACHE_VERSION', v);
  }
  return v;
}

function menuBumpApiCacheVersion_(){
  PropertiesService.getScriptProperties().setProperty('MENU_API_CACHE_VERSION', String(Date.now()));
}

function menuApiCacheKey_(name, parts){
  const version = menuApiCacheVersion_();
  const raw = [name, version].concat(parts || []).join('|');
  const digest = Utilities.base64EncodeWebSafe(Utilities.computeDigest(Utilities.DigestAlgorithm.MD5, raw)).replace(/=+$/,'');
  return 'menuapi:' + digest;
}

function menuReadApiCache_(key){
  try {
    const raw = CacheService.getScriptCache().get(key);
    if (!raw) return null;
    const parsed = JSON.parse(raw);
    return parsed && typeof parsed === 'object' ? parsed : null;
  } catch (err) {
    return null;
  }
}

function menuWriteApiCache_(key, data, seconds){
  try {
    CacheService.getScriptCache().put(key, JSON.stringify(data), Math.max(30, Number(seconds || 90) || 90));
  } catch (err) {}
}

function apiWeeklyMenu_(p, respond){
  const offset = parseInt(String(p.offset || "0"), 10) || 0;
  const cacheKey = menuApiCacheKey_('weeklyMenu', [String(offset)]);
  const cached = menuReadApiCache_(cacheKey);
  if (cached) return respond(cached);

  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sh = ensureMenuSheet_(ss);
  const ingredientData = loadMenuIngredientMap_(ss);
  const bomData = loadMenuBomMap_(ss);
  const allRows = parseMenuRows_(sh);
  const baseWeekStart = startOfWeekMonday_(new Date());
  const weekStart = addDaysLocal_(baseWeekStart, offset * 7);
  const weekEnd = addDaysLocal_(weekStart, 6);
  const startIso = Utilities.formatDate(weekStart, "Asia/Taipei", "yyyy-MM-dd");
  const endIso = Utilities.formatDate(weekEnd, "Asia/Taipei", "yyyy-MM-dd");
  const dayMap = {};
  for (let i = 0; i < 7; i++) {
    const dateObj = addDaysLocal_(weekStart, i);
    const iso = Utilities.formatDate(dateObj, "Asia/Taipei", "yyyy-MM-dd");
    dayMap[iso] = createMenuDayShell_(iso);
  }
  allRows.filter(row => row.date >= startIso && row.date <= endIso).forEach(row => {
    const day = dayMap[row.date];
    if (!day) return;
    if (row.lunar && !day.lunar) day.lunar = row.lunar;
    day.meals[row.meal_key] = { label: row.meal_label, main: row.main || "", side: row.side || "", light_veg: row.light_veg || "", dark_veg: row.dark_veg || "", root_veg: row.root_veg || "" };
    day.has_menu = true;
  });
  Object.keys(dayMap).forEach(dateKey => buildMenuDayPayload_(dayMap[dateKey], ingredientData, bomData));
  const days = Object.keys(dayMap).sort().map(dateKey => dayMap[dateKey]);
  const result = { status: "ok", sheet_name: sh.getName(), ingredient_sheet_name: ingredientData.sheet.getName(), bom_sheet_name: bomData.sheet.getName(), week_start: startIso, week_end: endIso, week_start_label: buildDateLabelZh_(startIso), week_end_label: buildDateLabelZh_(endIso), days: days, expected_headers: menuSheetHeaders_(), ingredient_expected_headers: menuIngredientSheetHeaders_(), bom_expected_headers: menuBomSheetHeaders_() };
  menuWriteApiCache_(cacheKey, result, 90);
  return respond(result);
}

function apiMenuDashboard_(p, respond){
  const monthKey = parseMenuMonthKey_(p.month || p.date || "");
  const requestedDate = parseMenuDateToISO_(p.date || "") || monthKey + '-01';
  const cacheKey = menuApiCacheKey_('menuDashboard', [monthKey, requestedDate]);
  const cached = menuReadApiCache_(cacheKey);
  if (cached) return respond(cached);

  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sh = ensureMenuSheet_(ss);
  const ingredientData = loadMenuIngredientMap_(ss);
  const bomData = loadMenuBomMap_(ss);
  const allRows = parseMenuRows_(sh);
  const monthStart = monthStartLocal_(monthKey);
  const monthEnd = monthEndLocal_(monthKey);
  const startIso = Utilities.formatDate(monthStart, "Asia/Taipei", "yyyy-MM-dd");
  const endIso = Utilities.formatDate(monthEnd, "Asia/Taipei", "yyyy-MM-dd");
  const calendarMap = {};
  let cursor = new Date(monthStart.getTime());
  while (cursor <= monthEnd) {
    const iso = Utilities.formatDate(cursor, "Asia/Taipei", "yyyy-MM-dd");
    calendarMap[iso] = createMenuDayShell_(iso);
    cursor.setDate(cursor.getDate() + 1);
  }
  allRows.filter(row => row.date >= startIso && row.date <= endIso).forEach(row => {
    const day = calendarMap[row.date];
    if (!day) return;
    if (row.lunar && !day.lunar) day.lunar = row.lunar;
    day.meals[row.meal_key] = { label: row.meal_label, main: row.main || "", side: row.side || "", light_veg: row.light_veg || "", dark_veg: row.dark_veg || "", root_veg: row.root_veg || "" };
    day.has_menu = true;
  });
  let selectedDate = requestedDate && calendarMap[requestedDate] ? requestedDate : "";
  if (!selectedDate) { const todayIso = Utilities.formatDate(new Date(), "Asia/Taipei", "yyyy-MM-dd"); if (calendarMap[todayIso]) selectedDate = todayIso; }
  if (!selectedDate) { const dates = Object.keys(calendarMap).sort(); selectedDate = dates.length ? dates[0] : startIso; }
  const selectedDay = calendarMap[selectedDate] || createMenuDayShell_(selectedDate);
  buildMenuDayPayload_(selectedDay, ingredientData, bomData);
  const calendar_days = Object.keys(calendarMap).sort().map(dateKey => {
    const day = calendarMap[dateKey];
    return { date: dateKey, day: Number(dateKey.slice(-2)), weekday_index: new Date(dateKey + 'T00:00:00').getDay(), weekday_label: weekdayLabelZh_(dateKey), display_date: buildDateLabelZh_(dateKey), lunar: day.lunar || "", has_menu: !!day.has_menu, is_selected: dateKey === selectedDate };
  });
  const result = { status: "ok", sheet_name: sh.getName(), ingredient_sheet_name: ingredientData.sheet.getName(), bom_sheet_name: bomData.sheet.getName(), month: monthKey, month_label: Utilities.formatDate(monthStart, "Asia/Taipei", "yyyy/MM") + "月", selected_date: selectedDate, selected_day: selectedDay, calendar_days: calendar_days, nutrition_targets: menuNutritionTargets_(), expected_headers: menuSheetHeaders_(), ingredient_expected_headers: menuIngredientSheetHeaders_(), bom_expected_headers: menuBomSheetHeaders_() };
  menuWriteApiCache_(cacheKey, result, 90);
  return respond(result);
}


function collectMenuDishOptions_(menuRows){
  const out = [];
  const seen = {};
  (menuRows || []).forEach(row => {
    [row.main, row.side, row.light_veg, row.dark_veg, row.root_veg].forEach(v => {
      const name = String(v || '').trim();
      if (!name) return;
      const key = normalizeMenuDishMatchText_(name);
      if (!key || seen[key]) return;
      seen[key] = true;
      out.push(name);
    });
  });
  return out.sort((a, b) => a.localeCompare(b, 'zh-Hant'));
}

function apiManageMenuMapping_(p, respond){
  const mode = String(p.mode || '').trim().toLowerCase();
  const action = String(p.action || '').trim().toLowerCase();
  if (!['ingredient', 'bom'].includes(mode)) return respond({ status: 'error', message: '缺少 mode' });
  if (!['add', 'update', 'delete'].includes(action)) return respond({ status: 'error', message: '缺少 action' });

  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sheet = mode === 'bom' ? ensureMenuBomSheet_(ss) : ensureMenuIngredientSheet_(ss);
  const headers = mode === 'bom' ? menuBomSheetHeaders_() : menuIngredientSheetHeaders_();
  ensureMenuMappingHeaders_(sheet, headers);

  if (action === 'delete') {
    const rowNo = normalizeMenuMappingRowNo_(p.row_no);
    if (!(rowNo > 1) || rowNo > sheet.getLastRow()) return respond({ status: 'error', message: '找不到要刪除的資料列' });
    sheet.deleteRow(rowNo);
    menuBumpApiCacheVersion_();
    return respond({ status: 'ok', message: '刪除成功' });
  }

  const payload = mustJson_(p.payload);
  if (!payload || typeof payload !== 'object') return respond({ status: 'error', message: 'payload 格式錯誤' });
  const built = mode === 'bom' ? buildMenuBomPayload_(payload) : buildMenuIngredientPayload_(payload);
  if (!built.dish_name) return respond({ status: 'error', message: 'dish_name 不可為空' });
  if (!built.product_id) return respond({ status: 'error', message: 'product_id 不可為空' });
  if (mode === 'bom' && !(Number(built.per_person_qty || 0) > 0)) return respond({ status: 'error', message: 'per_person_qty 必須大於 0' });

  if (action === 'add') {
    appendObject_(sheet, built);
    menuBumpApiCacheVersion_();
    return respond({ status: 'ok', message: '新增成功', row_no: sheet.getLastRow() });
  }

  const rowNo = normalizeMenuMappingRowNo_(payload.row_no || p.row_no);
  if (!(rowNo > 1) || rowNo > sheet.getLastRow()) return respond({ status: 'error', message: '找不到要更新的資料列' });
  writeMenuMappingRow_(sheet, rowNo, built);
  menuBumpApiCacheVersion_();
  return respond({ status: 'ok', message: '更新成功', row_no: rowNo });
}

function normalizeMenuMappingTargets_(v){
  const values = Array.isArray(v) ? v : String(v || '').split(/[\s,|/、]+/);
  let ingredient = false;
  let bom = false;
  values.forEach(function(item){
    const s = String(item || '').trim().toLowerCase();
    if (!s) return;
    if (['both','all','全部','雙用途'].indexOf(s) >= 0) { ingredient = true; bom = true; return; }
    if (s.indexOf('ingredient') >= 0 || s.indexOf('cart') >= 0 || s.indexOf('shop') >= 0 || s.indexOf('購物') >= 0 || s.indexOf('食材') >= 0) ingredient = true;
    if (s.indexOf('bom') >= 0 || s.indexOf('procurement') >= 0 || s.indexOf('採購') >= 0 || s.indexOf('試算') >= 0) bom = true;
  });
  if (!ingredient && !bom) ingredient = true;
  return {
    ingredient: !!ingredient,
    bom: !!bom,
    text: ingredient && bom ? 'both' : (bom ? 'bom' : 'ingredient')
  };
}

function mergeMenuMappingNotes_(a, b){
  const parts = {};
  [a, b].forEach(function(v){
    const s = String(v || '').trim();
    if (!s) return;
    parts[s] = true;
  });
  return Object.keys(parts).join('｜');
}

function mergeUnifiedMenuRow_(base, incoming){
  if (!base) return incoming;
  const a = normalizeMenuMappingTargets_(base.mapping_targets);
  const b = normalizeMenuMappingTargets_(incoming.mapping_targets);
  base.mapping_targets = (a.ingredient || b.ingredient) && (a.bom || b.bom) ? 'both' : ((a.bom || b.bom) ? 'bom' : 'ingredient');
  if (!base.product_name && incoming.product_name) base.product_name = incoming.product_name;
  if (!base.spec && incoming.spec) base.spec = incoming.spec;
  if (!base.product_id && incoming.product_id) base.product_id = incoming.product_id;
  if (!(Number(base.sort_no || 0) > 0) || (Number(incoming.sort_no || 0) > 0 && Number(incoming.sort_no || 0) < Number(base.sort_no || 0))) base.sort_no = incoming.sort_no;
  if (incoming.default_qty !== '' && incoming.default_qty != null) base.default_qty = incoming.default_qty;
  if (incoming.category) base.category = incoming.category;
  if (incoming.per_person_qty !== '' && incoming.per_person_qty != null) base.per_person_qty = incoming.per_person_qty;
  if (incoming.loss_rate) base.loss_rate = incoming.loss_rate;
  if (incoming.qty_unit) base.qty_unit = incoming.qty_unit;
  if (String(base.enabled || '1') === '0' && String(incoming.enabled || '1') !== '0') base.enabled = incoming.enabled;
  base.note = mergeMenuMappingNotes_(base.note, incoming.note);
  return base;
}

function migrateLegacyMenuMappingsToUnified_(ss, targetSheet){
  const hasBody = targetSheet.getLastRow() > 1 && readObjects_(targetSheet).some(function(row){
    return Object.keys(row || {}).some(function(k){ return String(row[k] || '').trim() !== ''; });
  });
  if (hasBody) return { migrated: false, count: 0 };

  const ingSh = legacyMenuIngredientSheetOrNull_(ss);
  const bomSh = legacyMenuBomSheetOrNull_(ss);
  const ingredientRows = ingSh ? readObjects_(ingSh) : [];
  const bomRows = bomSh ? readObjects_(bomSh) : [];
  if (!ingredientRows.length && !bomRows.length) return { migrated: false, count: 0 };

  const merged = {};
  ingredientRows.forEach(function(row){
    const built = buildUnifiedIngredientRow_(row);
    if (!built.dish_name || !(built.product_id || built.product_name)) return;
    const key = menuMappingMergeKey_(built);
    merged[key] = mergeUnifiedMenuRow_(merged[key], built);
  });
  bomRows.forEach(function(row){
    const built = buildUnifiedBomRow_(row);
    if (!built.dish_name || !(built.product_id || built.product_name)) return;
    const key = menuMappingMergeKey_(built);
    merged[key] = mergeUnifiedMenuRow_(merged[key], built);
  });

  const rows = Object.keys(merged).map(function(key){ return merged[key]; }).sort(function(a, b){
    const da = String(a.dish_name || '');
    const db = String(b.dish_name || '');
    if (da !== db) return da.localeCompare(db, 'zh-Hant');
    const sa = Number(a.sort_no || 0);
    const sb = Number(b.sort_no || 0);
    if (sa !== sb) return sa - sb;
    return String(a.product_name || '').localeCompare(String(b.product_name || ''), 'zh-Hant');
  });
  if (!rows.length) return { migrated: false, count: 0 };
  rows.forEach(function(row){ appendObject_(targetSheet, row); });
  return { migrated: true, count: rows.length };
}

function loadUnifiedMenuMappingRows_(ss){
  const sh = ensureMenuUnifiedSheet_(ss);
  migrateLegacyMenuMappingsToUnified_(ss, sh);
  return { sheet: sh, rows: readObjectsWithRowNo_(sh) };
}

function loadMenuIngredientMap_(ss){
  const data = loadUnifiedMenuMappingRows_(ss);
  const productLookup = loadMenuProductLookup_(ss);
  const map = {};
  data.rows.forEach(function(row, index){
    const dishName = String(row.dish_name || '').trim();
    if (!dishName) return;
    if (normalizeShopEnabled_(row.enabled, true) === '0') return;
    const targets = normalizeMenuMappingTargets_(row.mapping_targets);
    if (!targets.ingredient) return;
    const product = resolveMenuIngredientProduct_(row, productLookup);
    if (!product || normalizeShopEnabled_(product.shop_enabled, true) === '0') return;
    const key = normalizeMenuDishMatchText_(dishName);
    if (!key) return;
    if (!map[key]) map[key] = [];
    map[key].push({
      dish_name: dishName,
      sort_no: Number(row.sort_no || index + 1) || (index + 1),
      default_qty: Math.max(1, parseInt(String(row.default_qty || '1'), 10) || 1),
      note: String(row.note || '').trim(),
      product: product
    });
  });
  Object.keys(map).forEach(function(key){
    const seen = {};
    map[key] = map[key].sort(function(a, b){
      const sa = Number(a.sort_no || 0), sb = Number(b.sort_no || 0);
      if (sa !== sb) return sa - sb;
      return String(a.product.name || '').localeCompare(String(b.product.name || ''), 'zh-Hant');
    }).filter(function(row){
      const pid = String((row.product && row.product.id) || '').trim();
      if (!pid) return false;
      if (seen[pid]) return false;
      seen[pid] = true;
      return true;
    });
  });
  return { sheet: data.sheet, map: map };
}

function loadMenuBomMap_(ss){
  const data = loadUnifiedMenuMappingRows_(ss);
  const productLookup = loadMenuProductLookup_(ss);
  const map = {};
  data.rows.forEach(function(row, index){
    const dishName = String(row.dish_name || '').trim();
    if (!dishName) return;
    if (normalizeShopEnabled_(row.enabled, true) === '0') return;
    const targets = normalizeMenuMappingTargets_(row.mapping_targets);
    if (!targets.bom) return;
    const resolvedProduct = resolveMenuIngredientProduct_(row, productLookup);
    const product = resolvedProduct ? resolvedProduct : {
      id: '',
      sku: '',
      name: String(row.product_name || '').trim(),
      spec: String(row.spec || '').trim(),
      price: 0,
      image: '',
      stock: 0,
      unit: String(row.qty_unit || '').trim(),
      shop_enabled: '1'
    };
    if (!String(product.name || '').trim()) return;
    if (resolvedProduct && normalizeShopEnabled_(product.shop_enabled, true) === '0') return;
    const perPersonQty = Math.max(0, normalizeMenuBomNumber_(row.per_person_qty, 0));
    if (!(perPersonQty > 0)) return;
    const key = normalizeMenuDishMatchText_(dishName);
    if (!key) return;
    if (!map[key]) map[key] = [];
    map[key].push({
      dish_name: dishName,
      sort_no: Number(row.sort_no || index + 1) || (index + 1),
      category: String(row.category || '').trim(),
      per_person_qty: perPersonQty,
      loss_rate: normalizeMenuBomLossRate_(row.loss_rate),
      qty_unit: normalizeMenuBomQtyUnit_(row, product),
      note: String(row.note || '').trim(),
      product: product,
      has_linked_product: !!(resolvedProduct && String(resolvedProduct.id || '').trim())
    });
  });
  Object.keys(map).forEach(function(key){
    const seen = {};
    map[key] = map[key].sort(function(a, b){
      const sa = Number(a.sort_no || 0), sb = Number(b.sort_no || 0);
      if (sa !== sb) return sa - sb;
      return String((a.product && a.product.name) || '').localeCompare(String((b.product && b.product.name) || ''), 'zh-Hant');
    }).filter(function(row){
      const pid = String((row.product && row.product.id) || '').trim();
      const pname = normalizeMenuDishMatchText_((row.product && row.product.name) || '');
      const pspec = normalizeMenuDishMatchText_((row.product && row.product.spec) || '');
      const q = String(row.qty_unit || '').trim();
      const uniq = (pid ? pid : (pname + '__' + pspec)) + '__' + q;
      if (!uniq || /^__+$/.test(uniq)) return false;
      if (seen[uniq]) return false;
      seen[uniq] = true;
      return true;
    });
  });
  return { sheet: data.sheet, map: map };
}

  }

  const payload = mustJson_(p.payload);
  if (!payload || typeof payload !== 'object') return respond({ status: 'error', message: 'payload 格式錯誤' });
  const built = buildMenuUnifiedPayload_(payload, mode);
  const targetInfo = normalizeMenuMappingTargets_(built.mapping_targets);
  if (!built.dish_name) return respond({ status: 'error', message: 'dish_name 不可為空' });
  if (!(built.product_id || built.product_name)) return respond({ status: 'error', message: '請先選擇商品' });
  if (!targetInfo.ingredient && !targetInfo.bom) return respond({ status: 'error', message: '請至少選擇一種用途' });
  if (targetInfo.bom && !(Number(built.per_person_qty || 0) > 0)) return respond({ status: 'error', message: 'per_person_qty 必須大於 0' });

  if (action === 'add') {
    appendObject_(sheet, built);
    menuBumpApiCacheVersion_();
    return respond({ status: 'ok', message: '新增成功', row_no: sheet.getLastRow() });
  }

  const rowNo = normalizeMenuMappingRowNo_(payload.row_no || p.row_no);
  if (!(rowNo > 1) || rowNo > sheet.getLastRow()) return respond({ status: 'error', message: '找不到要更新的資料列' });
  writeMenuMappingRow_(sheet, rowNo, built);
  menuBumpApiCacheVersion_();
  return respond({ status: 'ok', message: '更新成功', row_no: rowNo });
}
