function doGet(e) {
  const p = (e && e.parameter) ? e.parameter : {};
  const type = String(p.type || "").trim();

  const callback = String(p.callback || "").trim();
  const jsonpOk = callback && /^[A-Za-z_$][0-9A-Za-z_$]*(\.[0-9A-Za-z_$]+)*$/.test(callback);

  const respond = (obj) => {
    const json = JSON.stringify(obj);
    const body = jsonpOk ? `${callback}(${json})` : json;
    const out = ContentService.createTextOutput(body);
    return out.setMimeType(jsonpOk ? ContentService.MimeType.JAVASCRIPT : ContentService.MimeType.JSON);
  };

  try {
    if (type === "products") return apiProducts_(respond);
    if (type === "manageProduct") return apiManageProduct_(p, respond);

    if (type === "members") return apiMembersLogin_(p, respond);

    // 客戶（電商）登入
    if (type === "customerLogin") return apiCustomerLogin_(p, respond);

    // 客戶（電商）註冊（寫入 customers）
    if (type === "register") return apiCustomerRegister_(p, respond);

    // 後台使用者（Members）管理
    if (type === "manageMember") return apiManageMember_(p, respond);

    // 前台（商城）下單/查單
    if (type === "order") return apiOrderCreate_(p, respond);
    if (type === "myOrders") return apiMyOrders_(p, respond);
    if (type === "getOrder") return apiGetOrder_(p, respond);

    if (type === "suppliers") return apiSuppliers_(respond);
    if (type === "manageSupplier") return apiManageSupplier_(p, respond);

    if (type === "customers") return apiCustomers_(respond);
    if (type === "manageCustomer") return apiManageCustomer_(p, respond);

    if (type === "purchases") return apiPurchases_(p, respond);
    if (type === "managePurchase") return apiManagePurchase_(p, respond);

    if (type === "pickups") return apiPickups_(respond);
    if (type === "managePickup") return apiManagePickup_(p, respond);

    if (type === "orders") return apiOrders_(respond);
    if (type === "manageOrder") return apiManageOrder_(p, respond);

    if (type === "manageSale") return apiManageSale_(p, respond);

    if (type === "stockLedger") return apiStockLedger_(respond);
    if (type === "productLedger") return apiProductLedger_(p, respond);
    if (type === "stockAdjust") return apiStockAdjust_(p, respond);

    if (type === "syncReferencePrices") return apiSyncReferencePrices_(respond);
    if (type === "marketPriceBoard") return apiMarketPriceBoard_(respond);

    if (type === "systemSettings") return apiSystemSettings_(respond);
    if (type === "manageSystemSettings") return apiManageSystemSettings_(p, respond);

    if (type === "lastUpdate") return apiLastUpdate_(respond);

    return respond({ status: "error", message: "Invalid Request" });
  } catch (err) {
    return respond({ status: "error", message: String(err && err.message ? err.message : err) });
  }
}

// ----------------- Helpers -----------------
function findSheetByTrimName_(ss, name){
  const target = String(name || "").trim();
  const sheets = ss.getSheets();
  for (let i = 0; i < sheets.length; i++){
    const n = String(sheets[i].getName() || "").trim();
    if (n === target) return sheets[i];
  }
  return null;
}

function getSheetByAnyName_(ss, names) {
  for (let i = 0; i < names.length; i++) {
    const sh = ss.getSheetByName(names[i]);
    if (sh) return sh;
  }
  for (let i = 0; i < names.length; i++) {
    const sh = ss.getSheetByName(String(names[i]).trim());
    if (sh) return sh;
  }
  return null;
}

function ensureSheet_(ss, name, headers) {
  let sh = ss.getSheetByName(name) || findSheetByTrimName_(ss, name);
  if (!sh) {
    sh = ss.insertSheet(name);
  } else {
    // 若是被空白或大小寫造成的名字差異，嘗試統一名稱
    if (sh.getName() !== name) {
      try { sh.setName(name); } catch(e) {}
    }
  }
  const lastCol = Math.max(sh.getLastColumn(), headers.length);
  const row1 = sh.getRange(1, 1, 1, lastCol).getValues()[0];
  const hasAny = row1.some(v => String(v || "").trim() !== "");
  if (!hasAny) {
    sh.getRange(1, 1, 1, headers.length).setValues([headers]);
  } else {
    // Fill missing header cells in order (do not overwrite existing)
    for (let i = 0; i < headers.length; i++) {
      if (!row1[i] || String(row1[i]).trim() === "") sh.getRange(1, i + 1).setValue(headers[i]);
    }
  }
  return sh;
}

function getHeaders_(sheet) {
  const lastCol = Math.max(sheet.getLastColumn() || 1, 1);
  return sheet.getRange(1, 1, 1, lastCol).getValues()[0].map(h => String(h || "").trim());
}


function normalizeShopEnabled_(v, defaultEnabled) {
  const raw = String(v == null ? "" : v).trim().toLowerCase();
  if (!raw) return defaultEnabled === false ? "0" : "1";
  return (["0","false","no","off","n","hide","hidden"].includes(raw)) ? "0" : "1";
}

function headerIndexMap_(headers) {
  const m = {};
  headers.forEach((h, i) => { if (h) m[h] = i; });
  return m;
}

function readObjects_(sheet) {
  const values = sheet.getDataRange().getValues();
  if (values.length <= 1) return [];
  const headers = values[0].map(h => String(h || "").trim());
  const out = [];
  for (let r = 1; r < values.length; r++) {
    const row = values[r];
    if (!row || row.every(v => v === "" || v === null)) continue;
    const obj = {};
    for (let c = 0; c < headers.length; c++) {
      const key = headers[c];
      if (!key) continue;
      obj[key] = row[c];
    }
    out.push(obj);
  }
  return out;
}

function appendObject_(sheet, obj) {
  const headers = getHeaders_(sheet);
  const row = new Array(headers.length).fill("");
  headers.forEach((h, i) => {
    if (!h) return;
    if (Object.prototype.hasOwnProperty.call(obj, h)) row[i] = obj[h];
  });
  sheet.appendRow(row);
}

function toROCDateString_(d) {
  if (!(d instanceof Date)) return String(d || "").trim();
  const y = d.getFullYear() - 1911;
  const m = ("0" + (d.getMonth() + 1)).slice(-2);
  const day = ("0" + d.getDate()).slice(-2);
  return `${y}.${m}.${day}`;
}

function toISODateLocal_(d){
  if (!d) return "";
  try{
    // Date object
    if (Object.prototype.toString.call(d) === "[object Date]" && !isNaN(d.getTime())) {
      return Utilities.formatDate(d, "Asia/Taipei", "yyyy-MM-dd");
    }
    const s = String(d).trim();
    if (!s) return "";
    // ISO with time (Z) -> parse then format local
    if (/^\d{4}-\d{2}-\d{2}T/.test(s)) {
      const dt = new Date(s);
      if (!isNaN(dt.getTime())) return Utilities.formatDate(dt, "Asia/Taipei", "yyyy-MM-dd");
      return s.slice(0,10);
    }
    // ISO date
    if (/^\d{4}-\d{2}-\d{2}/.test(s)) return s.slice(0,10);
    // ROC YYY.MM.DD / etc
    if (/^\d{2,3}[\.\/\-]\d{1,2}[\.\/\-]\d{1,2}/.test(s) && !/^\d{4}[\.\/\-]/.test(s)) {
      const base = s.split(/[\sT]/)[0];
      const parts = base.split(/[\.\/\-]/);
      const y = (parseInt(parts[0],10) + 1911);
      const m = ("0"+parts[1]).slice(-2);
      const dd = ("0"+parts[2]).slice(-2);
      return `${y}-${m}-${dd}`;
    }
    // Fallback Date parse
    const dt2 = new Date(s);
    if (!isNaN(dt2.getTime())) return Utilities.formatDate(dt2, "Asia/Taipei", "yyyy-MM-dd");
    return "";
  } catch(e){
    return "";
  }
}


function decimalPlacesText_(v){
  const s = String(v == null ? "" : v).replace(/,/g, "").trim();
  const m = s.match(/\.(\d+)$/);
  return m ? m[1].length : 0;
}

function cleanDecimalInput_(v, maxScale){
  maxScale = (maxScale === undefined) ? 6 : Number(maxScale || 0);
  const s = String(v == null ? "" : v).replace(/,/g, "").trim();
  if (!s) return 0;
  const n = Number(s);
  if (!isFinite(n)) return 0;
  const scale = Math.min(decimalPlacesText_(s), maxScale);
  return Number(n.toFixed(scale));
}

function mulDecimalInput_(a, b, maxScale){
  maxScale = (maxScale === undefined) ? 6 : Number(maxScale || 0);
  const na = Number(String(a == null ? "" : a).replace(/,/g, "").trim() || 0);
  const nb = Number(String(b == null ? "" : b).replace(/,/g, "").trim() || 0);
  if (!isFinite(na) || !isFinite(nb)) return 0;
  const scale = Math.min(decimalPlacesText_(a) + decimalPlacesText_(b), maxScale);
  return Number((na * nb).toFixed(scale));
}

function addDecimalInput_(a, b, maxScale){
  maxScale = (maxScale === undefined) ? 6 : Number(maxScale || 0);
  const na = Number(a || 0);
  const nb = Number(b || 0);
  if (!isFinite(na) || !isFinite(nb)) return 0;
  const scale = Math.min(Math.max(decimalPlacesText_(a), decimalPlacesText_(b)), maxScale);
  return Number((na + nb).toFixed(scale));
}

function normalizeDate_(d) {
  if (d instanceof Date) return toROCDateString_(d);
  return String(d || "").trim();
}

function compareDate_(d1, d2) {
  d1 = normalizeDate_(d1);
  d2 = normalizeDate_(d2);
  if (!d1 && !d2) return 0;
  if (!d1) return -1;
  if (!d2) return 1;
  const a = d1.split(".").map(Number);
  const b = d2.split(".").map(Number);
  if (a.length < 3 || b.length < 3 || a.some(isNaN) || b.some(isNaN)) {
    return d1 === d2 ? 0 : (d1 > d2 ? 1 : -1);
  }
  if (a[0] !== b[0]) return a[0] - b[0];
  if (a[1] !== b[1]) return a[1] - b[1];
  return a[2] - b[2];
}

function getLedgerSheet_(ss) {
  return getSheetByAnyName_(ss, ["stock_ledger", "stockLedger", "StockLedger", "inventoryLog", "InventoryLog", "lnventoryLog"]);
}

function ensureLedgerSheet_(ss) {
  // 以既有欄位為主（不重排、不覆蓋），必要時才補齊缺漏欄位
  const baseHeaders = ["ledger_id","product_id","product_name","change","direction","reason","ref_id","operator","date","target"];
  const sh = getLedgerSheet_(ss) || ensureSheet_(ss, "stock_ledger", baseHeaders);

  // 若 sheet 已有表頭，確保 baseHeaders 都存在（缺的就補在最後，不改動既有順序）
  const headers = getHeaders_(sh);
  const idx = headerIndexMap_(headers);
  let changed = false;
  let lastCol = sh.getLastColumn();
  baseHeaders.forEach(h => {
    if (idx[h] === undefined) {
      lastCol += 1;
      sh.getRange(1, lastCol).setValue(h);
      changed = true;
    }
  });
  if (changed) {
    // No-op, just makes sure header row is committed
  }

// 若需要顯示/追蹤歷史成本，追加欄位（只加在最後，不重排）
const extraHeaders = ["cost","sku","unit"];
extraHeaders.forEach(h => {
  if (idx[h] === undefined) {
    lastCol += 1;
    sh.getRange(1, lastCol).setValue(h);
    changed = true;
  }
});
  return sh;
}

function writeLedger_(ss, obj) {
  const ledger = ensureLedgerSheet_(ss);
  appendObject_(ledger, obj);
}

/**
 * 供應商代碼規則：yyyyMMdd + 3位流水（例：20260317001）
 * - 每日序號從 001 起算
 * - 只掃描 suppliers.id 欄位中符合當日前綴的純數字代碼
 * - 建議搭配 LockService 使用，避免併發重號
 */
function generateSupplierDailyId_(sheet, dateObj) {
  const d = dateObj instanceof Date ? dateObj : new Date();
  const prefix = Utilities.formatDate(d, "Asia/Taipei", "yyyyMMdd");
  const headers = getHeaders_(sheet);
  const idx = headerIndexMap_(headers);
  const col = idx.id;
  if (col === undefined) throw new Error('找不到欄位：id');

  const lastRow = sheet.getLastRow();
  let maxSeq = 0;
  if (lastRow >= 2) {
    const vals = sheet.getRange(2, col + 1, lastRow - 1, 1).getValues();
    for (let i = 0; i < vals.length; i++) {
      const s = String(vals[i][0] || '').trim();
      if (!s) continue;
      if (/^\d+$/.test(s) && s.indexOf(prefix) === 0 && s.length >= 11) {
        const suf = parseInt(s.slice(8), 10);
        if (!isNaN(suf)) maxSeq = Math.max(maxSeq, suf);
      }
    }
  }

  const next = maxSeq + 1;
  const suf3 = ('000' + next).slice(-3);
  return prefix + suf3;
}

/**
 * 單號規則：yyyyMMdd + 4位流水（例：202602090001）
 * - 每日序號從 0001 起算
 * - 以指定 sheet + 欄位掃描當日最大值後 +1
 * - 建議在 LockService 區塊內呼叫，避免併發重號
 */
function generateDailyDocNo_(sheet, idField, dateObj) {
  const d = dateObj instanceof Date ? dateObj : new Date();
  const prefix = Utilities.formatDate(d, "Asia/Taipei", "yyyyMMdd");

  const headers = getHeaders_(sheet);
  const idx = headerIndexMap_(headers);
  const col = idx[idField];
  if (col === undefined) throw new Error(`找不到欄位：${idField}`);

  const lastRow = sheet.getLastRow();
  let maxSeq = 0;

  if (lastRow >= 2) {
    const vals = sheet.getRange(2, col + 1, lastRow - 1, 1).getValues();
    for (let i = 0; i < vals.length; i++) {
      const s = String(vals[i][0] || "").trim();
      if (!s) continue;
      // 只看符合 prefix 的純數字單號（舊格式會被忽略）
      if (s.length >= 12 && s.indexOf(prefix) === 0) {
        const suf = parseInt(s.slice(8), 10);
        if (!isNaN(suf)) maxSeq = Math.max(maxSeq, suf);
      }
    }
  }

  const next = maxSeq + 1;
  const suf4 = ("0000" + next).slice(-4);
  return prefix + suf4;
}

function toISODateOnly_(s){
  // Accept YYYY-MM-DD or ROC like 115.02.03 or 115/02/03
  s = String(s || "").trim();
  if (!s) return Utilities.formatDate(new Date(), Session.getScriptTimeZone(), "yyyy-MM-dd");
  if (/^\d{4}-\d{2}-\d{2}$/.test(s)) return s;

  // ROC format
  const m = s.match(/^(\d{2,3})[\.\/\-](\d{1,2})[\.\/\-](\d{1,2})/);
  if (m){
    const y = Number(m[1]) + 1911;
    const mo = String(m[2]).padStart(2,"0");
    const d = String(m[3]).padStart(2,"0");
    return `${y}-${mo}-${d}`;
  }

  // Fallback parse
  const dt = new Date(s);
  if (!isNaN(dt.getTime())) return Utilities.formatDate(dt, Session.getScriptTimeZone(), "yyyy-MM-dd");
  return Utilities.formatDate(new Date(), Session.getScriptTimeZone(), "yyyy-MM-dd");
}

function toLocalDateTime_(d){
  const dt = (d instanceof Date) ? d : new Date(d);
  return Utilities.formatDate(dt, "Asia/Taipei", "yyyy-MM-dd HH:mm:ss");
}


function mustJson_(s) {
  const raw = String(s == null ? "" : s);
  if (!raw) return null;
  try {
    return JSON.parse(raw);
  } catch (e1) {
    try {
      return JSON.parse(decodeURIComponent(raw));
    } catch (e2) {
      throw e2;
    }
  }
}

function syncPingtungRawAndReferencePrices_() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const rawSh = ensureSheet_(ss, "PingtungRaw", PINGTUNG_RAW_HEADERS_);
  const existingRows = readNormalizedPingtungRawRows_(rawSh);

  let fetched = null;
  let usedFallback = false;
  try {
    fetched = fetchPingtungRawFromGist_();
  } catch (err) {
    const msg = String(err && err.message || err || '');
    const looksLikeHoliday = msg.indexOf('沒有可辨識的蔬菜行情資料') >= 0 || msg.indexOf('沒有可辨識的行情資料') >= 0;
    if (!looksLikeHoliday || !existingRows.length) throw err;
    fetched = { rows: [] };
    usedFallback = true;
  }

  const mergedRows = mergeLatestRawRows_(existingRows, fetched.rows || []);
  if (!mergedRows.length) throw new Error('目前沒有可保留的果菜市場價格資料');

  rawSh.clearContents();
  rawSh.getRange(1, 1, 1, PINGTUNG_RAW_HEADERS_.length).setValues([PINGTUNG_RAW_HEADERS_]);
  rawSh.getRange(2, 1, mergedRows.length, PINGTUNG_RAW_HEADERS_.length).setValues(mergedRows.map(r => [
    r.sync_time,
    r.source_file,
    r.market_date,
    r.item_name,
    r.spec,
    r.market,
    r.upper_price,
    r.middle_price,
    r.lower_price,
    r.raw_json
  ]));

  const applied = applyReferencePricesFromRaw_(ss, mergedRows);
  const latestMarketDate = mergedRows.reduce((acc, r) => String(r.market_date || '') > acc ? String(r.market_date || '') : acc, '');
  const suffix = usedFallback
    ? `（本次未取得可用新行情，已保留最後一次更新價格，最新行情日期 ${latestMarketDate || '—'}）`
    : '';

  return {
    message: `同步完成：PingtungRaw ${mergedRows.length} 筆，商品主檔參考價格與售價更新 ${applied.updated_count} 筆${suffix}`,
    raw_count: mergedRows.length,
    fetched_count: (fetched && Array.isArray(fetched.rows)) ? fetched.rows.length : 0,
    preserved_count: Math.max(0, mergedRows.length - (((fetched && Array.isArray(fetched.rows)) ? fetched.rows.length : 0))),
    updated_count: applied.updated_count,
    mapped_count: applied.mapped_count,
    unmatched_mapping_count: applied.unmatched_mapping_count,
    latest_market_date: latestMarketDate,
    used_fallback: usedFallback,
    synced_at: Utilities.formatDate(new Date(), Session.getScriptTimeZone(), "yyyy-MM-dd HH:mm:ss")
  };
}

function readNormalizedPingtungRawRows_(sh) {
  if (!sh) return [];
  return readObjects_(sh).map(r => ({
    sync_time: String(r.sync_time || '').trim(),
    source_file: String(r.source_file || '').trim(),
    market_date: normalizeLooseDate_(r.market_date || ''),
    item_name: String(r.item_name || '').trim(),
    spec: String(r.spec || '').trim(),
    market: normalizeMarketLabel_(r.market || ''),
    upper_price: parseLooseNumber_(r.upper_price),
    middle_price: parseLooseNumber_(r.middle_price),
    lower_price: parseLooseNumber_(r.lower_price),
    raw_json: String(r.raw_json || '')
  })).filter(r => r.item_name && r.market);
}

function mergeLatestRawRows_(existingRows, fetchedRows) {
  const bucket = {};
  const put_ = (row, preferNew) => {
    if (!row) return;
    const market = normalizeMarketLabel_(row.market || '');
    const itemName = String(row.item_name || '').trim();
    const spec = String(row.spec || '').trim();
    if (!market || !itemName) return;
    const key = [market, normalizeLookupKey_(itemName), normalizeLookupKey_(spec)].join('||');
    const normalized = {
      sync_time: String(row.sync_time || '').trim(),
      source_file: String(row.source_file || '').trim(),
      market_date: normalizeLooseDate_(row.market_date || ''),
      item_name: itemName,
      spec: spec,
      market: market,
      upper_price: parseLooseNumber_(row.upper_price),
      middle_price: parseLooseNumber_(row.middle_price),
      lower_price: parseLooseNumber_(row.lower_price),
      raw_json: String(row.raw_json || '')
    };
    const current = bucket[key];
    if (!current) {
      bucket[key] = normalized;
      return;
    }
    const currentDate = String(current.market_date || '');
    const nextDate = String(normalized.market_date || '');
    if (nextDate > currentDate) {
      bucket[key] = normalized;
      return;
    }
    if (nextDate < currentDate) return;

    const currentScore = [current.upper_price, current.middle_price, current.lower_price].filter(v => v !== '' && v !== null && v !== undefined).length;
    const nextScore = [normalized.upper_price, normalized.middle_price, normalized.lower_price].filter(v => v !== '' && v !== null && v !== undefined).length;
    if (nextScore > currentScore) {
      bucket[key] = normalized;
      return;
    }
    if (nextScore < currentScore) return;

    const currentSync = String(current.sync_time || '');
    const nextSync = String(normalized.sync_time || '');
    if (nextSync > currentSync || (preferNew && nextSync === currentSync)) {
      bucket[key] = normalized;
    }
  };

  (Array.isArray(existingRows) ? existingRows : []).forEach(row => put_(row, false));
  (Array.isArray(fetchedRows) ? fetchedRows : []).forEach(row => put_(row, true));

  return Object.keys(bucket).map(k => bucket[k]).sort((a, b) => {
    const marketCmp = marketSortPriority_(b.market) - marketSortPriority_(a.market);
    if (marketCmp !== 0) return marketCmp;
    const dateCmp = String(b.market_date || '').localeCompare(String(a.market_date || ''));
    if (dateCmp !== 0) return dateCmp;
    const nameCmp = String(a.item_name || '').localeCompare(String(b.item_name || ''), 'zh-Hant');
    if (nameCmp !== 0) return nameCmp;
    return String(a.spec || '').localeCompare(String(b.spec || ''), 'zh-Hant');
  });
}

function fetchPingtungRawFromGist_() {
  const res = UrlFetchApp.fetch(PINGTUNG_GIST_API_URL_, {
    muteHttpExceptions: true,
    headers: {
      Accept: "application/vnd.github+json",
      "User-Agent": "Google-Apps-Script"
    }
  });
  const code = res.getResponseCode();
  if (code < 200 || code >= 300) throw new Error(`抓取 GitHub Gist 失敗（HTTP ${code}）`);

  const gist = JSON.parse(res.getContentText() || "{}");
  const files = gist.files || {};
  const syncTime = Utilities.formatDate(new Date(), Session.getScriptTimeZone(), "yyyy-MM-dd HH:mm:ss");
  const out = [];

  Object.keys(files).forEach(k => {
    const f = files[k] || {};
    const filename = String(f.filename || k || "").trim() || k;
    let content = String(f.content || "");
    if ((!content || f.truncated) && f.raw_url) {
      const rawRes = UrlFetchApp.fetch(String(f.raw_url), {
        muteHttpExceptions: true,
        headers: { "User-Agent": "Google-Apps-Script" }
      });
      if (rawRes.getResponseCode() >= 200 && rawRes.getResponseCode() < 300) content = rawRes.getContentText();
    }
    const parsedRows = parsePingtungMarketFile_(content, filename, String(f.type || ""));
    parsedRows.forEach(row => {
      row.sync_time = syncTime;
      row.source_file = filename;
      out.push(row);
    });
  });

  if (!out.length) throw new Error("Gist 內沒有可辨識的蔬菜行情資料");
  return { rows: out };
}

function parsePingtungMarketFile_(content, filename, mimeType) {
  const text = String(content || "").replace(/^﻿/, "").trim();
  if (!text) return [];

  const lowerName = String(filename || "").toLowerCase();
  let candidates = [];

  const tryJson = () => {
    try {
      const data = JSON.parse(text);
      candidates = extractRowsFromAnyJson_(data);
      return true;
    } catch (e) {
      return false;
    }
  };

  const tryDelimited = (delimiter) => {
    try {
      const rows = Utilities.parseCsv(text, delimiter);
      if (!rows || rows.length < 2) return false;
      const headers = rows[0].map(v => String(v || "").trim());
      candidates = rows.slice(1).map(r => {
        const obj = {};
        headers.forEach((h, idx) => { if (h) obj[h] = r[idx]; });
        return obj;
      });
      return true;
    } catch (e) {
      return false;
    }
  };

  if (lowerName.endsWith('.json') || /json/i.test(mimeType)) {
    if (!tryJson()) { tryDelimited(',') || tryDelimited('\t'); }
  } else if (lowerName.endsWith('.csv')) {
    if (!tryDelimited(',')) tryJson();
  } else if (lowerName.endsWith('.tsv') || lowerName.endsWith('.txt')) {
    if (!tryDelimited('\t')) { if (!tryDelimited(',')) tryJson(); }
  } else {
    if (!tryJson()) { if (!tryDelimited(',')) tryDelimited('\t'); }
  }

  if (!candidates.length && looksLikeMarketTextBlock_(text)) {
    candidates = parseMarketTextBlock_(text, filename, '');
  }

  const out = [];
  candidates.forEach(obj => {
    const norm = normalizeMarketRow_(obj);
    if (!norm) return;
    out.push({
      market_date: norm.market_date,
      item_name: norm.item_name,
      spec: norm.spec,
      market: norm.market,
      upper_price: norm.upper_price,
      middle_price: norm.middle_price,
      lower_price: norm.lower_price,
      raw_json: JSON.stringify(obj)
    });
  });
  return out;
}

function extractRowsFromAnyJson_(node) {
  const out = [];
  const seenTextBlocks = {};

  function pushTextBlock_(text, marketHint, dateHint) {
    const s = String(text || '').trim();
    if (!s) return;
    const key = `${String(marketHint || '')}__${String(dateHint || '')}__${s.slice(0, 120)}`;
    if (seenTextBlocks[key]) return;
    seenTextBlocks[key] = true;
    parseMarketTextBlock_(s, marketHint, dateHint).forEach(row => out.push(row));
  }

  function walk(v, keyHint, inheritedDateHint) {
    if (Array.isArray(v)) {
      v.forEach(item => walk(item, keyHint, inheritedDateHint));
      return;
    }
    if (typeof v === 'string') {
      if (looksLikeMarketTextBlock_(v)) pushTextBlock_(v, keyHint, inheritedDateHint);
      return;
    }
    if (!v || typeof v !== 'object') return;

    if (looksLikeMarketRowObject_(v)) out.push(v);

    const localDateHint = pickValueByAliases_(v, ['updatedAt','updated_at','date','market_date','trade_date','日期','交易日期']) || inheritedDateHint || '';
    Object.keys(v).forEach(k => {
      const child = v[k];
      if (typeof child === 'string') {
        if (looksLikeMarketTextBlock_(child)) pushTextBlock_(child, k, localDateHint);
        return;
      }
      if (Array.isArray(child) || (child && typeof child === 'object')) walk(child, k, localDateHint);
    });
  }

  walk(node, '', '');
  return out;
}


function looksLikeMarketTextBlock_(text) {
  const s = String(text || '').trim();
  if (!s) return false;
  return /上價\s*[\d\-]/.test(s) && /中價\s*[\d\-]/.test(s) && /下價\s*[\d\-]/.test(s) && /交易日期/.test(s);
}

function parseMarketTextBlock_(text, marketHint, dateHint) {
  const out = [];
  const lines = String(text || '').split(/\r?\n+/).map(v => String(v || '').trim()).filter(Boolean);
  lines.forEach(line => {
    const cleaned = line.replace(/^[•‧・●○◦▪■□◆◇★☆※◎◉\-]+\s*/, '').trim();
    if (!cleaned) return;

    const itemPart = String((cleaned.split(/[：:]/)[0] || '')).trim();
    const upperM = cleaned.match(/上價\s*([^，,]+)/);
    const middleM = cleaned.match(/中價\s*([^，,]+)/);
    const lowerM = cleaned.match(/下價\s*([^，,]+)/);
    const dateM = cleaned.match(/交易日期\s*([0-9\.\/-]+)/);

    const upper = parseLooseNumber_(upperM ? upperM[1] : '');
    const middle = parseLooseNumber_(middleM ? middleM[1] : '');
    const lower = parseLooseNumber_(lowerM ? lowerM[1] : '');
    const marketDate = normalizeLooseDate_(dateM ? dateM[1] : dateHint);

    if (!itemPart || (upper === '' && middle === '' && lower === '')) return;

    const split = splitMarketItemAndSpec_(itemPart);
    out.push({
      item_name: split.item_name,
      spec: split.spec,
      market: normalizeMarketLabel_(marketHint),
      market_date: marketDate,
      upper_price: upper,
      middle_price: middle,
      lower_price: lower,
      raw_line: cleaned
    });
  });
  return out;
}

function splitMarketItemAndSpec_(label) {
  const s = String(label || '').trim();
  if (!s) return { item_name: '', spec: '' };
  const parts = s.split(/\s*[\-－–—]\s*/);
  if (parts.length <= 1) return { item_name: s, spec: '' };
  return {
    item_name: String(parts.shift() || '').trim(),
    spec: String(parts.join('-') || '').trim()
  };
}

function normalizeMarketLabel_(marketHint) {
  const raw = String(marketHint || '').trim();
  const nk = normalizeLookupKey_(raw);
  if (nk.indexOf('pingtung') >= 0 || raw.indexOf('屏東') >= 0) return 'Pingtung';
  if (nk.indexOf('firstmarket') >= 0 || raw.indexOf('第一市場') >= 0) return 'FirstMarket';
  if (nk.indexOf('secondmarket') >= 0 || raw.indexOf('第二市場') >= 0) return 'SecondMarket';
  return raw;
}

function marketSortPriority_(market) {
  const raw = String(market || '').trim();
  const nk = normalizeLookupKey_(raw);
  if (nk.indexOf('pingtung') >= 0 || raw.indexOf('屏東') >= 0) return 30;
  if (nk.indexOf('secondmarket') >= 0 || raw.indexOf('第二市場') >= 0) return 20;
  if (nk.indexOf('firstmarket') >= 0 || raw.indexOf('第一市場') >= 0) return 10;
  return 0;
}


function looksLikeMarketRowObject_(obj) {
  const name = pickValueByAliases_(obj, ["item_name","name","item","product","crop","commodity","菜名","品名","名稱","作物名稱","作物","蔬菜名稱"]);
  const upper = pickValueByAliases_(obj, ["upper_price","high_price","high","upper","max_price","最高價","上價"]);
  const middle = pickValueByAliases_(obj, ["middle_price","mid_price","mid","middle","avg_price","中價","平均價"]);
  const lower = pickValueByAliases_(obj, ["lower_price","low_price","low","lower","min_price","最低價","下價"]);
  return !!String(name || "").trim() && [upper, middle, lower].some(v => String(v || "").trim() !== "");
}

function normalizeMarketRow_(obj) {
  const combinedName = String(pickValueByAliases_(obj, ['item_name','name','item','product','crop','commodity','菜名','品名','名稱','作物名稱','作物','蔬菜名稱']) || '').trim();
  if (!combinedName) return null;

  let spec = String(pickValueByAliases_(obj, ['spec','grade','size','規格','規格別','品級','品規']) || '').trim();
  let itemName = combinedName;
  if (!spec) {
    const split = splitMarketItemAndSpec_(combinedName);
    itemName = split.item_name;
    spec = split.spec;
  }

  const market = String(pickValueByAliases_(obj, ['market','market_name','市場','交易市場']) || '').trim();
  const marketDate = normalizeLooseDate_(pickValueByAliases_(obj, ['date','market_date','trade_date','updatedAt','updated_at','日期','交易日期','更新日期','查詢日期']) || '');
  const upper = parseLooseNumber_(pickValueByAliases_(obj, ['upper_price','high_price','high','upper','max_price','最高價','上價']));
  const middle = parseLooseNumber_(pickValueByAliases_(obj, ['middle_price','mid_price','mid','middle','avg_price','中價','平均價']));
  const lower = parseLooseNumber_(pickValueByAliases_(obj, ['lower_price','low_price','low','lower','min_price','最低價','下價']));
  if (upper === '' && middle === '' && lower === '') return null;
  return { item_name: itemName, spec: spec, market: market, market_date: marketDate, upper_price: upper, middle_price: middle, lower_price: lower };
}

function pickValueByAliases_(obj, aliases) {
  const map = {};
  Object.keys(obj || {}).forEach(k => { map[normalizeLookupKey_(k)] = obj[k]; });
  for (let i = 0; i < aliases.length; i++) {
    const nk = normalizeLookupKey_(aliases[i]);
    if (Object.prototype.hasOwnProperty.call(map, nk) && map[nk] !== "" && map[nk] !== null && map[nk] !== undefined) return map[nk];
  }
  const normKeys = Object.keys(map);
  for (let i = 0; i < aliases.length; i++) {
    const ak = normalizeLookupKey_(aliases[i]);
    const hit = normKeys.find(k => k.indexOf(ak) >= 0 || ak.indexOf(k) >= 0);
    if (hit && map[hit] !== "" && map[hit] !== null && map[hit] !== undefined) return map[hit];
  }
  return "";
}

function normalizeLookupKey_(s) {
  return String(s || "").toLowerCase().replace(/[\s_\-\/\\(\)\[\]\{\}\.。:：]/g, "").trim();
}

function parseLooseNumber_(v) {
  if (v === null || v === undefined) return "";
  if (typeof v === 'number') return Number.isFinite(v) ? v : "";
  const s = String(v).trim();
  if (!s || s === '-' || s === '--') return "";
  const m = s.replace(/[,，\s]/g, '').match(/-?\d+(?:\.\d+)?/);
  if (!m) return "";
  const n = Number(m[0]);
  return Number.isFinite(n) ? n : "";
}

function normalizeLooseDate_(v) {
  if (!v) return '';
  if (Object.prototype.toString.call(v) === '[object Date]' && !isNaN(v.getTime())) return Utilities.formatDate(v, Session.getScriptTimeZone(), 'yyyy-MM-dd');
  const s = String(v).trim();
  if (!s) return '';
  if (/^\d{4}-\d{2}-\d{2}/.test(s)) return s.slice(0,10);
  if (/^\d{4}[\.\/]\d{1,2}[\.\/]\d{1,2}/.test(s)) {
    const parts = s.split(/[\sT]/)[0].split(/[\.\/]/);
    return `${parts[0]}-${('0'+parts[1]).slice(-2)}-${('0'+parts[2]).slice(-2)}`;
  }
  if (/^\d{2,3}[\.\/-]\d{1,2}[\.\/-]\d{1,2}/.test(s)) {
    const parts = s.split(/[\sT]/)[0].split(/[\.\/-]/);
    const rocYear = Number(parts[0]);
    if (Number.isFinite(rocYear)) {
      const year = rocYear < 1911 ? (rocYear + 1911) : rocYear;
      return `${year}-${('0'+parts[1]).slice(-2)}-${('0'+parts[2]).slice(-2)}`;
    }
  }
  const dt = new Date(s);
  if (!isNaN(dt.getTime())) return Utilities.formatDate(dt, Session.getScriptTimeZone(), 'yyyy-MM-dd');
  return s;
}

function applyReferencePricesFromRaw_(ss, rawRows) {
  const synSh = ensureSheet_(ss, 'Synonyms', ['raw_name','product_name','spec','product_id','enabled','note']);
  const synRows = readObjects_(synSh);
  const prodSh = ensureSheet_(ss, 'Products', ['id','name','sku','price','image','stock','category','date','cost','unit','safety_stock','supplier_ids','expiry_date','last_purchase_date','spec','reference_price','reference_price_date','reference_price_name','shop_enabled']);
  const values = prodSh.getDataRange().getValues();
  if (values.length <= 1) return { updated_count: 0, mapped_count: 0, unmatched_mapping_count: 0 };
  const headers = values[0].map(h => String(h || '').trim());
  const idx = headerIndexMap_(headers);

  const rawLookup = {};
  const addRawLookup_ = (key, row) => {
    const nk = normalizeLookupKey_(key);
    if (!nk) return;
    if (!rawLookup[nk]) rawLookup[nk] = [];
    rawLookup[nk].push(row);
  };

  rawRows.forEach(r => {
    const itemName = String(r.item_name || '').trim();
    const spec = String(r.spec || '').trim();
    if (!itemName) return;
    addRawLookup_(itemName, r);
    if (spec) {
      addRawLookup_(`${itemName}-${spec}`, r);
    }
  });

  Object.keys(rawLookup).forEach(k => rawLookup[k].sort((a, b) => {
    const dateCmp = String(b.market_date || '').localeCompare(String(a.market_date || ''));
    if (dateCmp !== 0) return dateCmp;
    return marketSortPriority_(b.market) - marketSortPriority_(a.market);
  }));

  let updatedCount = 0, mappedCount = 0, unmatchedMappingCount = 0;

  const chooseReferencePrice_ = (hit) => {
    if (hit.upper_price !== '' && hit.upper_price !== null && hit.upper_price !== undefined) return hit.upper_price;
    if (hit.middle_price !== '' && hit.middle_price !== null && hit.middle_price !== undefined) return hit.middle_price;
    if (hit.lower_price !== '' && hit.lower_price !== null && hit.lower_price !== undefined) return hit.lower_price;
    return '';
  };

  const setRowPrice_ = (rowNo, hit, hitName) => {
    const pickedPrice = chooseReferencePrice_(hit);
    if (idx.reference_price !== undefined) values[rowNo][idx.reference_price] = pickedPrice;
    if (idx.reference_price_date !== undefined) values[rowNo][idx.reference_price_date] = String(hit.market_date || '');
    if (idx.reference_price_name !== undefined) values[rowNo][idx.reference_price_name] = String(hitName || hit.item_name || '');
    if (pickedPrice !== '' && idx.price !== undefined) values[rowNo][idx.price] = pickedPrice;
    updatedCount++;
  };

  const findProductRow_ = (productId, productName, spec) => {
    const pid = String(productId || '').trim();
    const pname = String(productName || '').trim();
    const pspec = String(spec || '').trim();
    for (let r = 1; r < values.length; r++) {
      if (pid && idx.id !== undefined && String(values[r][idx.id] || '').trim() === pid) return r;
      const rowName = idx.name !== undefined ? String(values[r][idx.name] || '').trim() : '';
      const rowSpec = idx.spec !== undefined ? String(values[r][idx.spec] || '').trim() : '';
      if (pname && rowName === pname && (!pspec || rowSpec === pspec)) return r;
    }
    return -1;
  };

  const enabledValue_ = (rowObj) => {
    const v = pickValueByAliases_(rowObj, ['enabled','enable','active','啟用','是否啟用']);
    if (v === '' || v === null || v === undefined) return true;
    const s = String(v).trim().toLowerCase();
    return !(s === '0' || s === 'false' || s === 'no' || s === 'n' || s === '停用' || s === '否');
  };

  if (synRows.length) {
    synRows.forEach(row => {
      if (!enabledValue_(row)) return;
      const rawName = String(pickValueByAliases_(row, ['raw_name','market_name','raw','alias','synonym','keyword','原始名稱','外部名稱','行情名稱','對照名稱','比對名稱']) || '').trim();
      const productName = String(pickValueByAliases_(row, ['product_name','product','商品名稱','商品','系統商品','對應商品','name']) || '').trim();
      const spec = String(pickValueByAliases_(row, ['spec','規格','對應規格']) || '').trim();
      const productId = String(pickValueByAliases_(row, ['product_id','id','商品id','商品ID']) || '').trim();
      const lookupName = rawName || productName;
      if (!lookupName) return;
      const hits = (rawLookup[normalizeLookupKey_(lookupName)] || []).filter(hit => !spec || String(hit.spec || '').trim() === spec);
      if (!hits.length) { unmatchedMappingCount++; return; }
      const rowNo = findProductRow_(productId, productName || lookupName, spec);
      if (rowNo < 0) { unmatchedMappingCount++; return; }
      mappedCount++;
      setRowPrice_(rowNo, hits[0], lookupName);
    });
  } else {
    for (let r = 1; r < values.length; r++) {
      const rowName = idx.name !== undefined ? String(values[r][idx.name] || '').trim() : '';
      const rowSpec = idx.spec !== undefined ? String(values[r][idx.spec] || '').trim() : '';
      if (!rowName) continue;
      const baseHits = (rawLookup[normalizeLookupKey_(rowName)] || []).filter(hit => !rowSpec || String(hit.spec || '').trim() === rowSpec);
      const combinedHits = rowSpec ? (rawLookup[normalizeLookupKey_(`${rowName}-${rowSpec}`)] || []) : [];
      const hits = baseHits.length ? baseHits : combinedHits;
      if (!hits.length) continue;
      mappedCount++;
      setRowPrice_(r, hits[0], rowName);
    }
  }

  prodSh.getRange(1, 1, values.length, values[0].length).setValues(values);
  return { updated_count: updatedCount, mapped_count: mappedCount, unmatched_mapping_count: unmatchedMappingCount };
}


function buildMarketPriceBoard_(ss) {
  const rawSh = ensureSheet_(ss, "PingtungRaw", PINGTUNG_RAW_HEADERS_);
  const synSh = ensureSheet_(ss, 'Synonyms', ['raw_name','product_name','spec','product_id','enabled','note']);
  const prodSh = ensureSheet_(ss, 'Products', ['id','name','sku','price','image','stock','category','date','cost','unit','safety_stock','supplier_ids','expiry_date','last_purchase_date','spec','reference_price','reference_price_date','reference_price_name','shop_enabled']);

  const rawRows = readObjects_(rawSh).map(r => ({
    sync_time: String(r.sync_time || '').trim(),
    source_file: String(r.source_file || '').trim(),
    market_date: normalizeLooseDate_(r.market_date || ''),
    item_name: String(r.item_name || '').trim(),
    spec: String(r.spec || '').trim(),
    market: normalizeMarketLabel_(r.market || ''),
    upper_price: parseLooseNumber_(r.upper_price),
    middle_price: parseLooseNumber_(r.middle_price),
    lower_price: parseLooseNumber_(r.lower_price),
    raw_json: String(r.raw_json || '')
  })).filter(r => r.item_name);
  const synRows = readObjects_(synSh);
  const products = readObjects_(prodSh).map(p => ({
    id: String(p.id || '').trim(),
    name: String(p.name || '').trim(),
    sku: String(p.sku || p.part_no || p.code || '').trim(),
    spec: String(p.spec || '').trim(),
    unit: String(p.unit || '').trim(),
    category: String(p.category || '').trim(),
    cost: parseLooseNumber_(p.cost),
    price: parseLooseNumber_(p.price),
    last_purchase_date: normalizeLooseDate_(p.last_purchase_date || '')
  })).filter(p => p.name);

  const rawByMarket = buildRawLookupByMarket_(rawRows);
  const groups = { FirstMarket: [], SecondMarket: [], Pingtung: [] };

  products.forEach(product => {
    const aliases = getProductMarketAliases_(product, synRows);
    ['FirstMarket', 'SecondMarket', 'Pingtung'].forEach(marketKey => {
      const hit = chooseLatestHitByMarket_(rawByMarket[marketKey] || {}, aliases, product.name, product.spec);
      if (!hit) return;
      groups[marketKey].push({
        product_id: product.id,
        sku: product.sku,
        product_name: product.name,
        product_spec: product.spec,
        unit: product.unit,
        category: product.category,
        cost: product.cost,
        price: product.price,
        last_purchase_date: String(product.last_purchase_date || ''),
        market: marketKey,
        market_label: marketDisplayLabel_(marketKey),
        market_date: String(hit.market_date || ''),
        raw_name: String(hit.item_name || ''),
        raw_spec: String(hit.spec || ''),
        upper_price: hit.upper_price,
        middle_price: hit.middle_price,
        lower_price: hit.lower_price
      });
    });
  });

  Object.keys(groups).forEach(k => groups[k].sort((a, b) => {
    const catCmp = String(a.category || '').localeCompare(String(b.category || ''), 'zh-Hant');
    if (catCmp !== 0) return catCmp;
    const nameCmp = String(a.product_name || '').localeCompare(String(b.product_name || ''), 'zh-Hant');
    if (nameCmp !== 0) return nameCmp;
    return String(a.product_spec || '').localeCompare(String(b.product_spec || ''), 'zh-Hant');
  }));

  const latestSync = rawRows.reduce((acc, r) => String(r.sync_time || '') > acc ? String(r.sync_time || '') : acc, '');
  const latestMarketDate = rawRows.reduce((acc, r) => String(r.market_date || '') > acc ? String(r.market_date || '') : acc, '');
  const matchedCount = Object.keys(groups).reduce((sum, k) => sum + groups[k].length, 0);

  return {
    synced_at: latestSync,
    latest_market_date: latestMarketDate,
    raw_count: rawRows.length,
    product_count: products.length,
    matched_count: matchedCount,
    groups: groups
  };
}

function buildRawLookupByMarket_(rawRows) {
  const byMarket = { FirstMarket: {}, SecondMarket: {}, Pingtung: {} };
  const addHit_ = (marketKey, key, row) => {
    if (!byMarket[marketKey]) byMarket[marketKey] = {};
    const nk = normalizeLookupKey_(key);
    if (!nk) return;
    if (!byMarket[marketKey][nk]) byMarket[marketKey][nk] = [];
    byMarket[marketKey][nk].push(row);
  };

  rawRows.forEach(row => {
    const marketKey = normalizeMarketLabel_(row.market || '');
    if (!byMarket[marketKey]) return;
    const itemName = String(row.item_name || '').trim();
    const spec = String(row.spec || '').trim();
    if (!itemName) return;
    addHit_(marketKey, itemName, row);
    if (spec) addHit_(marketKey, `${itemName}-${spec}`, row);
  });

  Object.keys(byMarket).forEach(marketKey => {
    Object.keys(byMarket[marketKey]).forEach(key => {
      byMarket[marketKey][key].sort((a, b) => compareMarketHitDesc_(a, b));
    });
  });
  return byMarket;
}

function compareMarketHitDesc_(a, b) {
  const ad = String(a && a.market_date || '');
  const bd = String(b && b.market_date || '');
  const dateCmp = bd.localeCompare(ad);
  if (dateCmp !== 0) return dateCmp;
  const as = String(a && a.spec || '');
  const bs = String(b && b.spec || '');
  if (!!bs !== !!as) return bs ? 1 : -1;
  return String(b && b.item_name || '').localeCompare(String(a && a.item_name || ''), 'zh-Hant');
}

function marketDisplayLabel_(market) {
  const key = normalizeMarketLabel_(market);
  if (key === 'FirstMarket') return '台北第一果菜市場';
  if (key === 'SecondMarket') return '台北第二果菜市場';
  if (key === 'Pingtung') return '屏東果菜市場';
  return String(market || '');
}

function getProductMarketAliases_(product, synRows) {
  const pid = String(product && product.id || '').trim();
  const pname = String(product && product.name || '').trim();
  const pspec = String(product && product.spec || '').trim();
  const out = [];
  const seen = {};
  const pushAlias_ = (rawName, spec) => {
    const rn = String(rawName || '').trim();
    const sp = String(spec || '').trim();
    if (!rn) return;
    const key = `${normalizeLookupKey_(rn)}__${normalizeLookupKey_(sp)}`;
    if (seen[key]) return;
    seen[key] = true;
    out.push({ raw_name: rn, spec: sp });
  };

  synRows.forEach(row => {
    if (!enabledSynonymRow_(row)) return;
    const targetId = String(pickValueByAliases_(row, ['product_id','id','商品id','商品ID']) || '').trim();
    const targetName = String(pickValueByAliases_(row, ['product_name','product','商品名稱','商品','系統商品','對應商品','name']) || '').trim();
    const targetSpec = String(pickValueByAliases_(row, ['spec','規格','對應規格']) || '').trim();
    const rawName = String(pickValueByAliases_(row, ['raw_name','market_name','raw','alias','synonym','keyword','原始名稱','外部名稱','行情名稱','對照名稱','比對名稱']) || '').trim();

    let matched = false;
    if (pid && targetId && pid === targetId) matched = true;
    if (!matched && targetName && pname === targetName) {
      matched = (!targetSpec || !pspec || targetSpec === pspec);
    }
    if (!matched) return;
    pushAlias_(rawName || targetName || pname, targetSpec || pspec);
  });

  pushAlias_(pname, pspec);
  return out;
}

function enabledSynonymRow_(rowObj) {
  const v = pickValueByAliases_(rowObj, ['enabled','enable','active','啟用','是否啟用']);
  if (v === '' || v === null || v === undefined) return true;
  const s = String(v).trim().toLowerCase();
  return !(s === '0' || s === 'false' || s === 'no' || s === 'n' || s === '停用' || s === '否');
}

function chooseLatestHitByMarket_(marketLookup, aliases, productName, productSpec) {
  let best = null;
  const candidateAliases = Array.isArray(aliases) && aliases.length ? aliases : [{ raw_name: productName, spec: productSpec }];
  candidateAliases.forEach(alias => {
    const rawName = String(alias && alias.raw_name || productName || '').trim();
    const targetSpec = String(alias && alias.spec || productSpec || '').trim();
    if (!rawName) return;

    let hits = [];
    const byName = marketLookup[normalizeLookupKey_(rawName)] || [];
    const byCombo = targetSpec ? (marketLookup[normalizeLookupKey_(`${rawName}-${targetSpec}`)] || []) : [];
    if (targetSpec) {
      const exact = byName.filter(hit => String(hit.spec || '').trim() === targetSpec);
      hits = exact.length ? exact.concat(byCombo) : byCombo.concat(byName);
    } else {
      hits = byName.concat(byCombo);
    }

    hits.forEach(hit => {
      if (!best || compareMarketHitDesc_(hit, best) < 0) {
        best = hit;
      }
    });
  });
  return best;
}

      id: id,
      customer_id: id,
      name: name,
      username: (customer && customer.username) || p.username || "",
      password: (customer && customer.password) || p.password || "",
      role: (customer && customer.role) || p.role || "customer",
      phone: (customer && customer.phone) || p.phone || "",
      email: (customer && customer.email) || p.email || "",
      address: (customer && customer.address) || p.address || "",
      note: (customer && customer.note) || p.note || "",
      created_at: new Date()
    };
    appendObject_(sh, obj);
    return respond({ status: "ok", message: "客戶新增成功", id: id });
  }

  if (action === "update"){
    const id = String(p.id || "").trim();
    if (!id) return respond({ status: "error", message: "缺少客戶 id" });

    const rows = sh.getDataRange().getValues();
    for (let r = 1; r < rows.length; r++){
      if (String(rows[r][idx.id]).trim() === id){
        if (p.name !== undefined && idx.name !== undefined) sh.getRange(r + 1, idx.name + 1).setValue(p.name);
        if (p.username !== undefined && idx.username !== undefined) sh.getRange(r + 1, idx.username + 1).setValue(p.username);
        if (p.password !== undefined && idx.password !== undefined) sh.getRange(r + 1, idx.password + 1).setValue(p.password);
        if (p.role !== undefined && idx.role !== undefined) sh.getRange(r + 1, idx.role + 1).setValue(p.role);
        if (p.phone !== undefined && idx.phone !== undefined) sh.getRange(r + 1, idx.phone + 1).setValue(p.phone);
        if (p.address !== undefined && idx.address !== undefined) sh.getRange(r + 1, idx.address + 1).setValue(p.address);
        if (p.note !== undefined && idx.note !== undefined) sh.getRange(r + 1, idx.note + 1).setValue(p.note);
        return respond({ status: "ok", message: "客戶更新成功" });
      }
    }
    return respond({ status: "error", message: "找不到客戶" });
  }

  if (action === "delete"){
    const id = String(p.id || "").trim();
    if (!id) return respond({ status: "error", message: "缺少客戶 id" });
    const rows = sh.getDataRange().getValues();
    for (let r = rows.length - 1; r >= 1; r--){
      if (String(rows[r][idx.id]).trim() === id){
        sh.deleteRow(r + 1);
        return respond({ status: "ok", message: "客戶刪除成功" });
      }
    }
    return respond({ status: "error", message: "找不到客戶" });
  }

  return respond({ status: "error", message: "Invalid action" });
}

function findOrderIdByLinkedPurchaseId_(ss, poId){
  const target = String(poId || "").trim();
  if (!target) return "";
  const sh = ensureSheet_(ss, "Orders", orderHeaders_());
  const rows = sh.getDataRange().getValues();
  if (rows.length <= 1) return "";
  const idx = headerIndexMap_(rows[0].map(h => String(h || "").trim()));
  if (idx.order_id === undefined || idx.linked_purchase_id === undefined) return "";
  for (let r = 1; r < rows.length; r++) {
    if (String(rows[r][idx.linked_purchase_id] || "").trim() === target) {
      return String(rows[r][idx.order_id] || "").trim();
    }
  }
  return "";
}

function ensurePurchasesSheet_(ss){
  const expected = purchaseHeaders_();
  const sh = ensureSheet_(ss, "purchases", expected);
  const current = getHeaders_(sh);
  const normalizedCurrent = current.map(h => String(h || "").trim());
  const hasDup = normalizedCurrent.some((h, i) => !!h && normalizedCurrent.indexOf(h) !== i);
  const sameOrder = expected.length === normalizedCurrent.length && expected.every((h, i) => normalizedCurrent[i] === h);
  if (sameOrder && !hasDup) return sh;

  const vals = sh.getDataRange().getValues();
  const currentHeaders = (vals[0] || []).map(h => String(h || "").trim());
  const firstIdx = {};
  currentHeaders.forEach((h, i) => {
    if (h && firstIdx[h] === undefined) firstIdx[h] = i;
  });
  const aliases = {
    po_id: ["po_id", "purchase_id", "purchaseId", "id", "ref_id"],
    date: ["date"],
    arrival_date: ["arrival_date", "receive_date"],
    form_no: ["form_no", "form_code"],
    form_name: ["form_name", "form_title"],
    supplier_id: ["supplier_id", "supplierId", "supplier"],
    supplier_name: ["supplier_name", "supplierName"],
    total: ["total"],
    items: ["items"],
    expiry_date: ["expiry_date"],
    status: ["status"],
    operator: ["operator"],
    created_at: ["created_at"],
    source_order_id: ["source_order_id"],
    stock_applied: ["stock_applied"],
    completed_at: ["completed_at"],
    auto_generated: ["auto_generated"]
  };

  const out = [expected.slice()];
  for (let r = 1; r < vals.length; r++) {
    const row = vals[r] || [];
    if (row.every(v => v === "" || v === null)) continue;
    const obj = {};
    expected.forEach((h) => {
      const names = aliases[h] || [h];
      let hitIdx = -1;
      for (let i = 0; i < names.length; i++) {
        const idx = firstIdx[names[i]];
        if (idx !== undefined) { hitIdx = idx; break; }
      }
      obj[h] = hitIdx >= 0 ? row[hitIdx] : "";
    });

    if (!String(obj.form_no || "").trim()) {
      const info = getPurchaseFormInfoByCategory_(String(obj.form_name || "").trim());
      obj.form_no = info.form_no;
    }
    if (!String(obj.form_name || "").trim()) {
      const info = getPurchaseFormInfoByCategory_(String(obj.form_name || "").trim());
      obj.form_name = info.form_name;
    }
    if (!String(obj.source_order_id || "").trim() && Number(obj.auto_generated || 0)) {
      const linkedOrderId = findOrderIdByLinkedPurchaseId_(ss, obj.po_id);
      if (linkedOrderId) obj.source_order_id = linkedOrderId;
    }

    out.push(expected.map(h => Object.prototype.hasOwnProperty.call(obj, h) ? obj[h] : ""));
  }

  const maxRows = sh.getMaxRows();
  const maxCols = sh.getMaxColumns();
  if (maxRows < out.length) sh.insertRowsAfter(maxRows, out.length - maxRows);
  if (maxCols < expected.length) sh.insertColumnsAfter(maxCols, expected.length - maxCols);
  sh.clearContents();
  sh.getRange(1, 1, out.length, expected.length).setValues(out);
  if (sh.getMaxColumns() > expected.length) {
    sh.deleteColumns(expected.length + 1, sh.getMaxColumns() - expected.length);
  }
  return sh;
}

function getOrderDateById_(ss, orderId) {

  for (let r = 1; r < rows.length; r++) {

  }
  return "";
}

function getPurchaseFormInfoByCategory_(category){
  const s = String(category || "").trim();
  if (/冷凍/.test(s)) return { form_no: "F-02-B-01-2", form_name: "冷凍食品類" };
  if (/(乾貨|素料)/.test(s)) return { form_no: "F-02-B-01-4", form_name: "乾貨素料類" };
  if (/(南北|雜貨|包材)/.test(s)) return { form_no: "F-02-B-01-3", form_name: "南北雜貨包材類" };
  return { form_no: "F-02-B-01-1", form_name: "生鮮蔬果類" };
}

function parseSupplierIdsFromProductRow_(row, pidx){
  const set = {};
  if (pidx.supplier_id !== undefined) {
    const v = String(row[pidx.supplier_id] || "").trim();
    if (v) set[v] = true;
  }
  if (pidx.supplier_ids !== undefined) {
    const raw = String(row[pidx.supplier_ids] || "").trim();
    if (raw) raw.split(",").map(s => String(s).trim()).filter(Boolean).forEach(v => set[v] = true);
  }
  return Object.keys(set);
}

function applyPurchaseStock_(ss, prodSh, po_id, items, operator, dateValue){
  const ph = getHeaders_(prodSh);
  const pidx = headerIndexMap_(ph);
  const prodVals = prodSh.getDataRange().getValues();
  const byId = {};
  for (let r = 1; r < prodVals.length; r++) {
    byId[String(prodVals[r][pidx.id]).trim()] = { r: r, row: prodVals[r] };
  }
  const now = new Date();
  const purchaseDateIso = toISODateLocal_(dateValue || now);
  for (let i = 0; i < items.length; i++) {
    const it = items[i] || {};
    const pid = String(it.product_id || it.id || "").trim();
    const qtyRaw = (it.qty_raw !== undefined && it.qty_raw !== null && String(it.qty_raw).trim() !== "") ? it.qty_raw : it.qty;
    const costRaw = (it.cost_raw !== undefined && it.cost_raw !== null && String(it.cost_raw).trim() !== "") ? it.cost_raw : (it.cost ?? 0);
    const qty = cleanDecimalInput_(qtyRaw, 6);
    const unitCost = cleanDecimalInput_(costRaw, 6);
    if (!pid || !(qty > 0)) throw new Error("品項資料不完整，無法完成驗收入庫");
    const hit = byId[pid];
    if (!hit) throw new Error(`找不到商品 id=${pid}`);
    const current = cleanDecimalInput_(hit.row[pidx.stock] || 0, 6);
    const next = addDecimalInput_(current, qty, 6);
    prodSh.getRange(hit.r + 1, pidx.stock + 1).setValue(next);
    if (isFinite(unitCost) && unitCost > 0 && pidx.cost !== undefined) {
      prodSh.getRange(hit.r + 1, pidx.cost + 1).setValue(unitCost);
      hit.row[pidx.cost] = unitCost;
    }
    if (purchaseDateIso && pidx.last_purchase_date !== undefined) {
      const oldLast = toISODateLocal_(hit.row[pidx.last_purchase_date] || "");
      const finalLast = (!oldLast || oldLast < purchaseDateIso) ? purchaseDateIso : oldLast;
      prodSh.getRange(hit.r + 1, pidx.last_purchase_date + 1).setValue(finalLast);
      hit.row[pidx.last_purchase_date] = finalLast;
    }
    const itemExpiry = toISODateLocal_(it.expiry_date || it.expire_date || it.expiry || it.valid_date || it.valid_until || "");
    if (itemExpiry && pidx.expiry_date !== undefined) {
      prodSh.getRange(hit.r + 1, pidx.expiry_date + 1).setValue(itemExpiry);
      hit.row[pidx.expiry_date] = itemExpiry;
      it.expiry_date = itemExpiry;
    }
    writeLedger_(ss, {
      ledger_id: generateId_("LG"),
      product_id: pid,
      product_name: String(hit.row[pidx.name] || it.product_name || ""),
      sku: (pidx.sku !== undefined) ? String(hit.row[pidx.sku] || it.sku || "") : String(it.sku || ""),
      unit: (pidx.unit !== undefined) ? String(hit.row[pidx.unit] || it.unit || "") : String(it.unit || ""),
      change: qty,
      direction: "IN",
      reason: "purchase",
      ref_id: String(po_id),
      operator: operator,
      date: toLocalDateTime_(now),
      cost: (isFinite(unitCost) && unitCost > 0) ? unitCost : (pidx.cost !== undefined ? (Number(hit.row[pidx.cost] || 0) || "") : ""),
      target: String(it.supplier_name || "")
    });
  }
}

function applyOrderShipment_(ss, prodSh, order_id, items, target, operator){
  const ph = getHeaders_(prodSh);
  const pidx = headerIndexMap_(ph);
  const prodVals = prodSh.getDataRange().getValues();
  const byId = {};
  for (let r = 1; r < prodVals.length; r++) {
    byId[String(prodVals[r][pidx.id]).trim()] = { r: r, row: prodVals[r] };
  }
  const now = new Date();
  for (let i = 0; i < items.length; i++) {
    const it = items[i] || {};
    const pid = String(it.product_id || it.id || "").trim();
    const qty = cleanDecimalInput_(it.qty || it.quantity || 0, 6);
    if (!pid || !(qty > 0)) throw new Error("訂單品項資料不完整");
    const hit = byId[pid];
    if (!hit) throw new Error(`找不到商品 id=${pid}`);
    const current = cleanDecimalInput_(hit.row[pidx.stock] || 0, 6);
    const next = addDecimalInput_(current, -qty, 6);
    if (next < 0) throw new Error(`庫存不足：${hit.row[pidx.name]} current=${current}, qty=${qty}`);
    prodSh.getRange(hit.r + 1, pidx.stock + 1).setValue(next);
    writeLedger_(ss, {
      ledger_id: generateId_("LG"),
      product_id: pid,
      product_name: String(hit.row[pidx.name] || it.name || ""),
      sku: (pidx.sku !== undefined) ? String(hit.row[pidx.sku] || "") : "",
      unit: (pidx.unit !== undefined) ? String(hit.row[pidx.unit] || "") : "",
      change: -qty,
      direction: "OUT",
      reason: "order",
      ref_id: String(order_id),
      operator: operator,
      date: toLocalDateTime_(now),
      cost: (pidx.cost !== undefined) ? (Number(hit.row[pidx.cost] || 0) || "") : "",
      target: target
    });
  }
}

function apiPurchases_(p, respond) {
  if (typeof p === "function") {
    respond = p;
    p = {};
  }
  p = p || {};

  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sh = ensurePurchasesSheet_(ss);
  const rawList = readObjects_(sh);
  const targetPoId = String(p.po_id || p.id || "").trim();
  const includeItems = targetPoId || Number(p.detail || p.include_items || 0) ? 1 : 0;

  const list = rawList.map((x, i) => {
    x.date = toISODateLocal_(x.date);
    x.arrival_date = toISODateLocal_(x.arrival_date || x.receive_date || "");
    x.created_at = toISODateLocal_(x.created_at);
    x.completed_at = toISODateLocal_(x.completed_at || "");

    x.form_no = x.form_no || x.form_code || "";
    x.form_name = x.form_name || x.form_title || "";
    x.supplier_id = x.supplier_id || x.supplierId || x.supplier || "";
    x.supplier_name = x.supplier_name || x.supplierName || "";
    x.status = String(x.status || "待驗收").trim() || "待驗收";
    x.stock_applied = Number(x.stock_applied || 0) ? 1 : 0;
    x.source_order_id = String(x.source_order_id || "").trim();
    x.auto_generated = Number(x.auto_generated || 0) ? 1 : 0;
    if (!x.source_order_id) {
      const fallbackSourceOrderId = String(findOrderIdByLinkedPurchaseId_(ss, x.po_id) || "").trim();
      if (fallbackSourceOrderId) {
        x.source_order_id = fallbackSourceOrderId;
        if (!x.auto_generated) x.auto_generated = 1;
      }
    }
    if (!String(x.form_no || "").trim()) {
      const info = getPurchaseFormInfoByCategory_(String(x.form_name || "").trim());
      x.form_no = info.form_no;
      if (!String(x.form_name || "").trim()) x.form_name = info.form_name;
    }
    if (x.auto_generated && x.source_order_id) {
      const linkedOrderDate = getOrderDateById_(ss, x.source_order_id);
      if (linkedOrderDate) x.date = linkedOrderDate;
    }

    const raw = x.items;
    if (includeItems) {
      if (raw && typeof raw === "string") {
        try { x.items = JSON.parse(raw); } catch (e) { x.items = []; }
      }
      if (!Array.isArray(x.items)) x.items = [];
      x.items_loaded = 1;
      x.item_count = x.items.length;
    } else {
      x.items = [];
      x.items_loaded = 0;
      x.item_count = 0;
    }

    if (!String(x.po_id || "").trim()) {
      const hasPurchaseLikeData = String(x.date || "").trim() || String(x.supplier_id || x.supplier_name || "").trim() || String(x.total || "").trim() || String(raw || "").trim();
      if (hasPurchaseLikeData) x.po_id = `ROW${i + 2}`;
    }
    return x;
  }).filter(x => String(x.po_id || "").trim() !== "");

  const filtered = targetPoId ? list.filter(x => String(x.po_id || "").trim() === targetPoId) : list;
  return respond({ status: "ok", data: filtered, count: filtered.length });
}

          qty: qty,
          suggested_qty: cleanDecimalInput_(it.suggested_qty || qty, 6),
          customer_order_qty: (function(){

      const headerSupplierName = String(payload.supplier_name || ((uniqSupNames.length === 1) ? uniqSupNames[0] : "多供應商")).trim();
      const applyStockNow = !!payload.apply_stock && !oldStockApplied;
      if (applyStockNow) {
        applyPurchaseStock_(ss, prodSh, po_id, normalizedItems, operator, date);
      }

      const rowObj = {
        po_id: po_id,
        date: date,
        arrival_date: arrival_date,
        form_no: form_no,
        form_name: form_name,

        status: applyStockNow ? "已入庫" : String(payload.status || (oldStockApplied ? "已入庫" : "待驗收")).trim(),
        operator: operator,
        created_at: foundRow ? (rows[foundRow - 1][idx.created_at] || new Date()) : new Date(),
        source_order_id: linkedSourceOrderId,
        stock_applied: applyStockNow || oldStockApplied ? 1 : 0,
        completed_at: (applyStockNow ? new Date() : (foundRow ? rows[foundRow - 1][idx.completed_at] : "")),
        auto_generated: linkedAutoGenerated ? 1 : 0
      };

      if (foundRow) {
        setSheetRowByObject_(purchasesSh, foundRow, rowObj);
      } else {
        appendObject_(purchasesSh, rowObj);
      }

      return respond({ status: "ok", message: applyStockNow ? `採購驗收單已完成：${po_id}` : `採購驗收單已儲存：${po_id}`, po_id: po_id });
    }

    if (action === "delete") {
      const po_id = String(p.po_id || p.id || "").trim();
      if (!po_id) return respond({ status: "error", message: "缺少 po_id" });
      const rows = purchasesSh.getDataRange().getValues();
      const headers = rows[0].map(h => String(h || "").trim());
      const idx = headerIndexMap_(headers);
      for (let r = 1; r < rows.length; r++) {
        if (String(rows[r][idx.po_id]).trim() === po_id) {
          purchasesSh.deleteRow(r + 1);
          return respond({ status: "ok", message: "採購驗收單已刪除（若已入庫，不會自動回滾庫存）" });
        }
      }
      return respond({ status: "error", message: "找不到進貨單" });
    }

    return respond({ status: "error", message: "未知 action" });
  } finally {
    try { lock.releaseLock(); } catch (e2) {}
  }
}

/**
 * 領貨單號：yyyyMMdd + 4位流水（例：202602090001）
 * - 會同時掃描 pickups 上可能的單號欄位（pickup_id / id / doc_no / docNo / pickupId）
 * - 避免「舊資料單號在別欄位」造成序號重複
 */
function generatePickupDocNo_(pickupsSh, dateObj){
  const d = dateObj instanceof Date ? dateObj : new Date();
  const prefix = Utilities.formatDate(d, "Asia/Taipei", "yyyyMMdd");

  const headers = getHeaders_(pickupsSh);
  const idx = headerIndexMap_(headers);

  const candidates = ["pickup_id","id","doc_no","docNo","pickupId"];
  const cols = candidates.map(k => idx[k]).filter(v => v !== undefined);

  if (!cols.length) {
    // 若真的沒有任何欄位，就先確保 pickup_id 存在（已在 ensurePickupsSheet_ 保障）
    return generateDailyDocNo_(pickupsSh, "pickup_id", d);
  }

  const lastRow = pickupsSh.getLastRow();
  let maxSeq = 0;

  if (lastRow >= 2) {
    // 逐欄掃描（避免抓整表太大）
    for (let c = 0; c < cols.length; c++){
      const col = cols[c] + 1;
      const vals = pickupsSh.getRange(2, col, lastRow - 1, 1).getValues();
      for (let i = 0; i < vals.length; i++){
        const s = String(vals[i][0] || "").trim();
        if (!s) continue;
        if (s.length >= 12 && s.indexOf(prefix) === 0) {
          const suf = parseInt(s.slice(8), 10);
          if (!isNaN(suf)) maxSeq = Math.max(maxSeq, suf);
        }
      }
    }
  }

  const next = maxSeq + 1;
  const suf4 = ("0000" + next).slice(-4);
  return prefix + suf4;
}

function apiManagePickup_(p, respond){
  const action = String(p.action || "").trim(); // add/delete
  const ss = SpreadsheetApp.getActiveSpreadsheet();

  const pickupsSh = ensurePickupsSheet_(ss);
  const prodSh = ensureSheet_(ss, "Products", ["id","name","sku","price","image","stock","category","date","cost","unit","safety_stock","supplier_ids","expiry_date","last_purchase_date","spec","reference_price","reference_price_date","reference_price_name","shop_enabled"]);

  const lock = LockService.getScriptLock();
  lock.waitLock(10000);

  try{
    if (action === "add"){
      const payload = mustJson_(p.pickup);
      const department = String(payload.department || payload.dept || "").trim();
      const receiver = String(payload.receiver || "").trim();
      const note = String(payload.note || "").trim();
      const items = Array.isArray(payload.items) ? payload.items : [];
      const operator = String(payload.operator || p.operator || "").trim();

      if (!department) return respond({ status:"error", message:"缺少領用單位/門市（department）" });
      if (!items.length) return respond({ status:"error", message:"缺少 items" });

      // 領貨單號：yyyyMMdd + 4位流水（例：202602090001）
      const pickup_id = String(generatePickupDocNo_(pickupsSh, new Date())).trim();

      // products map
      const prodVals = prodSh.getDataRange().getValues();
      const ph = prodVals[0].map(h => String(h || "").trim());
      const pidx = headerIndexMap_(ph);
      const byId = {};
      for (let r = 1; r < prodVals.length; r++){
        byId[String(prodVals[r][pidx.id]).trim()] = { r: r, row: prodVals[r] };
      }

      const now = new Date();
      let total = 0;

      // 扣庫存 + 寫 ledger
      for (let i = 0; i < items.length; i++){
        const it = items[i];
        const pid = String(it.product_id || it.id || "").trim();
        const qty = Number(it.qty || it.quantity || 0);
        const unitCost = (it.cost !== undefined && it.cost !== "") ? Number(it.cost) : Number(it.unit_cost || 0);

        if (!pid || !qty || qty <= 0) return respond({ status:"error", message:"品項資料不完整（product_id/qty）" });

        const hit = byId[pid];
        if (!hit) return respond({ status:"error", message:`找不到商品 id=${pid}` });

        const current = Number(hit.row[pidx.stock] || 0);
        const next = current - qty;
        if (next < 0) return respond({ status:"error", message:`庫存不足：${hit.row[pidx.name]} current=${current}, qty=${qty}` });

        // update stock
        prodSh.getRange(hit.r + 1, pidx.stock + 1).setValue(next);

        const pname = String(hit.row[pidx.name] || "");
        const target = receiver ? `${department} / ${receiver}` : department;

        writeLedger_(ss, {
          ledger_id: generateId_("LG"),
          product_id: pid,
          product_name: pname,
                    sku: (pidx.sku !== undefined) ? String(hit.row[pidx.sku] || "") : "",
          unit: (pidx.unit !== undefined) ? String(hit.row[pidx.unit] || "") : "",
change: -qty,                // signed change: out = negative
          direction: "OUT",
          reason: "pickup",
          ref_id: pickup_id,
          operator: operator,
          date: toLocalDateTime_(now),
          cost: (pidx.cost !== undefined) ? (Number(hit.row[pidx.cost] || 0) || "") : "",
          target: target
        });

        // total cost (optional)
        if (!isNaN(unitCost) && unitCost > 0) total += unitCost * qty;
      }

      // 寫 pickups（items 存 JSON）
      // 同時寫入 pickup_id 與 id（若有）以兼容舊表頭（appendObject_ 只會寫入表頭存在的欄位）
      appendObject_(pickupsSh, {
        pickup_id: pickup_id,
        id: pickup_id,
        doc_no: pickup_id,
        date: payload.date || toROCDateString_(new Date()),
        department: department,
        receiver: receiver,
        total: total,
        items: JSON.stringify(items),
        note: note,
        operator: operator,
        created_at: now
      });

      return respond({ status:"ok", message:`領貨完成：${pickup_id}`, pickup_id: pickup_id });
    }

    if (action === "delete"){
      const pickup_id = String(p.pickup_id || p.id || "").trim();
      if (!pickup_id) return respond({ status:"error", message:"缺少 pickup_id" });

      const rows = pickupsSh.getDataRange().getValues();
      const headers = rows[0].map(h => String(h || "").trim());
      const idx = headerIndexMap_(headers);

      const candidates = ["pickup_id","id","doc_no","docNo","pickupId"];
      const cols = candidates.map(k => idx[k]).filter(v => v !== undefined);
      const col = cols.length ? cols[0] : 0;

      for (let r = 1; r < rows.length; r++){
        if (String(rows[r][col] || "").trim() === pickup_id){
          pickupsSh.deleteRow(r + 1);
          return respond({ status:"ok", message:"領貨單已刪除（未回滾庫存）" });
        }
      }
      return respond({ status:"error", message:"找不到領貨單" });
    }

    return respond({ status:"error", message:"未知 action" });
  } finally {
    try { lock.releaseLock(); } catch(e2) {}
  }
}

function apiManageOrder_(p, respond) {
  const action = String(p.action || "").trim();
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sh = ensureSheet_(ss, "Orders", orderHeaders_());
  const prodSh = ensureSheet_(ss, "Products", ["id","name","sku","price","image","stock","category","date","cost","unit","safety_stock","supplier_ids","expiry_date","last_purchase_date","spec","reference_price","reference_price_date","reference_price_name","shop_enabled"]);

    const status = String(p.status || "").trim();
    if (!status) return respond({ status: "error", message: "缺少 status" });
    for (let r = 1; r < rows.length; r++) {
      if (String(rows[r][idx.order_id]).trim() === order_id) {
        const source = String(rows[r][idx.source] || "").trim();
        const linkedPurchaseId = String(rows[r][idx.linked_purchase_id] || "").trim();
        const stockApplied = Number(rows[r][idx.stock_applied] || 0) ? 1 : 0;
        let items = rows[r][idx.items];
        if (typeof items === "string" && items) {
          try { items = JSON.parse(items); } catch (e) { items = []; }
        }
        if (!Array.isArray(items)) items = [];

        const incomingOperator = String(p.operator || "").trim();
        const existingOperator = String(rows[r][idx.operator] || "").trim();
        const shipmentOperator = incomingOperator || existingOperator;

        if ((status === "已出貨" || status === "已完成") && !stockApplied && (source === "EC" || linkedPurchaseId)) {
          applyOrderShipment_(ss, prodSh, order_id, items, String(rows[r][idx.name] || "").trim(), shipmentOperator);
          if (idx.stock_applied !== undefined) sh.getRange(r + 1, idx.stock_applied + 1).setValue(1);
        }
        if (idx.status !== undefined) sh.getRange(r + 1, idx.status + 1).setValue(status);
        if (idx.operator !== undefined && incomingOperator) {
          if (status === "已出貨" || (!existingOperator && status === "已完成")) {
            sh.getRange(r + 1, idx.operator + 1).setValue(incomingOperator);
          }
        }
        return respond({ status: "ok", message: "訂單已更新" });
      }
    }
    return respond({ status: "error", message: "找不到訂單" });
  }

  if (action === "delete") {
    for (let r = 1; r < rows.length; r++) {
      if (String(rows[r][idx.order_id]).trim() === order_id) {
        sh.deleteRow(r + 1);
        return respond({ status: "ok", message: "訂單已刪除（未回滾庫存）" });
      }
    }
    return respond({ status: "error", message: "找不到訂單" });
  }

  return respond({ status: "error", message: "未知 action" });
}

function apiProductLedger_(p, respond) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sh = ensureLedgerSheet_(ss);
  const pid = String(p.product_id || p.id || "").trim();
  const sku = String(p.sku || "").trim();
  if (!sku && !pid) return respond({ status: "error", message: "缺少 sku 或 product_id" });

  const limit = Math.max(1, Math.min(1000, Number(p.limit || 300)));

  const from = String(p.from || "").trim(); // yyyy-MM-dd (optional)
  const to = String(p.to || "").trim();     // yyyy-MM-dd (optional)

  const list = readObjects_(sh).filter(x => {
    const xs = String(x.sku || "").trim();
    if (sku && xs && xs === sku) return true;
    return pid ? (String(x.product_id || x.id || "").trim() === pid) : false;
  });

  // 日期篩選（用 date，若無則不篩）
  let rows = list;
  if (from) {
    const ft = Date.parse(from + "T00:00:00");
    rows = rows.filter(x => {
      const t = Date.parse(String(x.date || "").replace(" ", "T"));
      return isNaN(t) ? true : (t >= ft);
    });
  }
  if (to) {
    const tt = Date.parse(to + "T23:59:59");
    rows = rows.filter(x => {
      const t = Date.parse(String(x.date || "").replace(" ", "T"));
      return isNaN(t) ? true : (t <= tt);
    });
  }

  // 依時間倒序（date）
  rows.sort((a,b) => {
    const ta = Date.parse(String(a.date || "").replace(" ", "T"));
    const tb = Date.parse(String(b.date || "").replace(" ", "T"));
    return (isNaN(tb)?0:tb) - (isNaN(ta)?0:ta);
  });

  return respond({ status: "ok", data: rows.slice(0, limit) });
}


function apiStockAdjust_(p, respond) {
  const product_id = String(p.product_id || "").trim();
  const delta = Number(p.delta || 0);
  const reason = String(p.reason || "adjust").trim();
  const operator = String(p.operator || "").trim();
  if (!product_id) return respond({ status: "error", message: "缺少 product_id" });
  if (!delta || isNaN(delta)) return respond({ status: "error", message: "delta 必須是非 0 數字" });

  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const prodSh = ensureSheet_(ss, "Products", ["id","name","sku","price","image","stock","category","date","cost","unit","safety_stock","supplier_ids","expiry_date","last_purchase_date","spec","reference_price","reference_price_date","reference_price_name","shop_enabled"]);

  const lock = LockService.getScriptLock();
  lock.waitLock(10000);
  try {
    const vals = prodSh.getDataRange().getValues();
    const headers = vals[0].map(h => String(h || "").trim());
    const idx = headerIndexMap_(headers);

    let row = -1;
    for (let r = 1; r < vals.length; r++) {
      if (String(vals[r][idx.id]).trim() === product_id) { row = r; break; }
    }
    if (row === -1) return respond({ status: "error", message: "找不到商品" });

    const current = Number(vals[row][idx.stock] || 0);
    const next = current + delta;
    if (next < 0) return respond({ status: "error", message: `庫存不足 current=${current}, delta=${delta}` });

    prodSh.getRange(row + 1, idx.stock + 1).setValue(next);

    const pname = String(vals[row][idx.name] || "");
    const now = new Date();
    writeLedger_(ss, {
      ledger_id: generateId_("LG"),
      product_id: product_id,
      product_name: pname,
            sku: (idx.sku !== undefined) ? String(vals[row][idx.sku] || "") : "",
      unit: (idx.unit !== undefined) ? String(vals[row][idx.unit] || "") : "",
change: delta,                 // signed change：增加為正、減少為負
      direction: "ADJ",
      reason: "adjust",
      ref_id: "ADJ-" + Utilities.formatDate(now, "Asia/Taipei", "yyyyMMddHHmmss"),
      operator: operator,
      date: toLocalDateTime_(now),
      cost: (idx.cost !== undefined) ? (Number(vals[row][idx.cost] || 0) || "") : "",
      target: reason                 // 調整原因放在 target（便於前台顯示）
    });

    return respond({ status: "ok", message: "庫存已調整", before: current, after: next });
  } finally {
    try { lock.releaseLock(); } catch (e2) {}
  }
}

function systemSettingsHeaders_(){
  return ["key","value","updated_at"];
}

function getSystemSettingsMap_(ss){
  const sh = ensureSheet_(ss, "SystemSettings", systemSettingsHeaders_());
  const rows = sh.getDataRange().getValues();
  const map = {};
  for (let r = 1; r < rows.length; r++) {
    const key = String(rows[r][0] || "").trim();
    if (!key) continue;
    map[key] = String(rows[r][1] || "");
  }
  return map;
}

function upsertSystemSetting_(sh, key, value){
  const rows = sh.getDataRange().getValues();
  for (let r = 1; r < rows.length; r++) {
    if (String(rows[r][0] || "").trim() === key) {
      sh.getRange(r + 1, 2).setValue(value);
      sh.getRange(r + 1, 3).setValue(new Date());
      return;
    }
  }
  appendObject_(sh, { key: key, value: value, updated_at: new Date() });
}

function apiSystemSettings_(respond){

    }
  });
}

  upsertSystemSetting_(sh, "sales_name", String(payload.sales_name || p.sales_name || "").trim());
  return respond({ status: "ok", message: "系統設定已儲存" });
}

function apiLastUpdate_(respond) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sh = ensureSheet_(ss, "Products", ["id","name","sku","price","image","stock","category","date","cost","unit","safety_stock","supplier_ids","expiry_date","last_purchase_date","spec","reference_price","reference_price_date","reference_price_name","shop_enabled"]);
  const rows = sh.getDataRange().getValues();
  const headers = rows[0].map(h => String(h || "").trim());
  const dateIndex = headers.indexOf("date");
  if (dateIndex === -1) return respond({ status: "error", message: "找不到 date 欄位" });

  let latest = "";
  for (let r = 1; r < rows.length; r++) {
    const d = rows[r][dateIndex];
    if (!d) continue;
    const ds = normalizeDate_(d);
    if (!latest || compareDate_(ds, latest) > 0) latest = ds;
  }
  return respond({ status: "ok", lastUpdate: latest });
}

function apiMemberRegister_(p, respond) {
  const name = String(p.name || "").trim();
  const username = String(p.username || "").trim();
  const password = String(p.password || "").trim();

  if (!name || !username || !password) {
    return respond({ status: "error", message: "缺少註冊資料" });
  }

  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sh = ensureSheet_(ss, "Members", ["id","name","username","password","created_at","role"]);
  const rows = sh.getDataRange().getValues();
  const headers = rows[0].map(h => String(h || "").trim());
  const idx = headerIndexMap_(headers);

  // 帳號重複檢查
  for (let i = 1; i < rows.length; i++) {
    const u = String(rows[i][idx.username] || "").trim();
    if (u === username) {
      return respond({ status: "error", message: "帳號已存在" });
    }
  }

  // 產生新 id（取目前最大值 + 1）
  let newId = 1;
  if (rows.length > 1 && idx.id !== undefined) {
    let maxId = 0;
    for (let i = 1; i < rows.length; i++) {
      const v = Number(String(rows[i][idx.id] || "").trim());
      if (!isNaN(v)) maxId = Math.max(maxId, v);
    }
    newId = maxId + 1;
  }

  appendObject_(sh, {
    id: newId,
    name: name,
    username: username,
    password: password,
    created_at: new Date(),
    role: "user"
  });

  return respond({ status: "ok", message: "註冊成功", id: newId });
}

function apiOrderCreate_(p, respond) {
  const name = String(p.name || "").trim();
  const phone = String(p.phone || "").trim();
  const address = String(p.address || "").trim();
  const member_id = String(p.member_id || "").trim();

  if (!name || !phone || !address) {
    return respond({ status: "error", message: "缺少收件資訊" });
  }

  let cart = [];
  try {
    const raw = String(p.cart || "[]");
    const decoded = decodeURIComponent(raw);
    cart = JSON.parse(decoded);
  } catch (e) {
    try { cart = JSON.parse(String(p.cart || "[]")); } catch (e2) {}
  }
  if (!Array.isArray(cart) || cart.length === 0) {
    return respond({ status: "error", message: "購物車為空" });
  }

  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const ordersSh = ensureSheet_(ss, "Orders", orderHeaders_());
  const prodSh = ensureSheet_(ss, "Products", ["id","name","sku","price","image","stock","category","date","cost","unit","safety_stock","supplier_ids","expiry_date","last_purchase_date","spec","reference_price","reference_price_date","reference_price_name","shop_enabled"]);
  const purchasesSh = ensurePurchasesSheet_(ss);

  const lock = LockService.getScriptLock();
  lock.waitLock(10000);

  try {
    const now = new Date();
    const orderDate = toISODateLocal_(now);
    const order_id = String(generateDailyDocNo_(ordersSh, "order_id", now)).trim();
    const po_id = String(generateDailyDocNo_(purchasesSh, "po_id", now)).trim();

    const prodVals = prodSh.getDataRange().getValues();
    const ph = prodVals[0].map(h => String(h || "").trim());
    const pidx = headerIndexMap_(ph);
    const byId = {};
    for (let r = 1; r < prodVals.length; r++) {
      byId[String(prodVals[r][pidx.id]).trim()] = { r: r, row: prodVals[r] };
    }
    const supplierMap = buildSupplierNameMap_(ss);

    const items = [];
    const purchaseItems = [];
    let total = 0;
    let firstCategory = "";
    let purchaseTotal = 0;

    for (let i = 0; i < cart.length; i++) {
      const it = cart[i] || {};
      const pid = String(it.id || it.product_id || "").trim();
      const qty = cleanDecimalInput_(it.qty || it.quantity || 0, 6);
      const price = cleanDecimalInput_(it.price || 0, 6);
      if (!pid || !(qty > 0)) return respond({ status: "error", message: "購物車品項資料不完整" });
      const hit = byId[pid];
      if (!hit) return respond({ status: "error", message: `找不到商品 id=${pid}` });
      const row = hit.row;
      const pname = String(it.name || row[pidx.name] || "");
      const category = String(row[pidx.category] || "").trim();
      if (!firstCategory && category) firstCategory = category;
      const stock = cleanDecimalInput_(row[pidx.stock] || 0, 6);
      // 電商下單不在此階段檢查或扣減庫存；一律允許下訂，後續由採購驗收與出貨流程處理。
      const safety = cleanDecimalInput_(row[pidx.safety_stock] || 0, 6);
      const cost = cleanDecimalInput_(row[pidx.cost] || 0, 6);
      const suggestedQty = addDecimalInput_(qty, Math.max(0, safety - stock), 6);
      const supplierIds = parseSupplierIdsFromProductRow_(row, pidx);
      const supplier_id = String(supplierIds[0] || "").trim();
      const supplier_name = String(supplierMap[supplier_id] || "").trim();
      const sub = mulDecimalInput_(qty, price, 6);
      total = addDecimalInput_(total, sub, 6);
      // 電商自動生成的採購驗收單草稿，不預帶單價；由後續人工驗收時填寫。
      purchaseTotal = addDecimalInput_(purchaseTotal, 0, 6);

      items.push({
        product_id: pid,
        name: pname,
        qty: qty,
        price: price,
        subtotal: sub,
        spec: (pidx.spec !== undefined) ? String(row[pidx.spec] || "") : "",
        unit: (pidx.unit !== undefined) ? String(row[pidx.unit] || "") : "",
        sku: (pidx.sku !== undefined) ? String(row[pidx.sku] || "") : ""
      });

      purchaseItems.push({
        product_id: pid,
        product_name: pname,
        qty: qty,
        qty_raw: qty,
        suggested_qty: suggestedQty,
        customer_order_qty: qty,
        cost: "",
        cost_raw: "",
        supplier_id: supplier_id,
        supplier_name: supplier_name,
        spec: (pidx.spec !== undefined) ? String(row[pidx.spec] || "") : "",
        unit: (pidx.unit !== undefined) ? String(row[pidx.unit] || "") : "",
        sku: (pidx.sku !== undefined) ? String(row[pidx.sku] || "") : "",
        receive_date: "",
        inspection_priority: "",
        receipt_weight: "",
        accept_weight: "",
        acceptance_result: "",
        pesticide_result: "",
        note: ""
      });
    }

    const formInfo = getPurchaseFormInfoByCategory_(firstCategory);

    const orderItemsSh = getSheetByAnyName_(ss, ["orderItems"]);
    if (orderItemsSh) {
      ensureSheet_(ss, orderItemsSh.getName(), ["OrderItemID","OrderID","ProductID","Quantity","UnitPrice","Subtotal"]);
      for (let i = 0; i < items.length; i++) {
        const it = items[i];
        appendObject_(orderItemsSh, {
          OrderItemID: generateId_("OI"),
          OrderID: order_id,
          ProductID: it.product_id,
          Quantity: Number(it.qty),
          UnitPrice: Number(it.price),
          Subtotal: Number(it.subtotal)
        });
      }
    }

    appendObject_(ordersSh, {
      order_id: order_id,
      name: name,
      phone: phone,
      address: address,
      total: total,
      items: JSON.stringify(items),
      date: orderDate,
      member_id: member_id,
      status: "待出貨",
      operator: "",
      source: "EC",
      stock_applied: 0,
      linked_purchase_id: po_id
    });

    appendObject_(purchasesSh, {
      po_id: po_id,
      date: orderDate,
      arrival_date: "",
      form_no: formInfo.form_no,
      form_name: formInfo.form_name,
      supplier_id: (function(){ const ids = Array.from(new Set(purchaseItems.map(x => String(x.supplier_id || "").trim()).filter(Boolean))); return ids.length === 1 ? ids[0] : "MULTI"; })(),
      supplier_name: (function(){ const names = Array.from(new Set(purchaseItems.map(x => String(x.supplier_name || "").trim()).filter(Boolean))); return names.length === 1 ? names[0] : "多供應商"; })(),
      total: purchaseTotal,
      items: JSON.stringify(purchaseItems),
      expiry_date: "",
      status: "待驗收",
      operator: "",
      created_at: now,
      source_order_id: order_id,
      stock_applied: 0,
      completed_at: "",
      auto_generated: 1
    });

    return respond({ status: "ok", message: "下單成功", order_id: order_id, po_id: po_id, total: total });
  } finally {
    try { lock.releaseLock(); } catch (e2) {}
  }
}

function apiMyOrders_(p, respond) {
  const member_id = String(p.member_id || "").trim();
  if (!member_id) return respond({ status: "error", message: "缺少 member_id" });

  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sh = ensureSheet_(ss, "Orders", orderHeaders_());

  const list = readObjects_(sh)
    .filter(o => String(o.member_id || "").trim() === member_id)
    .map(o => {
      o.date = toISODateLocal_(o.date);
      // items 不要在列表展開，避免太長
      return o;
    });

  // 新到舊（以 order_id 文字排序足夠：yyyyMMddXXXX）
  list.sort((a, b) => String(b.order_id || "").localeCompare(String(a.order_id || "")));

  return respond({ status: "ok", orders: list });
}

function apiGetOrder_(p, respond) {
  const order_id = String(p.order_id || p.id || "").trim();
  if (!order_id) return respond({ status: "error", message: "缺少 order_id" });

  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sh = ensureSheet_(ss, "Orders", orderHeaders_());

  const list = readObjects_(sh);
  const hit = list.find(o => String(o.order_id || "").trim() === order_id);
  if (!hit) return respond({ status: "error", message: "查無此訂單" });

  let items = hit.items;
  if (items && typeof items === "string") {
    try { items = JSON.parse(items); } catch (e) {}
  }
  hit.items = formatOrderItemsForDisplay_(items);
  hit.date = toISODateLocal_(hit.date);

  return respond({ status: "ok", order: hit });
}

function formatOrderItemsForDisplay_(items) {
  if (!Array.isArray(items) || items.length === 0) return "";
  return items.map(it => {
    const name = String(it.name || it.product_name || it.product || "");
    const qty = Number(it.qty || it.quantity || it.Quantity || 0);
    const price = Number(it.price || it.UnitPrice || 0);
    const subtotal = (!isNaN(Number(it.subtotal))) ? Number(it.subtotal) : (qty * price);
    return `${name} x${qty} = $${subtotal}`;
  }).join("<br>");
}
